1"""Sterba curtain, transmission-line sister design of `sterba.py`. 

2 

3Where `wire.sterba` models the curtain as a single continuous wire 

4(Cebik's offset-pair trick, verticals as real wires), this variant models 

5the vertical phasing sections as **ideal transmission lines** via 

6`build_network()`. 

7 

8Simplified single-conductor form: a flat meander at x=0 made of 

9`n_cells + 2` horizontal sections (a quarter-wave at each end, `n_cells` 

10half-waves in the middle) at alternating heights — bottom rail (z = base) 

11and top rail (z = base + h), h = half a wavelength. Consecutive sections 

12are NOT joined by wire; each junction is bridged by a half-wave TL that 

13plays the role of the vertical phasing line. With n_cells = 3 that's five 

14sections joined by four TLs. The feed is a delta-gap at the centre of the 

15central (lower) section. 

16 

17Differences from the all-wires `sterba`: 

18 - One conductor, not the offset pair, so the curtain is one-high (a 

19 staircase of sections) rather than two-high stacked. Expect somewhat 

20 lower gain than the all-wires version; this is a modelling study of 

21 the TL abstraction, not a higher-performance antenna. 

22 - The verticals carry no geometry: their current/phase lives entirely 

23 in the TL stamp, so there is nothing to radiate from them by 

24 construction (the all-wires version achieves the same end by current 

25 cancellation in the offset pair). 

26 

27Note on the half-wave TL: this engine's nodal TL admittance 

281/(jZ0 sin βl)·[[cos βl,-1],[-1,cos βl]] is SINGULAR at βl = π (length = 

29λ/2). The phasing lines are nominally λ/2, so `length_factor` defaults to 

300.99 (TL length ≈ 0.495 λ) to sit just off the singularity. lf = 1.0 

31exactly will raise in `network_reduce.tl_admittance_2x2`; keep lf away from 1.0. 

32""" 

33 

34from ... import AntennaBuilder 

35from ...network import Driven, Network, PortAtEdge, TL 

36from types import MappingProxyType

39class Builder(AntennaBuilder): 

40 default_params = MappingProxyType( 

41 { 

42 "design_freq": 28.47, 

43 "freq": 28.47, 

44 "base": 7.0, 

45 # Off 1.0 on purpose: at lf = 1.0 the half-wave phasing TLs hit 

46 # the βl = π nodal-admittance singularity. 0.99 sits just off it. 

47 "length_factor": 0.99, 

48 # Odd; number of full half-wave middle sections. TLs = n_cells + 1. 

49 "n_cells": 3, 

50 # Characteristic impedance of the phasing lines (the twisted pair 

51 # the TL stands in for); ~500 ohm for a close-spaced 2-wire line. 

52 "z0": 500.0, 

53 "ui_params": MappingProxyType( 

54 { 

55 # Unlike the all-wires (~600 ohm) Sterba, the single- 

56 # conductor TL-fed form presents ~70 ohm at the central 

57 # delta-gap (near resonance at lf ~ 0.997), so reference 

58 # SWR to 75 ohm. 

59 "target_z0": 75.0, 

60 # Single-band 10m antenna: snap the GUI sweep to the band. 

61 "sweep_policy": {"band_locked": True}, 

62 # Keep length_factor near (but not at) 1.0; below ~0.96 

63 # the phasing degrades, and 1.0 is the TL singularity. 

64 "length_factor": { 

65 "min": 0.96, 

66 "max": 0.999, 

67 }, 

68 "n_cells": { 

69 "min": 1, 

70 "max": 7, 

71 "step": 2, 

72 }, 

73 } 

74 ), 

75 }

78 def build_wires(self): 

79 wavelength = 299.792458 / self.design_freq 

80 h = 0.5 * wavelength * self.length_factor # half-wave unit 

81 q = 0.5 * h # quarter-wave end section 

82 bot = self.base 

83 top = self.base + h 

84 n = int(self.n_cells) 

85 assert n >= 1 and n % 2 == 1, "n_cells must be a positive odd integer" 

86 

87 n0 = self.nominal_nsegs 

88 pe = max(0.2, 0.06 * h) # short port/feed edge length (m) 

89 

90 # y breakpoints and per-section heights (alternate, start on bottom 

91 # so both ends and the centre sit on the lower rail). 

92 yb = [0.0] + [q + k * h for k in range(n + 1)] + [2 * q + n * h] 

93 n_sections = n + 2 

94 center_sec = (n_sections - 1) // 2 

95 

96 def level(i): 

97 return bot if i % 2 == 0 else top 

98 

99 self._tl_length = h # consumed by build_network() 

100 

101 tups = [] 

102 

103 def edge(ya, yb_, z, name): 

104 seg_len = abs(yb_ - ya) 

105 ns = 3 if name is not None else max(5, round(n0 * seg_len / h)) 

106 if name is not None and ns % 2 == 0: 

107 ns += 1 

108 tups.append(((0.0, ya, z), (0.0, yb_, z), ns, None, name)) 

109 

110 for i in range(n_sections): 

111 y0, y1, z = yb[i], yb[i + 1], level(i) 

112 left = f"p{i}_b" if i > 0 else None # joins TL from junction i 

113 right = f"p{i + 1}_a" if i < n_sections - 1 else None 

114 feed = "feed" if i == center_sec else None 

115 

116 cur = y0 

117 if left is not None: 

118 edge(cur, cur + pe, z, left) 

119 cur += pe 

120 body_end = (y1 - pe) if right is not None else y1 

121 if feed is not None: 

122 yc = 0.5 * (y0 + y1) 

123 edge(cur, yc - 0.5 * pe, z, None) 

124 edge(yc - 0.5 * pe, yc + 0.5 * pe, z, "feed") 

125 cur = yc + 0.5 * pe 

126 edge(cur, body_end, z, None) 

127 cur = body_end 

128 if right is not None: 

129 edge(cur, y1, z, right) 

130 

131 return tups 

132 

133 def build_network(self): 

134 # build_wires must run first to populate _tl_length. 

135 self.build_wires() 

136 n = int(self.n_cells) 

137 n_sections = n + 2 

138 z0 = self.z0 

139 length = self._tl_length 

140 

141 ports = {"feed": PortAtEdge("feed")} 

142 branches = [] 

143 # One TL per junction j (1..n_sections-1): connects the right-end 

144 # port of section j-1 to the left-end port of section j — the 

145 # half-wave vertical phasing line, now ideal. 

146 for j in range(1, n_sections): 

147 a, b = f"p{j}_a", f"p{j}_b" 

148 ports[a] = PortAtEdge(a) 

149 ports[b] = PortAtEdge(b) 

150 branches.append(TL(a=a, b=b, z0=z0, length=length)) 

151 

152 return Network( 

153 ports=ports, 

154 branches=branches, 

155 sources=[Driven(port="feed", voltage=1 + 0j)], 

156 )
