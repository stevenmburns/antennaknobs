21def _parity_for_solver(solver, solver_kwargs): 

22 """The basis types have fixed parity expectations: 

23 - BSplineSolver degree=1 → tent basis, even (feed straddles 2 segs) 

24 - BSplineSolver degree=2 → quadratic → odd 

25 - SinusoidalSolver → odd 

26 Anything else falls through as "any" (no coercion).""" 

27 name = getattr(solver, "__name__", "") 

28 if name == "SinusoidalSolver": 

29 return "odd" 

30 if name in ("BSplineSolver", "HMatrixSolver", "ArrayBlockSolver"): 

31 # HMatrixSolver and ArrayBlockSolver are BSplineSolver subclasses (same 

32 # basis), so they share the degree-driven parity. Getting this right 

33 # matters for cross-solver comparison: a mismatched parity would build 

34 # a *different* mesh and silently invalidate any A/B against the dense 

35 # bspline path. 

36 degree = (solver_kwargs or {}).get("degree", 2) 

37 return "even" if int(degree) == 1 else "odd" 

38 return "any"

46def _polyline_knots(polyline, npe_list): 

47 """Concatenated per-edge knot positions (shared corners deduped). 

48 Mirrors momwire/web/server.py:_polyline_knots.""" 

49 parts = [] 

50 for i, n_e in enumerate(npe_list): 

51 seg = np.linspace(polyline[i], polyline[i + 1], n_e + 1) 

52 parts.append(seg if i == 0 else seg[1:]) 

53 return np.vstack(parts)

56def _normalise_ground(ground): 

57 if ground is None or ground == "free": 

58 return None 

59 if ground == "pec": 

60 return ("pec",) 

61 if ( 

62 isinstance(ground, tuple) 

63 and len(ground) == 3 

64 and ground[0] 

65 in ( 

66 "finite", 

67 "finite-fast", 

68 ) 

69 ): 

70 # The variant is preserved: "finite" means the true Sommerfeld/ 

71 # Norton ground (NEC gn 2 — momwire BSplineSolver implements it 

72 # since 0.6.0), "finite-fast" the reflection-coefficient 

73 # approximation (NEC gn 0). Solvers without the requested model 

74 # fall back to their best available one — see the ground-model 

75 # mapping in __init__. 

76 return (ground[0],) + tuple(ground[1:]) 

77 raise ValueError(f"unrecognised ground spec: {ground!r}")

119 def __init__( 

120 self, 

121 builder, 

122 *, 

123 solver=BSplineSolver, 

124 wire_radius=0.0005, 

125 solver_kwargs=None, 

126 ground=None, 

127 ground_z=0.0, 

128 cancel=None, 

129 ): 

130 """ 

131 solver: 

132 A momwire solver class — BSplineSolver (default), SinusoidalSolver, 

133 or the fast BSpline subclasses (HMatrixSolver, ArrayBlockSolver). 

134 Different bases trade speed vs impedance fidelity; on the hentenna 

135 sinusoidal is typically closer to PyNEC at modest segmentation. 

136 solver_kwargs: 

137 Dict of solver-specific kwargs passed straight to the constructor 

138 (e.g. `{"degree": 1}` for BSplineSolver, or `{"n_qp_const": 16}` 

139 for SinusoidalSolver). None = solver defaults. 

140 ground: 

141 None or "free" — no ground (default) 

142 "pec" — PEC plane at z=ground_z (image method) 

143 ("finite", eps_r, sigma) — far-field uses PEC image + Fresnel 

144 coefficients on the reflected component. 

145 The impedance solve uses momwire's TRUE 

146 Sommerfeld ground (NEC gn 2 style, 

147 ground_model="sommerfeld") on every 

148 momwire solver (momwire >= 0.8.0: 

149 BSpline dense, Sinusoidal field-based, 

150 HMatrix/ArrayBlock fast paths) — the 

151 accurate-at-any-height model. 

152 ("finite-fast", eps_r, sigma) — the reflection-coefficient model 

153 (NEC gn 0 style, ground_eps) on every 

154 solver that supports it. Matches 

155 "finite" to ~2 ohm above ~0.1λ heights 

156 but diverges hard below (~22 ohm at 

157 0.05λ, >100 ohm at 0.02λ). 

158 """ 

159 super().__init__(builder) 

160 

161 # Cooperative-cancellation token (momwire.CancelToken or None). Forwarded 

162 # into every solver construction so a mid-solve cancel aborts the C++ 

163 # fill / Python checkpoints, and polled between this engine's own phases. 

164 self._cancel = cancel 

165 self._solver = solver 

166 self._solver_kwargs = dict(solver_kwargs) if solver_kwargs else {} 

167 # Per-instance parity: sinusoidal wants odd, bspline depends on 

168 # degree. Set before _coerce_wire_tuples runs. 

169 self.segment_parity = _parity_for_solver(self._solver, self._solver_kwargs) 

170 

171 # build_wires() must run before build_tls() — some designs populate 

172 # self.tls inside build_wires() (delta_looparray_with_tls). 

173 tups = self._coerce_wire_tuples(builder.build_wires()) 

174 self._network = builder.build_network() 

175 self._tls = [] if self._network is not None else list(builder.build_tls()) 

176 

177 # Resolve TL endpoint tags into augmented tups: any tag whose ev was 

178 # nullified gets a passive feed (V=0) so momwire assembles the full 

179 # multi-port Y matrix. Driven ports keep their original voltages. 

180 augmented_tags = set() 

181 if self._tls: 

182 tups = list(tups) 

183 for tag1, _seg1, tag2, _seg2, _z0, _length in self._tls: 

184 for tag in (tag1, tag2): 

185 t = tups[tag - 1] 

186 p0, p1, n_seg, ev = t[0], t[1], t[2], t[3] 

187 if ev is None: 

188 tups[tag - 1] = (p0, p1, n_seg, 0 + 0j) 

189 augmented_tags.add(tag) 

190 

191 translated = flat_wires_to_polylines(tups) 

192 self._polylines = translated["polylines"] 

193 self._edge_segments = translated["edge_segments"] 

194 self._feeds = translated["feeds"] 

195 self._feed_names = translated["feed_names"] 

196 self._junctions = translated["junctions"] 

197 # Wire material (issue #316): a design-declared WireSpec supplies the 

198 # conductor radius plus the distributed-loading kwargs. Precedence 

199 # for the radius: an explicit non-default `wire_radius` (the web 

200 # model-options control) wins; the stock 0.0005 acts as "auto" so a 

201 # spec design's skin-effect loss is computed at its real radius. 

202 self._wire_spec = builder.build_wire_material() 

203 if wire_radius != 0.0005 and wire_radius is not None: 

204 self._wire_radius = wire_radius 

205 elif self._wire_spec is not None: 

206 self._wire_radius = self._wire_spec.radius 

207 else: 

208 self._wire_radius = 0.0005 

209 # Distributed loading rides only the solvers that model it; warn 

210 # once (not raise) so a matched-basis sinusoidal comparison of a 

211 # lossy design still solves — as the ideal wire, stated plainly. 

212 self._loading_kwargs = {} 

213 spec = self._wire_spec 

214 if spec is not None and ( 

215 spec.conductivity is not None or spec.insulation_radius is not None 

216 ): 

217 if _solver_supports_wire_loading(self._solver): 

218 if spec.conductivity is not None: 

219 self._loading_kwargs["wire_conductivity"] = spec.conductivity 

220 if spec.insulation_radius is not None: 

221 self._loading_kwargs["insulation_radius"] = spec.insulation_radius 

222 self._loading_kwargs["insulation_eps_r"] = spec.insulation_eps_r 

223 else: 

224 _logger.warning( 

225 "%s doesn't model distributed wire loading; solving the " 

226 "design's %s wire as ideal (PEC, bare)", 

227 getattr(self._solver, "__name__", self._solver), 

228 type(builder).__name__, 

229 ) 

230 self._ground = _normalise_ground(ground) 

231 self._ground_z = ground_z if self._ground is not None else None 

232 # Finite ground constants forwarded to the impedance solve when the 

233 # solver supports the reflection-coefficient model; None otherwise 

234 # (PEC image). 

235 self._ground_eps = None 

236 self._ground_model = None 

237 if ( 

238 self._ground is not None 

239 and self._ground[0] in ("finite", "finite-fast") 

240 and _solver_supports_ground_eps(self._solver) 

241 ): 

242 self._ground_eps = (float(self._ground[1]), float(self._ground[2])) 

243 # "finite" means true Sommerfeld on EVERY momwire solver since 

244 # momwire 0.8.0 (sinusoidal grew the model; HMatrix/ArrayBlock 

245 # solve it on their fast paths — C2-scaled image blocks + one 

246 # global low-rank remainder — so the old dense-solve cliff that 

247 # kept them on refl-coef is gone). "finite-fast" is refl-coef 

248 # everywhere. 

249 self._ground_model = ( 

250 "sommerfeld" if self._ground[0] == "finite" else "refl-coef" 

251 ) 

252 

253 # Map TL tags to feed indices for the legacy build_tls() path. 

254 self._tag_to_feed = {} 

255 feed_i = 0 

256 for tag, t in enumerate(tups, start=1): 

257 ev = t[3] 

258 if ev is not None: 

259 self._tag_to_feed[tag] = feed_i 

260 feed_i += 1 

261 # 0-based feed indices that are TL passive ports (V=0, floating). 

262 self._tl_passive_feed_idx = {self._tag_to_feed[t] for t in augmented_tags} 

263 

264 if self._network is not None: 

265 self._init_network()

267 def _init_network(self): 

268 """Build the port-index map (real feeds first, virtual ports after) 

269 and the engine-agnostic NetworkReducer that stamps the branches and 

270 reduces to driven-port impedance. Validates that every PortOnWire 

271 resolves to a translated feed name.""" 

272 net = self._network 

273 feed_name_to_idx = {n: i for i, n in enumerate(self._feed_names) if n} 

274 

275 port_to_idx = {} 

276 for name, port in net.ports.items(): 

277 if isinstance(port, PortOnWire): 

278 if name not in feed_name_to_idx: 

279 raise ValueError( 

280 f"network port {name!r} is a PortOnWire but no wire in " 

281 f"build_wires() carries that name; named wires: " 

282 f"{sorted(feed_name_to_idx)}" 

283 ) 

284 port_to_idx[name] = feed_name_to_idx[name] 

285 

286 # Virtual ports are indexed after the real feeds. 

287 next_idx = len(self._feeds) 

288 for name, port in net.ports.items(): 

289 if isinstance(port, PortVirtual): 

290 port_to_idx[name] = next_idx 

291 next_idx += 1 

292 

293 self._reducer = NetworkReducer(net, port_to_idx, next_idx)

296 """Extra ground kwargs for solver construction. Only spliced in when 

297 set (free-space / PEC solves pass no ground_eps).""" 

298 if self._ground_eps is None: 

299 return {}

327 def _apply_tls(self, Y, wavelength): 

328 """Y + per-TL stamps at the corresponding feed-index pairs (legacy 

329 build_tls() path; the network-spec path goes through NetworkReducer).""" 

330 Y = Y.copy() 

331 for tag1, _s1, tag2, _s2, z0, length in self._tls: 

332 a = self._tag_to_feed[tag1] 

333 b = self._tag_to_feed[tag2] 

334 y_tl = tl_admittance_2x2(z0, length, wavelength) 

335 Y[np.ix_([a, b], [a, b])] += y_tl 

336 return Y

338 def _resolve_feed_voltages(self, Y_total): 

339 """Return the full per-feed voltage vector V with passive ports' 

340 voltages set so I_ext=0 there. The driven ports keep their applied V.""" 

341 n = Y_total.shape[0] 

342 driven = [i for i in range(n) if i not in self._tl_passive_feed_idx] 

343 passive = sorted(self._tl_passive_feed_idx) 

344 v_driven = np.array([self._feeds[i][2] for i in driven], dtype=np.complex128) 

345 V = np.empty(n, dtype=np.complex128) 

346 V[driven] = v_driven 

347 if passive: 

348 Y_pp = Y_total[np.ix_(passive, passive)] 

349 Y_pd = Y_total[np.ix_(passive, driven)] 

350 V[passive] = np.linalg.solve(Y_pp, -Y_pd @ v_driven) 

351 return V, driven

353 def _impedance_from_y(self, Y_total): 

354 """Driving-point Z at each driven port, with passive (TL-only) ports 

355 floating (I_ext=0). Matches PyNECEngine's per-driven-port semantics 

356 when all drivers are excited simultaneously.""" 

357 V, driven = self._resolve_feed_voltages(Y_total) 

358 I = Y_total @ V 

359 return [complex(V[i] / I[i]) for i in driven]

371 def _solved_excited(self, wavelength): 

372 """Build the excitation-resolved solver and run compute_impedance 

373 once per (wavelength, feed-voltage) tuple, caching on the engine 

374 instance. Lets impedance(), current_distribution(), and far_field() 

375 share one MoM solve when the live UI tick calls them in sequence. 

376 

377 Cache lives for this engine instance only; the server constructs a 

378 fresh MomwireEngine each tick, so nothing leaks across requests. 

379 """ 

380 sim = self._make_excited_solver(wavelength=wavelength) 

381 v_key = tuple((complex(v).real, complex(v).imag) for *_, v in sim.feeds) 

382 key = (float(wavelength), v_key) 

383 cached = getattr(self, "_solved_cache", None) 

384 if cached is not None and cached[0] == key: 

385 sim, coeffs, z = cached[1] 

386 else: 

387 z, coeffs = sim.compute_impedance() 

388 self._solved_cache = (key, (sim, coeffs, z)) 

389 # _make_excited_solver reset the power bookkeeping either way, so 

390 # the wire-loss amendment below is applied exactly once per call. 

391 self._amend_wire_loss(sim, coeffs, z) 

392 return sim, coeffs, z

394 def _amend_wire_loss(self, sim, coeffs, z): 

395 """Fold ohmic wire loss (momwire#131 distributed loading) into the 

396 power bookkeeping (issue #317): a "wire loss (I²R)" budget row and 

397 the matching efficiency drop. The loss already lives inside P_in 

398 (the loading raised the driving-point R), so gain = 4π·U/P_in needs 

399 no change — this keeps the *reported* efficiency and budget rows 

400 truthful about where those watts went. Insulation is reactive and 

401 contributes nothing; lossless designs are untouched.""" 

402 self._excited_p_wire = 0.0 

403 if not self._loading_kwargs: 

404 return 

405 p_wire, _per_wire = sim.wire_loss_power(coeffs) 

406 if p_wire <= 0.0: 

407 return 

408 self._excited_p_wire = float(p_wire) 

409 self._excited_power_budget = list(self._excited_power_budget) + [ 

410 ("wire loss (I²R)", float(p_wire)) 

411 ] 

412 p_in = self._excited_p_in 

413 if p_in is None: # plain path records no port-model power 

414 p_in = self._p_in_from_excited(sim, z) 

415 if p_in > 0.0: 

416 self._excited_efficiency = max( 

417 0.0, min(1.0, self._excited_efficiency - p_wire / p_in) 

418 )

420 @staticmethod 

421 def _p_in_from_excited(sim, z): 

422 """Source input power ½·Σ|V_f|²·Re(Z_f)/|Z_f|² from an excited 

423 solve's driving-point impedances (the plain-path formula shared by 

424 `input_power` and the wire-loss amendment).""" 

425 z_arr = np.atleast_1d(np.asarray(z, dtype=np.complex128)) 

426 volts = np.array([complex(v) for *_, v in sim.feeds], dtype=np.complex128) 

427 with np.errstate(divide="ignore", invalid="ignore"): 

428 terms = 0.5 * np.abs(volts) ** 2 * z_arr.real / np.abs(z_arr) ** 2 

429 return float(np.sum(terms[np.isfinite(terms)]))

441 def impedance(self): 

442 self._raise_if_cancelled() 

443 wavelength = self._wavelength_for(self.builder.freq) 

444 if self._network is not None: 

445 Y = self._compute_y_matrix(wavelength) 

446 return self._reducer.driven_impedance(Y, wavelength) 

447 if self._tls: 

448 Y = self._compute_y_matrix(wavelength) 

449 Y_total = self._apply_tls(Y, wavelength) 

450 return self._impedance_from_y(Y_total) 

451 _sim, _coeffs, z = self._solved_excited(wavelength) 

452 # Single-feed path returns a scalar; multi-feed returns an array. 

453 # Match PyNECEngine's list-of-Z return shape. 

454 z_arr = np.atleast_1d(z) 

455 return [complex(zi) for zi in z_arr]

457 def impedance_sweep(self, freqs): 

458 freqs = np.asarray(freqs, dtype=float) 

459 if freqs.ndim != 1 or freqs.size == 0: 

460 raise ValueError("freqs must be a 1-D non-empty array") 

461 s = self._make_solver(wavelength=self._wavelength_for(freqs[0])) 

462 k_array = 2.0 * np.pi * freqs * 1e6 / C_LIGHT 

463 if self._network is not None: 

464 Y_swept = np.asarray( 

465 s.compute_y_matrix_swept(k_array), dtype=np.complex128 

466 ) # (n_k, n_real, n_real) 

467 zs = np.empty((freqs.size, self._reducer.n_driven), dtype=np.complex128) 

468 for ki, freq in enumerate(freqs): 

469 Y_total = self._reducer.apply_branches( 

470 Y_swept[ki], self._wavelength_for(freq) 

471 ) 

472 zs[ki] = self._reducer.impedance_from_y(Y_total) 

473 return zs 

474 if self._tls: 

475 # Batched Y at every frequency, then per-k TL stamping (βL is 

476 # frequency-dependent) and driven-port reduction. The Y assembly 

477 # is amortised across frequencies via the upstream swept solve; 

478 # the per-k post-processing is O(n_p³) and dwarfed by the solve. 

479 Y_swept = np.asarray( 

480 s.compute_y_matrix_swept(k_array), dtype=np.complex128 

481 ) # (n_k, n_p, n_p) 

482 n_driven = sum( 

483 1 for i in range(Y_swept.shape[1]) if i not in self._tl_passive_feed_idx 

484 ) 

485 zs = np.empty((freqs.size, n_driven), dtype=np.complex128) 

486 for ki, freq in enumerate(freqs): 

487 Y_total = self._apply_tls(Y_swept[ki], self._wavelength_for(freq)) 

488 zs[ki] = self._impedance_from_y(Y_total) 

489 return zs 

490 zs = s.compute_impedance_swept(k_array) 

491 # Single-feed: (n_k,); multi-feed: (n_k, n_feeds). Normalise to 

492 # (n_k, n_feeds) to match PyNECEngine. 

493 zs = np.asarray(zs) 

494 if zs.ndim == 1: 

495 zs = zs.reshape(-1, 1) 

496 return zs

498 def _make_excited_solver(self, *, wavelength): 

499 """Build a solver whose feed voltages match the actual excitation: 

500 for plain designs, just the build_wires() voltages; for TL or 

501 Network designs, the per-port voltages after the network reduction 

502 so basis coefficients reflect the branch-induced port voltages. 

503 Without this, network-spec designs (where every named feed carries 

504 a placeholder V=0) would solve with no excitation — every basis 

505 coefficient is zero and `compute_impedance`'s V/I returns NaN.""" 

506 if self._network is not None: 

507 # excited_state solves the same MNA system as the impedance 

508 # path, so resistive loads shape the current through their 

509 # physical series BC; it also returns the radiation efficiency 

510 # and the source input power the far field uses to normalise 

511 # gain. 

512 Y = self._compute_y_matrix(wavelength) 

513 ( 

514 V_full, 

515 self._excited_efficiency, 

516 self._excited_p_in, 

517 self._excited_power_budget, 

518 ) = self._reducer.excited_state(Y, wavelength) 

519 feeds_resolved = [ 

520 (w, arc, complex(V_full[i])) 

521 for i, (w, arc, _v) in enumerate(self._feeds) 

522 ] 

523 elif self._tls: 

524 self._excited_efficiency = 1.0 

525 self._excited_power_budget = [] 

526 Y = self._compute_y_matrix(wavelength) 

527 Y_total = self._apply_tls(Y, wavelength) 

528 V, driven = self._resolve_feed_voltages(Y_total) 

529 # Source input power at the driven ports of the TL-stamped port 

530 # model; the TLs are lossless so this equals the power entering 

531 # the wires. 

532 I_ports = Y_total @ V 

533 self._excited_p_in = 0.5 * float( 

534 sum((V[i] * np.conj(I_ports[i])).real for i in driven) 

535 ) 

536 feeds_resolved = [ 

537 (w, arc, complex(V[i])) for i, (w, arc, _v) in enumerate(self._feeds) 

538 ] 

539 else: 

540 self._excited_efficiency = 1.0 

541 self._excited_power_budget = [] 

542 # Plain path: no port model here; input_power() derives P_in from 

543 # the excited solve's driving-point impedance(s) on demand. 

544 self._excited_p_in = None 

545 return self._make_solver(wavelength=wavelength) 

546 return self._solver( 

547 wires=self._polylines, 

548 n_per_edge_per_wire=self._edge_segments, 

549 feeds=feeds_resolved, 

550 wavelength=wavelength, 

551 wire_radius=self._wire_radius, 

552 ground_z=self._ground_z, 

553 junctions=self._junctions or None, 

554 cancel=self._cancel, 

555 **self._loading_kwargs, 

556 **self._ground_solver_kwargs(), 

557 **self._solver_kwargs, 

558 )

560 def input_power(self): 

561 """Input power 1/2·Re(Σ V_f·I_f*) in watts over the SOURCE feeds of 

562 the excited solve — the gain normaliser (gain = 4π·U/P_in). Power 

563 burned in resistive loads and any ground absorption stay inside 

564 P_in (that is what makes 4π·U/P_in GAIN rather than directivity). 

565 

566 Network / TL designs record it while resolving the excitation in 

567 `_make_excited_solver` (a load port is excited too, but it is not a 

568 source, so its absorbed power must not be summed — the port model 

569 separates the two). The plain path derives it from the excited 

570 driving-point impedance(s): I_f = V_f/Z_f, so each feed contributes 

571 ½·|V_f|²·Re(Z_f)/|Z_f|². 

572 """ 

573 wavelength = self._wavelength_for(self.builder.freq) 

574 sim, _coeffs, z = self._solved_excited(wavelength) 

575 p_in = getattr(self, "_excited_p_in", None) 

576 if p_in is not None: 

577 return float(p_in) 

578 return self._p_in_from_excited(sim, z)

616 def _segment_dipoles(self, sim, coeffs): 

617 """Returns (mid, dr, i_mid) — concatenated per-segment midpoints, 

618 edge vectors, and midpoint currents from the MoM solution.""" 

619 knot_currents = sim.currents_at_knots(coeffs) 

620 mids, drs, i_mids = [], [], [] 

621 for w_idx, polyline in enumerate(self._polylines): 

622 knots = _polyline_knots(polyline, self._edge_segments[w_idx]) 

623 cur = knot_currents[w_idx] 

624 drs.append(knots[1:] - knots[:-1]) 

625 mids.append(0.5 * (knots[1:] + knots[:-1])) 

626 i_mids.append(0.5 * (cur[1:] + cur[:-1])) 

627 return ( 

628 np.concatenate(mids, axis=0), 

629 np.concatenate(drs, axis=0), 

630 np.concatenate(i_mids, axis=0), 

631 )

633 def _evaluate_M_perp(self, mid, dr, i_mid, k, theta, phi, freq_hz): 

634 """|M_perp(θ,φ)|² on the (theta, phi) grids (radians). 

635 

636 With ground enabled, adds the geometric-image contribution with PEC 

637 polarity, then layers Fresnel coefficients on the reflected wave so 

638 ρ_h=−1, ρ_v=+1 recovers the PEC limit exactly. Returns a real 

639 (n_theta, n_phi) array.""" 

640 sin_t, cos_t = np.sin(theta), np.cos(theta) 

641 cos_p, sin_p = np.cos(phi), np.sin(phi) 

642 

643 rx = sin_t[:, None] * cos_p[None, :] 

644 ry = sin_t[:, None] * sin_p[None, :] 

645 rz = np.broadcast_to(cos_t[:, None], rx.shape) 

646 rhat = np.stack([rx, ry, rz], axis=-1) 

647 

648 phase = k * np.einsum("ijc,nc->ijn", rhat, mid) 

649 expp = np.exp(1j * phase) 

650 weighted = i_mid[:, None] * dr 

651 M = np.einsum("ijn,nc->ijc", expp, weighted) 

652 m_dot_r = np.sum(M * rhat, axis=-1) 

653 M_perp = M - m_dot_r[..., None] * rhat 

654 

655 if self._ground is None: 

656 return np.sum(M_perp.real**2 + M_perp.imag**2, axis=-1) 

657 

658 # Geometric image — horizontal current flipped, vertical preserved, 

659 # mirrored across z = ground_z. 

660 z0 = self._ground_z 

661 mid_img = mid.copy() 

662 mid_img[:, 2] = 2 * z0 - mid[:, 2] 

663 dr_img = dr * np.array([-1.0, -1.0, 1.0]) 

664 weighted_img = i_mid[:, None] * dr_img 

665 phase_img = k * np.einsum("ijc,nc->ijn", rhat, mid_img) 

666 expp_img = np.exp(1j * phase_img) 

667 M_img = np.einsum("ijn,nc->ijc", expp_img, weighted_img) 

668 m_img_dot_r = np.sum(M_img * rhat, axis=-1) 

669 M_img_perp = M_img - m_img_dot_r[..., None] * rhat 

670 

671 if self._ground[0] == "pec": 

672 M_perp = M_perp + M_img_perp 

673 return np.sum(M_perp.real**2 + M_perp.imag**2, axis=-1) 

674 

675 # ("finite", eps_r, sigma): polarisation basis at each ray and 

676 # Fresnel reflection on the image wave. 

677 _, eps_r, sigma = self._ground 

678 # Vertical (TM, in-plane) and horizontal (TE, out-of-plane) 

679 # polarisation unit vectors for the reflected ray, written straight 

680 # from the azimuth/zenith angles rather than from the horizontal 

681 # projection s = sin(theta). Dividing by s degenerates at the zenith 

682 # (s -> 0, the plane of incidence is undefined): the old s->1.0 guard 

683 # collapsed BOTH vectors to zero there, dropping the reflected wave and 

684 # reading ~3 dB low. The phi-limit h_hat=(-sin phi, cos phi, 0), 

685 # v_hat=(-cos phi, -sin phi, 0) is exact, unit, orthonormal to rhat, 

686 # and recovers the PEC limit (rho_h=-1, rho_v=+1) at theta=0. 

687 s = np.sqrt(rx * rx + ry * ry) # = sin(theta); feeds sin2_ti below 

688 cos_p_g = np.broadcast_to(cos_p[None, :], rx.shape) 

689 sin_p_g = np.broadcast_to(sin_p[None, :], rx.shape) 

690 cos_t_g = np.broadcast_to(cos_t[:, None], rx.shape) 

691 sin_t_g = np.broadcast_to(sin_t[:, None], rx.shape) 

692 h_hat = np.stack([-sin_p_g, cos_p_g, np.zeros_like(rx)], axis=-1) 

693 v_hat = np.stack([-cos_p_g * cos_t_g, -sin_p_g * cos_t_g, sin_t_g], axis=-1) 

694 M_img_h = np.sum(M_img_perp * h_hat, axis=-1) 

695 M_img_v = np.sum(M_img_perp * v_hat, axis=-1) 

696 

697 omega = 2 * np.pi * freq_hz 

698 eps_c = eps_r - 1j * sigma / (omega * EPS0) 

699 cos_ti = rz 

700 sin2_ti = s * s 

701 Q = np.sqrt(eps_c - sin2_ti) 

702 rho_h = (cos_ti - Q) / (cos_ti + Q) 

703 rho_v = (eps_c * cos_ti - Q) / (eps_c * cos_ti + Q) 

704 

705 # PEC reflection corresponds to ρ_h=−1, ρ_v=+1. The image we built 

706 # already has the PEC sign convention baked in, so we need the 

707 # Fresnel-vs-PEC ratio per polarisation: 

708 # reflected_h = (−ρ_h) · M_img_h # PEC was +1·M_img_h 

709 # reflected_v = (+ρ_v) · M_img_v # PEC was +1·M_img_v 

710 M_refl = (rho_v * M_img_v)[..., None] * v_hat - (rho_h * M_img_h)[ 

711 ..., None 

712 ] * h_hat 

713 M_perp = M_perp + M_refl 

714 return np.sum(M_perp.real**2 + M_perp.imag**2, axis=-1)

716 def far_field(self, *, n_theta=90, n_phi=360, del_theta=1, del_phi=1): 

717 self._raise_if_cancelled() 

718 assert 90 % n_theta == 0 and 90 == del_theta * n_theta 

719 assert 360 % n_phi == 0 and 360 == del_phi * n_phi 

720 

721 wavelength = self._wavelength_for(self.builder.freq) 

722 k = 2.0 * np.pi / wavelength 

723 freq_hz = self.builder.freq * 1e6 

724 

725 sim, coeffs, _z = self._solved_excited(wavelength) 

726 mid, dr, i_mid = self._segment_dipoles(sim, coeffs) 

727 

728 # Gain normaliser from the source input power (same convention as the 

729 # web solve path): gain = 4π·U/P_in = η₀k²/(8π·P_in)·|M_perp|². Load 

730 # loss lives inside P_in, so terminated antennas come out as GAIN with 

731 # no efficiency multiply. 

732 p_in = self.input_power() 

733 if p_in > 0: 

734 directivity_norm = ETA0 * k * k / (8.0 * np.pi * p_in) 

735 else: 

736 # Defensive fallback (a pathological R_in ≤ 0): normalise by the 

737 # integrated pattern instead. Cell-centred grid over the sphere, 

738 # upper hemisphere only with ground (the image contribution is 

739 # already folded into |M_perp|² there). 

740 n_th_int, n_ph_int = 90, 180 

741 if self._ground is not None: 

742 theta_int = (np.arange(n_th_int) + 0.5) * (np.pi / 2 / n_th_int) 

743 dtheta = np.pi / 2 / n_th_int 

744 else: 

745 theta_int = (np.arange(n_th_int) + 0.5) * (np.pi / n_th_int) 

746 dtheta = np.pi / n_th_int 

747 phi_int = np.arange(n_ph_int) * (2 * np.pi / n_ph_int) 

748 dphi = 2 * np.pi / n_ph_int 

749 mag2_int = self._evaluate_M_perp( 

750 mid, dr, i_mid, k, theta_int, phi_int, freq_hz 

751 ) 

752 p_rad = float(np.sum(mag2_int * np.sin(theta_int)[:, None]) * dtheta * dphi) 

753 if p_rad <= 0: 

754 raise RuntimeError("computed zero radiated power") 

755 efficiency = getattr(self, "_excited_efficiency", 1.0) 

756 directivity_norm = 4 * np.pi / p_rad * efficiency 

757 

758 # Evaluate on the user grid (NEC convention: θ from 0 to 90−Δθ). 

759 theta_deg = np.linspace(0, 90 - del_theta, n_theta) 

760 phi_deg = np.linspace(0, 360, n_phi + 1) 

761 theta_user = np.deg2rad(theta_deg) 

762 phi_user = np.deg2rad(phi_deg) 

763 

764 mag2_user = self._evaluate_M_perp( 

765 mid, dr, i_mid, k, theta_user, phi_user, freq_hz 

766 ) 

767 D = directivity_norm * mag2_user 

768 # Floor before log so points where M_perp is exactly zero (poles, 

769 # nulls below quantisation) don't produce −inf. 

770 dBi = 10.0 * np.log10(np.maximum(D, 1e-30)) 

771 

772 rings = dBi.tolist() 

773 return FarField( 

774 rings=rings, 

775 max_gain=float(np.max(dBi)), 

776 min_gain=float(np.min(dBi)), 

777 thetas=theta_deg, 

778 phis=phi_deg, 

779 )
