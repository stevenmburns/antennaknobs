44 @staticmethod 

45 def coerce_n_seg(n_seg: int, parity: SegmentParity) -> int: 

46 # Floor below the parity step. n_seg=0 is invalid for every engine 

47 # (momwire divides edge length by it), and even-parity engines need 

48 # at least 2 segments to host a feed straddling a midpoint. 

49 if parity == "even": 

50 n_seg = max(2, n_seg) 

51 return n_seg + 1 if n_seg % 2 == 1 else n_seg 

52 if parity == "odd": 

53 n_seg = max(1, n_seg) 

54 return n_seg + 1 if n_seg % 2 == 0 else n_seg 

55 return max(1, n_seg)

57 def _coerce_wire_tuples(self, tups): 

58 """Returns the input tuples with each n_seg bumped to the engine's 

59 required parity. Logs once per distinct (n_in, n_out) shift so a 

60 converge sweep doesn't spam the log per-edge. Reads 

61 self.segment_parity so subclasses can set it per-instance (e.g. 

62 MomwireEngine, where the parity depends on the chosen solver).""" 

63 parity = self.segment_parity 

64 if parity == "any": 

65 return tups 

66 seen = set() 

67 out = [] 

68 for t in tups: 

69 p0, p1, n_seg, ev = t[0], t[1], t[2], t[3] 

70 name = t[4] if len(t) >= 5 else None 

71 n_new = self.coerce_n_seg(n_seg, parity) 

72 if n_new != n_seg and (n_seg, n_new) not in seen: 

73 seen.add((n_seg, n_new)) 

74 _logger.info( 

75 "%s bumped n_seg=%d → %d for %s parity", 

76 type(self).__name__, 

77 n_seg, 

78 n_new, 

79 parity, 

80 ) 

81 if name is None: 

82 out.append((p0, p1, n_new, ev)) 

83 else: 

84 out.append((p0, p1, n_new, ev, name)) 

85 return out
