23# MKL/OpenBLAS-LAPACKE pool for cores. vtune confirmed this was costing 

24# ~8% wall time at NP=4 on the gather-scatter fill. 

25# 

26# OMP_NUM_THREADS / MKL_NUM_THREADS: with the gather-scatter matrix fill 

27# (see PR #21) the per-source parallel-for inside cmset() and MKL/OpenBLAS'

46def _physical_cpu_count() -> int: 

47 """Number of physical cores (not logical / HT siblings). 

48 

49 Our quadrature kernels are FP-vector-saturated (libmvec AVX2 sin/cos 

50 inner loops, no spare FU bandwidth), so two HT siblings on one physical 

51 core contend for execution units rather than overlap. Ad-hoc bench on 

52 KBL-R 4C/8T showed 4-thread runs ~15% faster than 8-thread runs of the 

53 swept-ground hot path. Pin to physical-core count to skip that loss. 

54 

55 Uses psutil for a portable answer (Windows/macOS/Linux). The previous 

56 /proc/cpuinfo + "assume 2 HT siblings" fallback misfired on chips 

57 without HT (e.g. Intel N-series E-core SoCs), pinning to half the 

58 actual core count. 

59 """ 

60 try: 

61 import psutil 

62 

63 n = psutil.cpu_count(logical=False) 

64 if n: 

65 return n 

66 except ImportError: 

67 pass 

68 return max(1, os.cpu_count() or 1)

138def _attach_derived_em_fields(out: dict) -> None: 

139 """Augment the solve response with frequency-derived EM scalars the 

140 frontend would otherwise compute from raw physics constants. 

141 

142 Sets: 

143 - `k_meas_m_inv`: wavenumber 2π f / c at measurement freq (rad/m) 

144 - `ground_eps_im`: imaginary part of the complex relative permittivity 

145 of the ground, -σ / (ω ε₀); 0 when ground is off or σ=0. 

146 

147 The frontend reads these directly so it doesn't need to carry C_LIGHT 

148 or ε₀ literals. `lambda_design_m` is already shipped by each example. 

149 """ 

150 f_hz = float(out["measurement_freq_mhz"]) * 1e6 

151 omega = 2.0 * np.pi * f_hz 

152 out["k_meas_m_inv"] = omega / C_LIGHT 

153 sigma = float(out.get("ground_sigma", 0.0) or 0.0) 

154 out["ground_eps_im"] = -sigma / (omega * _EPS0) if omega > 0 else 0.0

160def _attach_gain_norm(out: dict) -> None: 

161 """Attach `directivity_norm` = η₀k²/(8π·P_in), the O(1) gain normaliser. 

162 

163 Multiplying this by the frontend's azimuth-cut |M_perp(π/2, φ)|² yields 

164 absolute GAIN (linear); 10·log10 is dBi. Derivation: the far field of the 

165 moment sum M = Σ I·dr·e^{jk·r̂·x} is E = (jkη₀/4πr)·e^{−jkr}·M_perp, so the 

166 radiation intensity is U = r²|E|²/(2η₀) = (η₀k²/32π²)·|M_perp|² and 

167 gain = 4π·U/P_in = (η₀k²/8π)·|M_perp|²/P_in. 

168 

169 Normalising by SOURCE input power is what makes this gain rather than 

170 directivity: power burned in resistive loads (terminated rhombic / T2FD) 

171 or absorbed by a lossy ground stays inside P_in, so no efficiency multiply 

172 — this replaces the old pattern-integral norm (4π/∮|M_perp|²dΩ)×efficiency, 

173 which equals it identically up to the solver's self-consistency gap (the 

174 NEC "average gain" diagnostic; `_pattern_integral_norm` measures it). 

175 

176 Falls back to the pattern-integral norm when the response carries no 

177 usable input power (defensive: a pathological R_in ≤ 0 from a nearly 

178 lossless, strongly reactive discretisation). 

179 """ 

180 p_in = float(out.get("input_power_w", 0.0) or 0.0) 

181 if p_in <= 0.0: 

182 _compute_directivity_norm(out) 

183 return 

184 k = float(out["k_meas_m_inv"]) 

185 out["directivity_norm"] = _ETA0 * k * k / (8.0 * np.pi * p_in)

188def _adaptive_norm_grid(k: float, lo: np.ndarray, hi: np.ndarray) -> tuple[int, int]: 

189 """Grid resolution (n_theta, n_phi) for the directivity-norm integral, 

190 sized to the structure's electrical extent. 

191 

192 The far-field pattern is band-limited by the source's largest dimension: a 

193 structure spanning D radiates angular detail up to spherical-harmonic degree 

194 ~k·D, and the integrand |M_perp|² has twice that bandwidth. We size n_theta 

195 off the bounding-box diagonal in wavelengths, D_λ, as a constant (the base 

196 pattern's irreducible complexity) plus a slope in D_λ, then clamp. 

197 n_phi = 2·n_theta mirrors the 2× azimuthal bandwidth. 

198 

199 The constant + slope are fit empirically (scripts/ 

200 profile_ws_postproc_serialization.py) to sit safely *above* the aliasing 

201 floor: sampling just below the floor doesn't merely lose precision, it 

202 corrupts the scalar by ~1 dB (a 13.8λ loop reads −0.9 dB at n_theta=14 then 

203 snaps to −0.007 dB at n_theta=20). The bbox diagonal upper-bounds the true 

204 source diameter, so this errs conservative (a finer grid than strictly 

205 needed) — safe, and still ~10× cheaper than the old fixed 45×90 on the 

206 common electrically-small design. 

207 """ 

208 lam = (2.0 * np.pi / k) if k > 0 else float("inf") 

209 d_lambda = float(np.linalg.norm(hi - lo)) / lam if np.isfinite(lam) else 0.0 

210 n_theta = int(np.clip(np.ceil(13.0 + 1.2 * d_lambda), 12, 90)) 

211 return n_theta, 2 * n_theta

214def _fine_norm_grid(n_theta_adaptive: int) -> tuple[int, int]: 

215 """A reference grid comfortably finer than the adaptive pick, for the

225def _moment_segments(out: dict) -> tuple[np.ndarray, np.ndarray, np.ndarray]: 

226 """Segment midpoints (Nseg,3), segment vectors dr (Nseg,3) and midpoint 

227 currents (Nseg,) complex — the discrete moment set behind the far-field 

228 sum M(r̂) = Σ I·dr·e^{jk·r̂·mid}, shared by every pattern normaliser. 

229 

230 Prefers the finer-grained sample arrays (knot + segment-midpoint) when 

231 the model produced them, so non-tent bases get their intra-segment 

232 curvature integrated. Falls back to knot arrays for any backend that 

233 only ships knot data (PyNEC). 

234 """ 

235 mids, drs, i_mids = [], [], [] 

236 for w in out["wires"]: 

237 if "sample_positions" in w: 

238 pts = np.asarray(w["sample_positions"], dtype=np.float64) 

239 cur = np.asarray( 

240 w["sample_currents_re"], dtype=np.float64 

241 ) + 1j * np.asarray(w["sample_currents_im"], dtype=np.float64) 

242 else: 

243 pts = np.asarray(w["knot_positions"], dtype=np.float64) 

244 cur = np.asarray(w["knot_currents_re"], dtype=np.float64) + 1j * np.asarray( 

245 w["knot_currents_im"], dtype=np.float64 

246 ) 

247 drs.append(pts[1:] - pts[:-1]) 

248 mids.append(0.5 * (pts[1:] + pts[:-1])) 

249 i_mids.append(0.5 * (cur[1:] + cur[:-1])) 

250 return ( 

251 np.concatenate(mids, axis=0), 

252 np.concatenate(drs, axis=0), 

253 np.concatenate(i_mids, axis=0), 

254 )

257def _pattern_integral_norm(out: dict) -> float: 

258 """The pattern-integral gain norm (4π/∮|M_perp|²dΩ)·efficiency evaluated 

259 in CLOSED FORM — no angular grid. Because the radiated power is quadratic 

260 in the currents, the sphere integral collapses to a pair sum over the 

261 moment set with the classical mutual-radiation-resistance kernel: 

262 

263 ∮ (I₃ − r̂r̂)·e^{jk·r̂·d} dΩ = 4π[ a(x)·I₃ − b(x)·d̂d̂ ], x = k|d| 

264 a(x) = j₀(x) − j₁(x)/x (→ 2/3 as x → 0) 

265 b(x) = j₀(x) − 3·j₁(x)/x (→ 0 as x → 0) 

266 

267 with spherical Bessels j₀, j₁ — real, smooth, exact. O(N²) pairs, no 

268 aliasing floor, no grid to size. 

269 

270 Ground (the web paths are PEC — eps_r is the 1e10 sentinel): image 

271 segments (x,y,z) → (x,y,−z) with horizontal moment components flipped 

272 reproduce the reflected wave exactly, and the imaged 2N system is 

273 mirror-symmetric, so the upper-hemisphere power is half its full-sphere 

274 power. The Fresnel coefficients at eps_r = 1e10 differ from the PEC 

275 limit only within ~1e-5 of grazing — beyond any displayed precision. 

276 

277 This is the same discrete functional the old grid integral sampled, so 

278 the delta against the P_in-based `directivity_norm` isolates the solver 

279 self-consistency gap (NEC's "average gain" diagnostic), not quadrature. 

280 """ 

281 k = float(out["k_meas_m_inv"]) 

282 mid, dr, i_mid = _moment_segments(out) 

283 w = i_mid[:, None] * dr # complex moment per segment (Nseg, 3) 

284 x_pts = mid 

285 half = 1.0 

286 if bool(out.get("ground", False)): 

287 x_pts = np.concatenate([mid, mid * np.array([1.0, 1.0, -1.0])], axis=0) 

288 w = np.concatenate([w, w * np.array([-1.0, -1.0, 1.0])], axis=0) 

289 half = 0.5 

290 

291 # Pair sum in row blocks so peak memory stays O(block·N) instead of 

292 # O(N²·3) — the terminated rhombic over ground is a ~2600-point set. 

293 n_pts = x_pts.shape[0] 

294 w_conj = np.conj(w) 

295 block = max(1, int(2e6) // max(n_pts, 1)) 

296 p_sum = 0.0 

297 for s in range(0, n_pts, block): 

298 e = min(s + block, n_pts) 

299 d = x_pts[s:e, None, :] - x_pts[None, :, :] # (B, N, 3) 

300 x = k * np.linalg.norm(d, axis=-1) # (B, N) 

301 small = x < 1e-3 

302 xs = np.where(small, 1.0, x) # avoid 0-division; small arm uses series 

303 sin_x, cos_x = np.sin(xs), np.cos(xs) 

304 j0 = sin_x / xs 

305 j1_over_x = (sin_x / xs - cos_x) / (xs * xs) 

306 x2 = x * x 

307 # 2-term series at small x (the exact forms lose precision to 

308 # cancellation): a = 2/3 − 2x²/15, b = −x²/15. 

309 a = np.where(small, 2.0 / 3.0 - 2.0 * x2 / 15.0, j0 - j1_over_x) 

310 b = np.where(small, -x2 / 15.0, j0 - 3.0 * j1_over_x) 

311 

312 # w_m*ᵀ [a·I − b·d̂d̂] w_n over the block's pairs. d̂ is undefined 

313 # at d=0 but b→0 there, so guard the denominator instead of 

314 # special-casing the diagonal. 

315 dot_ww = w_conj[s:e] @ w.T # (B, N) 

316 d_norm = np.where(small, 1.0, x / k) # |d| with the same guard 

317 proj_m = np.einsum("bnc,bc->bn", d, w_conj[s:e]) / d_norm 

318 proj_n = np.einsum("bnc,nc->bn", d, w) / d_norm 

319 p_sum += float(np.sum(a * dot_ww.real - b * (proj_m * proj_n).real)) 

320 p_rad = 4.0 * np.pi * half * p_sum 

321 if p_rad <= 0.0: 

322 return 0.0 

323 efficiency = float(out.get("radiation_efficiency", 1.0)) 

324 return 4.0 * np.pi / p_rad * efficiency

327def _compute_directivity_norm( 

328 out: dict, 

329 n_theta: int | None = None, 

330 n_phi: int | None = None, 

331 *, 

332 _theta_rule: str = "gl", 

333) -> None: 

334 """Attach `directivity_norm` = 4π / ∫|M_perp|² dΩ to the response. 

335 

336 Multiplying this by the frontend's azimuth-cut |M_perp(π/2, φ)|² yields 

337 absolute directivity D(φ) (linear); 10·log10(D) is dBi. 

338 

339 With ground enabled, integrates only the upper hemisphere and adds the 

340 Fresnel-reflected contribution from the geometric image so the 

341 normalization matches what the JS far-field code displays. 

342 

343 The θ direction uses Gauss–Legendre quadrature in u = cos θ (the sin θ 

344 Jacobian is absorbed into the weights); φ stays a uniform rectangle rule 

345 (periodic → spectrally accurate). By default the grid is sized to the 

346 structure's electrical extent via `_adaptive_norm_grid`; callers may pass an 

347 explicit `n_theta`/`n_phi` (e.g. the convergence harness). 

348 """ 

349 k = float(out["k_meas_m_inv"]) 

350 ground_on = bool(out.get("ground", False)) 

351 mid, dr, i_mid = _moment_segments(out) 

352 

353 if n_theta is None or n_phi is None: 

354 # Size the grid to the structure's electrical extent. Segment midpoints 

355 # under-cover the true endpoints by at most half a (sub-λ) segment — 

356 # negligible for the bounding-box diagonal used to pick the resolution. 

357 n_theta, n_phi = _adaptive_norm_grid(k, mid.min(axis=0), mid.max(axis=0)) 

358 

359 # θ integration in u = cos θ, with the sin θ Jacobian (du = −sin θ dθ) 

360 # folded into `w_theta` so the radiated-power sum below needs no extra sin θ 

361 # factor. Default is Gauss–Legendre (far more accurate per θ-point above the 

362 # resolution floor); `_theta_rule="uniform"` selects the legacy midpoint- 

363 # rectangle rule and exists only so the profiling harness can quantify the 

364 # GL win. With ground, integrate only the upper hemisphere (θ ∈ [0, π/2]). 

365 half = 0.5 if ground_on else 1.0 

366 if _theta_rule == "gl": 

367 gl_x, gl_w = np.polynomial.legendre.leggauss(n_theta) 

368 # Map the [−1, 1] rule onto u ∈ [0, 1] for a hemisphere, else keep [−1, 1]. 

369 u = 0.5 * (gl_x + 1.0) if ground_on else gl_x 

370 w_theta = half * gl_w 

371 elif _theta_rule == "uniform": 

372 theta = (np.arange(n_theta) + 0.5) * (half * np.pi / n_theta) 

373 u = np.cos(theta) 

374 w_theta = np.sin(theta) * (half * np.pi / n_theta) 

375 else: 

376 raise ValueError(f"unknown _theta_rule {_theta_rule!r}") 

377 cos_t = u 

378 sin_t = np.sqrt(np.clip(1.0 - u * u, 0.0, None)) 

379 phi = np.arange(n_phi) * (2 * np.pi / n_phi) 

380 cos_p, sin_p = np.cos(phi), np.sin(phi) 

381 

382 rx = sin_t[:, None] * cos_p[None, :] 

383 ry = sin_t[:, None] * sin_p[None, :] 

384 rz = np.broadcast_to(cos_t[:, None], (n_theta, n_phi)) 

385 rhat = np.stack([rx, ry, rz], axis=-1) # (nθ, nφ, 3) 

386 

387 phase = k * np.einsum("ijc,nc->ijn", rhat, mid) # (nθ, nφ, Nseg) 

388 expp = np.exp(1j * phase) 

389 weighted = i_mid[:, None] * dr # (Nseg, 3) 

390 M = np.einsum("ijn,nc->ijc", expp, weighted) # (nθ, nφ, 3) 

391 m_dot_r = np.sum(M * rhat, axis=-1) 

392 M_perp = M - m_dot_r[..., None] * rhat 

393 

394 if ground_on: 

395 # PEC-image method, then Fresnel-correct the reflected wave per-ray. 

396 # Image current: horizontal components flipped, vertical preserved. 

397 # This reproduces PEC reflection when ρ_h=-1, ρ_v=+1, and lets us 

398 # apply the actual finite-ground coefficients to that same image. 

399 mid_img = mid * np.array([1.0, 1.0, -1.0]) 

400 dr_img = dr * np.array([-1.0, -1.0, 1.0]) 

401 weighted_img = i_mid[:, None] * dr_img 

402 phase_img = k * np.einsum("ijc,nc->ijn", rhat, mid_img) 

403 expp_img = np.exp(1j * phase_img) 

404 M_img = np.einsum("ijn,nc->ijc", expp_img, weighted_img) 

405 m_img_dot_r = np.sum(M_img * rhat, axis=-1) 

406 M_img_perp = M_img - m_img_dot_r[..., None] * rhat 

407 

408 # Polarization basis at each ray: ĥ = ẑ × r̂ (perp to plane of 

409 # incidence), v̂ = r̂ × ĥ (in plane of incidence, perp to r̂). 

410 s = np.sqrt(rx * rx + ry * ry) 

411 s_safe = np.where(s > 1e-12, s, 1.0) 

412 h_hat = np.stack([-ry / s_safe, rx / s_safe, np.zeros_like(rx)], axis=-1) 

413 v_hat = np.stack([-rx * rz / s_safe, -ry * rz / s_safe, s], axis=-1) 

414 

415 M_img_h = np.sum(M_img_perp * h_hat, axis=-1) # complex (nθ, nφ) 

416 M_img_v = np.sum(M_img_perp * v_hat, axis=-1) 

417 

418 eps_c = out["ground_eps_r"] + 1j * out["ground_eps_im"] 

419 cos_ti = rz 

420 sin2_ti = s * s 

421 Q = np.sqrt(eps_c - sin2_ti) 

422 rho_h = (cos_ti - Q) / (cos_ti + Q) 

423 rho_v = (eps_c * cos_ti - Q) / (eps_c * cos_ti + Q) 

424 

425 # Reflected: ρ_v on the v-pol component, −ρ_h on the h-pol component 

426 # (the minus sign folds the PEC image's pre-applied horizontal flip 

427 # back out so ρ_h=−1 recovers the PEC limit exactly). 

428 M_refl = (rho_v * M_img_v)[..., None] * v_hat - (rho_h * M_img_h)[ 

429 ..., None 

430 ] * h_hat 

431 M_perp = M_perp + M_refl 

432 

433 mag2 = np.sum((M_perp.real**2 + M_perp.imag**2), axis=-1) # (nθ, nφ) 

434 

435 # Gauss–Legendre in θ (weight absorbs sin θ) × uniform rectangle in φ. 

436 dphi = 2 * np.pi / n_phi 

437 p_rad = float(np.sum(mag2 * w_theta[:, None]) * dphi) 

438 # Fold in the radiation efficiency (P_radiated / P_input) so a terminated / 

439 # loaded antenna plots GAIN, not directivity: 4π/p_rad is the directivity 

440 # normaliser, and multiplying by efficiency drops the peak by the fraction 

441 # of power burned in resistive loads. Defaults to 1.0 (lossless / no loads, 

442 # and the PyNEC path which doesn't report it), leaving every other design 

443 # unchanged. 

444 efficiency = float(out.get("radiation_efficiency", 1.0)) 

445 out["directivity_norm"] = (4 * np.pi / p_rad * efficiency) if p_rad > 0 else 0.0 

446 # Record the grid that produced this norm — the far-field grid-check overlay 

447 # reads it to derive a finer reference grid and to label the comparison. 

448 out["directivity_norm_grid"] = [int(n_theta), int(n_phi)]

451def _wire_record( 

452 knots: np.ndarray, 

453 currents: np.ndarray, 

454 label: str, 

455 sample_currents: np.ndarray | None = None, 

456) -> dict: 

457 """Package one wire's record for the JSON response. `currents` is a 

458 length-M_w complex array (one per mesh knot) as produced by each 

459 model's `currents_at_knots(coeffs)` method. 

460 

461 When `sample_currents` is provided, additional `sample_positions` / 

462 `sample_currents_re` / `sample_currents_im` arrays are attached at 

463 knots-and-midpoints interleaved (2*N_seg + 1 entries per wire). This is 

464 what `_compute_directivity_norm` and the frontend renderers consume to 

465 resolve intra-segment basis curvature (B-spline d=2, sinusoidal three- 

466 term) and the B-spline enrichment shape that vanishes at every knot. 

467 """ 

468 currents = np.asarray(currents, dtype=np.complex128) 

469 if currents.shape[0] != knots.shape[0]: 

470 raise ValueError( 

471 f"_wire_record: currents/knots length mismatch " 

472 f"({currents.shape[0]} vs {knots.shape[0]})" 

473 ) 

474 out = { 

475 "label": label, 

476 "knot_positions": knots.tolist(), 

477 "knot_currents_re": currents.real.tolist(), 

478 "knot_currents_im": currents.imag.tolist(), 

479 } 

480 if sample_currents is not None: 

481 sample_currents = np.asarray(sample_currents, dtype=np.complex128) 

482 n_seg = knots.shape[0] - 1 

483 expected = 2 * n_seg + 1 

484 if sample_currents.shape[0] != expected: 

485 raise ValueError( 

486 f"_wire_record: sample_currents length {sample_currents.shape[0]} " 

487 f"!= expected 2*N_seg+1 = {expected}" 

488 ) 

489 sample_positions = np.empty((expected, 3), dtype=np.float64) 

490 sample_positions[0::2] = knots 

491 sample_positions[1::2] = 0.5 * (knots[:-1] + knots[1:]) 

492 out["sample_positions"] = sample_positions.tolist() 

493 out["sample_currents_re"] = sample_currents.real.tolist() 

494 out["sample_currents_im"] = sample_currents.imag.tolist() 

495 return out

498def _sample_arc_for_wire(knots: np.ndarray) -> np.ndarray: 

499 """Build interleaved (knot_arc, midpoint_arc, knot_arc, ...) array from a 

500 wire's 3D knot positions. Segment lengths come from successive-knot 

501 distances along the polyline. 

502 """ 

503 knots = np.asarray(knots, dtype=np.float64) 

504 h_seg = np.linalg.norm(knots[1:] - knots[:-1], axis=1) 

505 arc_at_knot = np.concatenate([[0.0], np.cumsum(h_seg)]) 

506 mid_arc = 0.5 * (arc_at_knot[:-1] + arc_at_knot[1:]) 

507 sample_arc = np.empty(2 * h_seg.shape[0] + 1, dtype=np.float64) 

508 sample_arc[0::2] = arc_at_knot 

509 sample_arc[1::2] = mid_arc 

510 return sample_arc

513def _pack_momwire_wires(sim, coeffs, knot_arrays, labels) -> list[dict]: 

514 """Build wire records for every momwire wire with both knot-level currents 

515 AND finer-grained mid-segment samples (one extra sample per segment). 

516 

517 Calls `sim.currents_at_knots(coeffs)` once for the knot values and once 

518 more with an `s_array` of per-wire interleaved knot-and-midpoint arcs. 

519 The model's basis is then evaluated exactly at the midpoints — including 

520 the B-spline enrichment basis Φ_sing, which is zero at the knots but 

521 non-zero in the interior. 

522 """

524 knot_currents = sim.currents_at_knots(coeffs) 

525 sample_currents = sim.currents_at_knots(coeffs, s_array=sample_arcs) 

526 return [ 

527 _wire_record( 

528 np.asarray(knot_arrays[i]), 

529 knot_currents[i], 

530 labels[i], 

531 sample_currents=sample_currents[i], 

532 ) 

533 for i in range(len(knot_arrays)) 

534 ]

557# --- Live-engine size guard (hosted only) ---------------------------------- 

558# A solve builds a method-of-moments system whose dimension N ≈ the total wire 

559# segment count (one basis function per segment). The dense solvers — and PyNEC 

560# — form an N×N complex128 matrix (memory N²·16 bytes), so an unbounded N (a 

561# hand-edited request cranking "segments / wire", or a big array) can exhaust a 

562# small box's RAM. 

563# 

564# This guard is OFF by default, so the package a user `pip install`s and runs 

565# locally is unlocked — solve as big as your machine allows. It turns ON only 

566# when ANTENNAKNOBS_HOSTED is set (truthy), which the shared instance does via 

567# fly.toml's [env]. So the same wheel is unlocked locally and capped online. 

568# 

569# The caps are sized to keep a single solve's matrix under ~800 MB on the 2 GB 

570# Fly box (basis = √(800·2²⁰/16) ≈ 7000 for a dense N×N). Measured on 

571# arrays.bowtiearray2x4 (see scripts/measure_solve_memory.py): PyNEC's RSS 

572# tracks the full dense N×N (~1 GB at basis 8000), while arrayblock's block- 

573# low-rank uses ~0.6× of that — so it's allowed a proportionally higher cap. 

574# Caps are about MEMORY, not solve time (PyNEC's ~N³ LU is slow long before it 

575# is large; that's a responsiveness concern, deliberately not guarded here). 

576# All env-overridable for self-hosting on bigger boxes. 

577def _env_flag(name: str) -> bool: 

578 return os.environ.get(name, "").strip().lower() in {"1", "true", "yes", "on"}

602def _check_solve_size(req: dict, *, use_pynec: bool) -> None: 

603 """Reject a solve whose matrix would be too large for the hosted live engine. 

604 

605 No-op unless running hosted (ANTENNAKNOBS_HOSTED) — local instances are 

606 unlocked. Best-effort: counts segments via the geometry-only ``count_basis`` 

607 hook (cheap — no solve). If the count can't be determined (geometry won't 

608 build), return and let the normal solve path surface the real error. 

609 """ 

610 if not _HOSTED: 

611 return 

612 geometry = req.get("geometry", next(iter(EXAMPLES))) 

613 ex = EXAMPLES.get(geometry) 

614 if ex is None or ex.count_basis is None: 

615 return 

616 n = ex.count_basis(req) 

617 if n is None: 

618 return 

619 if use_pynec: 

620 cap, engine, compressed = _MAX_BASIS_PYNEC, "PyNEC", False 

621 elif req.get("momwire_model") in _COMPRESSED_MODELS: 

622 cap = _MAX_BASIS_COMPRESSED 

623 engine, compressed = str(req.get("momwire_model")), True 

624 else: 

625 cap, engine, compressed = _MAX_BASIS, "momwire", False 

626 if n > cap: 

627 # Dense momwire and PyNEC both form the full N×N matrix; point users at 

628 # the compressed engines (which don't) for big arrays. 

629 hint = ( 

630 "" 

631 if compressed 

632 else " — or switch to the array-block / H-matrix engine, which " 

633 "handles larger arrays without a dense matrix" 

634 ) 

635 raise SolveTooLargeError( 

636 f"This solve needs ~{n} wire segments, over the live {engine} " 

637 f"limit of {cap}. Reduce 'segments / wire (N)' or pick a smaller " 

638 f"design{hint}." 

639 )

642# Request fields that are pure metadata and never change the physics. Pop 

643# them before hashing so noisy frontend additions (timestamps, request ids) 

644# don't shred the hit rate. Anything else in `req` is treated as load- 

645# bearing — preferring "extra miss" over "wrong hit". 

646_CACHE_KEY_BLOCKLIST = frozenset( 

647 { 

648 "_request_id", 

649 "_client_ts", 

650 # Per-request sequence number for the /ws latest-wins protocol. Pure 

651 # metadata — echoed back so the client can order/prune responses; must 

652 # not shred the cache hit rate (a scrub back to an earlier value should 

653 # still hit even though its _seq is higher). 

654 "_seq", 

655 } 

656) 

657 

658# Quantisation grid for floats in the cache key. Slider grids in the UI 

659# are coarser than 1e-6, so this still lets back-and-forth scrubs land on 

660# identical values; finer than user-perceivable change so two genuinely 

661# different requests don't collide. 

662_CACHE_FLOAT_QUANT = 1e-6

665def _canonical_solve_key(req: dict) -> str: 

666 def quantise(x): 

667 if isinstance(x, float): 

668 return round(x / _CACHE_FLOAT_QUANT) * _CACHE_FLOAT_QUANT 

669 if isinstance(x, dict): 

670 return { 

671 k: quantise(v) for k, v in x.items() if k not in _CACHE_KEY_BLOCKLIST 

672 } 

673 if isinstance(x, (list, tuple)): 

674 return [quantise(v) for v in x] 

675 return x 

676 

677 blob = json.dumps(quantise(req), sort_keys=True, default=str).encode() 

678 return hashlib.blake2b(blob, digest_size=16).hexdigest()

681def _solve_uncached(req: dict, cancel=None) -> dict: 

682 geometry = req.get("geometry", next(iter(EXAMPLES))) 

683 use_pynec = req.get("solver") == "pynec" and pynec_backend.HAVE_PYNEC 

684 _check_solve_size(req, use_pynec=use_pynec) 

685 if use_pynec: 

686 # PyNEC start-gate only: a request already superseded before its solve 

687 # begins dies for free here; the native solve is one opaque call with no 

688 # mid-solve abort, so an in-flight one runs to completion (as today). 

689 if cancel is not None: 

690 cancel.raise_if_cancelled() 

691 out = pynec_backend.solve(req) 

692 else: 

693 ex = EXAMPLES.get(geometry) or next(iter(EXAMPLES.values())) 

694 out = ex.momwire_solve(req, cancel=cancel)

698 return out

701def solve(req: dict, cancel=None) -> dict: 

702 key = _canonical_solve_key(req) 

703 hit = _SOLVE_CACHE.get(key) 

704 # Cache hits are O(1) and never worth aborting — the token is only consulted 

705 # on the (expensive) miss path below. A SolveAborted from _solve_uncached 

706 # propagates before the cache-store line, so an aborted solve is never cached. 

707 if hit is not None: 

708 _SOLVE_CACHE.move_to_end(key) 

709 t0 = time.perf_counter() 

710 out = deepcopy(hit) 

711 # Overwrite the cached solve_ms with the actual cost of producing 

712 # this response (the lookup + deepcopy) — otherwise the frontend's 

713 # "solve time" indicator shows a stale value from whichever earlier 

714 # tick first populated this cache entry. 

715 out["solve_ms"] = (time.perf_counter() - t0) * 1e3 

716 out["cache_hit"] = True 

717 return out 

718 out = _solve_uncached(req, cancel=cancel) 

719 out["cache_hit"] = False 

720 _SOLVE_CACHE[key] = deepcopy(out) 

721 while len(_SOLVE_CACHE) > _SOLVE_CACHE_MAX: 

722 _SOLVE_CACHE.popitem(last=False) 

723 return out

726@app.post("/sweep") 

727async def sweep_endpoint(req: dict, request: Request): 

728 """Stream sweep points as NDJSON, one (freq, Z) per line. 

729 

730 Streaming so the UI can show partial results as they're computed, and 

731 so the server can stop mid-sweep when the client disconnects — without 

732 this the user's slider drags abort the fetch client-side but the server 

733 keeps grinding through all 41 expensive PyNEC ground solves, starving 

734 the live /ws solves of CPU. 

735 """ 

736 freqs = [float(f) for f in req.get("freqs_mhz", [])] 

737 geometry = req.get("geometry", next(iter(EXAMPLES))) 

738 sweep_ex = EXAMPLES.get(geometry) or next(iter(EXAMPLES.values())) 

739 use_pynec = req.get("solver") == "pynec" and pynec_backend.HAVE_PYNEC 

740 solver_name = "pynec" if use_pynec else "momwire" 

741 

742 async def gen(): 

743 if not freqs: 

744 yield json.dumps({"done": True, "solver": solver_name}) + "\n" 

745 return 

746 

747 if use_pynec: 

748 # Per-point loop with disconnect check; lets us bail before the 

749 # next ~100 ms PyNEC ground solve when the user moves a slider. 

750 # Multi-feed geometries take the multifeed sweep so per-feed Z 

751 # streams alongside the primary z_re / z_im. 

752 is_multifeed = sweep_ex.multi_feed 

753 for f in freqs: 

754 if await request.is_disconnected(): 

755 return 

756 if is_multifeed: 

757 primary, feeds_z = await run_in_threadpool( 

758 pynec_backend._sweep_at_multifeed, req, f 

759 ) 

760 record = { 

761 "freq_mhz": f, 

762 "z_re": float(primary.real), 

763 "z_im": float(primary.imag), 

764 "feeds_z_re": [float(z_.real) for z_ in feeds_z], 

765 "feeds_z_im": [float(z_.imag) for z_ in feeds_z], 

766 "solver": solver_name, 

767 } 

768 else: 

769 z = await run_in_threadpool(pynec_backend._sweep_at, req, f) 

770 record = { 

771 "freq_mhz": f, 

772 "z_re": float(z.real), 

773 "z_im": float(z.imag), 

774 "solver": solver_name, 

775 } 

776 yield json.dumps(record) + "\n" 

777 else: 

778 # momwire's batched sweep is ~10x faster per-call than per-point, 

779 # but a 5-band fan dipole sweep at n_per_wire=21, 41 freqs takes 

780 # ~6 s and holds several hundred MB of J tensors — long enough 

781 # that rapid slider drags would otherwise pile up concurrent 

782 # computes in the threadpool, exhausting threads or memory and 

783 # surfacing as a 500 at the Vite proxy. 

784 # 

785 # Chunk the sweep so we can check is_disconnected between 

786 # batches. Per-freq cost has a bowl curve in chunk size: 

787 # tiny chunks pay per-call overhead, huge chunks thrash memory 

788 # bandwidth. For the 5-band fan-dipole geometry the sweet spot 

789 # is chunk_size ≈ 8 (115 ms/freq); for an inverted V it's much 

790 # larger (single-digit ms/freq, all freqs in one go is fine). 

791 # 

792 # Aim each chunk at roughly _CHUNK_TARGET_MS so the cancellation 

793 # granularity is consistent across geometries. Start with an 

794 # 8-chunk heuristic, then after each chunk recompute the next 

795 # size from observed per-freq cost. Converges in ~1 iteration. 

796 sweep_fn = sweep_ex.momwire_sweep 

797 chunk_size = max(1, len(freqs) // 8) 

798 start = 0 

799 while start < len(freqs): 

800 if await request.is_disconnected(): 

801 return 

802 chunk = freqs[start : start + chunk_size] 

803 t0 = time.perf_counter() 

804 sweep_result = await run_in_threadpool(sweep_fn, req, chunk) 

805 # Multi-feed sweeps (bowtie array) return a 4-tuple with 

806 # per-feed Z appended. Everything else stays on the 

807 # original 2-tuple shape; the legacy z_re / z_im fields 

808 # always carry the primary feed for back-compat. 

809 feeds_re_chunk: list[list[float]] | None = None 

810 feeds_im_chunk: list[list[float]] | None = None 

811 if len(sweep_result) == 4: 

812 z_re, z_im, feeds_re_chunk, feeds_im_chunk = sweep_result 

813 else: 

814 z_re, z_im = sweep_result 

815 chunk_ms = (time.perf_counter() - t0) * 1000 

816 for i, f in enumerate(chunk): 

817 record: dict = { 

818 "freq_mhz": f, 

819 "z_re": z_re[i], 

820 "z_im": z_im[i], 

821 "solver": solver_name, 

822 } 

823 if feeds_re_chunk is not None: 

824 record["feeds_z_re"] = feeds_re_chunk[i] 

825 record["feeds_z_im"] = feeds_im_chunk[i] 

826 yield json.dumps(record) + "\n" 

827 start += len(chunk) 

828 # Adapt for the next chunk: target _CHUNK_TARGET_MS per 

829 # batch. Per-freq cost is a weak function of chunk size 

830 # (bowl curve), so this converges quickly. 

831 if chunk_ms > 0 and len(chunk) > 0: 

832 per_freq_ms = chunk_ms / len(chunk) 

833 chunk_size = max(1, round(_CHUNK_TARGET_MS / per_freq_ms)) 

834 

835 yield json.dumps({"done": True, "solver": solver_name}) + "\n" 

836 

837 return StreamingResponse(gen(), media_type="application/x-ndjson")

935def _norm_check(req: dict) -> dict: 

936 """Consistency check for the far-field normalisation, dwell-triggered. 

937 

938 The live `directivity_norm` comes from the circuit side (η₀k²/8π·P_in); 

939 here we recompute the same gain norm from the FIELD side — the closed-form 

940 pattern integral (`_pattern_integral_norm`) × efficiency. The two agree 

941 exactly for a self-consistent solve, so the dB gap between them is the 

942 discretisation's power-balance error: NEC's "average gain" diagnostic. 

943 The norm is a single scalar multiplying the whole pattern, so the frontend 

944 overlays it as a pure radial dBi shift of the live trace. 

945 

946 Reuses the settled solve (a cache hit on the dwell request, so no re-solve 

947 on the common path). Falls back to the fine-grid quadrature when the 

948 response carries a finite (non-PEC-sentinel) ground — the image identity 

949 behind the closed form is exact only for a perfect reflector.""" 

950 out = solve(dict(req)) 

951 if "directivity_norm" not in out or out["directivity_norm"] <= 0:

1255@app.websocket("/ws") 

1256async def ws_endpoint(ws: WebSocket): 

1257 # Latest-wins mailbox. A dedicated reader task drains the socket into a 

1258 # size-1 mailbox (overwriting any unsolved request), while the solver loop 

1259 # below pulls the newest request whenever it's free. This squashes 

1260 # superseded knob changes *server-side*: the client sends every change 

1261 # eagerly with a monotonic `_seq`, and only the freshest queued request is 

1262 # ever solved. Results known to be superseded (a newer request already sat 

1263 # in the mailbox before we could send) are skipped — the doomed payload 

1264 # never travels. The client renders monotonically by `_seq`, so a higher 

1265 # `_seq` response implicitly acknowledges every lower one. 

1266 await ws.accept() 

1267 mailbox: list[dict] = [] # size-1: newest unsolved request only 

1268 newer = asyncio.Event() # set when the mailbox is (re)filled 

1269 closed = asyncio.Event() # set when the socket disconnects 

1270 # In-flight solve's cancel token, shared between the reader and the solver 

1271 # loop (both coroutines on this event loop, so no lock needed — the token's 

1272 # flag is the only thing the threadpool worker touches). The reader trips it 

1273 # to preempt a solve the moment a newer request lands or the socket closes. 

1274 current: dict = {"token": None} 

1275 

1276 async def reader() -> None: 

1277 # Starlette requires a single reader on the socket, so *all* 

1278 # receive_text calls happen here; the solver loop never reads. 

1279 try: 

1280 while True: 

1281 req = json.loads(await ws.receive_text()) 

1282 mailbox[:] = [req] # overwrite → squash anything unsolved 

1283 token = current["token"] 

1284 if token is not None: 

1285 token.cancel() # preempt the now-superseded in-flight solve 

1286 newer.set() 

1287 except WebSocketDisconnect: 

1288 pass 

1289 finally: 

1290 closed.set() 

1291 token = current["token"] 

1292 if token is not None: 

1293 token.cancel() # disconnect: free the threadpool worker promptly 

1294 newer.set() # wake the solver so it can observe `closed` and exit 

1295 

1296 reader_task = asyncio.create_task(reader()) 

1297 try: 

1298 while True: 

1299 await newer.wait() 

1300 newer.clear() 

1301 if closed.is_set() and not mailbox: 

1302 return 

1303 if not mailbox: 

1304 continue 

1305 req = mailbox.pop() 

1306 # Publish the token BEFORE dispatch: a reader that fires in the gap 

1307 # cancels a not-yet-started solve, which then raises SolveAborted at 

1308 # its first checkpoint — no lost-wakeup window. 

1309 token = momwire.CancelToken() 

1310 current["token"] = token 

1311 try: 

1312 result = await run_in_threadpool(solve, req, cancel=token) 

1313 except momwire.SolveAborted: 

1314 # Superseded (or disconnected) mid-solve: a newer request already 

1315 # tripped our token. Send nothing — the superseding response will 

1316 # carry a higher _seq and the client renders monotonically. This 

1317 # catch MUST precede the generic handler, which would otherwise 

1318 # ship the abort to the client as a solve-error banner. 

1319 continue 

1320 except Exception as exc: # noqa: BLE001 — a user design's build_wires can raise 

1321 # A solve that raises must not tear down the socket (that drops 

1322 # every subsequent slider-driven solve). Send the cause so the 

1323 # frontend shows it in the solve-error banner, then keep serving. 

1324 result = { 

1325 "geometry": req.get("geometry"), 

1326 "error": user_designs.format_solve_error(exc), 

1327 } 

1328 finally: 

1329 current["token"] = None 

1330 # Echo the sequence number on EVERY response, error path included — 

1331 # the client keys ordering, RTT accounting, and solving-state off it, 

1332 # and a stuck request would leave `solving` true forever if any path 

1333 # dropped the echo. The stamp lands on the (deep)copied result solve() 

1334 # returns, never on a cached entry. 

1335 result["_seq"] = req.get("_seq") 

1336 # Superseded while we solved? A newer request is already queued, so 

1337 # skip this send entirely — its response will carry a higher `_seq` 

1338 # and the client renders monotonically. Saves the full doomed payload 

1339 # (wires + interleaved sample-current arrays) on the wire. 

1340 if mailbox: 

1341 continue 

1342 # The client can disconnect *during* the solve (rapid slider drag 

1343 # tears down the React effect's WS and opens a fresh one before our 

1344 # threadpool finishes). When that happens send_text races with the 

1345 # closed socket and uvicorn logs a noisy "socket.send() raised 

1346 # exception". Skip the send when we've already been disconnected, and 

1347 # treat any error during send as a disconnect. 

1348 if closed.is_set() or ws.client_state != WebSocketState.CONNECTED: 

1349 return 

1350 try: 

1351 await ws.send_text(json.dumps(result)) 

1352 except (WebSocketDisconnect, RuntimeError): 

1353 return 

1354 finally: 

1355 reader_task.cancel()
