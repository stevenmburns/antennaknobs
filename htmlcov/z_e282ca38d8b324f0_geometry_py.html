1"""Translate the flat (p0, p1, n_seg, excitation) wire list used by 

2AntennaBuilder.build_wires() into the polyline + feed-arclength shape 

3that momwire's solver classes consume. 

4 

5The flat list expresses connectivity implicitly: two tuples are part of 

6the same electrical wire when they share an endpoint (within `eps`). 

7momwire wants each electrical wire as one (M, 3) polyline with junction 

8information (KCL at shared nodes) explicit; this module recovers the 

9graph, decomposes it into maximal chains between junction/endpoint 

10nodes, and emits the junction list. 

11 

12Supported topologies: 

13 * Open chains and any number of junctions of any degree (tees, X's, 

14 hentenna-style multi-junction, fandipole-style multi-spoke feeds). 

15 * Pure cycles (closed loops). A cycle is cut at one edge into two 

16 polylines joined by a junction at each cut node, so momwire's KCL 

17 carries the current around the loop. The cut edge is the loop's port 

18 edge when it has one (driven loops); for a PARASITIC loop, which 

19 radiates only through mutual coupling, the cut edge is arbitrary. 

20 A cycle with two or more port edges is not yet handled. 

21 * One or more excited segments per geometry (each becomes a delta-gap 

22 feed in the returned `feeds` list). The geometry as a whole must 

23 carry at least one excitation, but individual loop components need 

24 not (a parasitic loop is excited only by coupling). 

25""" 

26 

27from __future__ import annotations 

28 

29import numpy as np 

30 

31 

32def _round_point(p, eps): 

33 # Quantize endpoints onto an eps-spaced grid so 1e-14 floating-point 

34 # noise doesn't fragment what is logically a shared node. 

35 return tuple(round(float(c) / eps) * eps for c in p) 

36 

37 

38def flat_wires_to_polylines(tups, *, eps=1e-6): 

39 """Convert flat wire tuples to momwire polyline form. 

40 

41 Returns a dict with keys: 

42 polylines : list of (M, 3) np.ndarray 

43 edge_segments : list of list[int] — n_seg per edge per polyline 

44 feeds : list of (polyline_idx, arclength, voltage) — 

45 one entry per excited tuple, in registration 

46 order. Suitable to pass directly to a 

47 momwire solver's feeds=... kwarg. 

48 feed_wire_index : int — polyline holding the first excited 

49 segment (back-compat: feeds[0][0]) 

50 feed_arclength : float — arclength of the first feed 

51 (back-compat: feeds[0][1]) 

52 feed_voltage : complex — voltage of the first feed 

53 (back-compat: feeds[0][2]) 

54 junctions : list of list[(wire_idx, "start"|"end")] — 

55 shared-node groups, suitable to pass directly 

56 to a momwire solver's junctions=... kwarg. Empty list 

57 if every component is a simple path. 

58 """ 

59 if not tups: 

60 raise ValueError("no wires to translate") 

61 

62 # Build endpoint->node map and per-tuple edge list. 

63 node_of = {} 

64 nodes = [] # list of np.ndarray(3,) 

65 edges = [] # list of (a, b, n_seg, ev, tup_index) 

66 

67 def node_id(p): 

68 key = _round_point(p, eps) 

69 if key not in node_of: 

70 node_of[key] = len(nodes) 

71 nodes.append(np.asarray(p, dtype=float)) 

72 return node_of[key] 

73 

74 # Names are an optional 5th tuple field. None (or absent) means 

75 # "unnamed"; otherwise a string identifying this edge as a network 

76 # port in `build_network()`. 

77 tup_names = [] 

78 for i, t in enumerate(tups): 

79 if len(t) == 4: 

80 p0, p1, n_seg, ev = t 

81 name = None 

82 elif len(t) == 5: 

83 p0, p1, n_seg, ev, name = t 

84 else: 

85 raise ValueError(f"tuple {i}: expected 4- or 5-tuple, got {len(t)}") 

86 a = node_id(p0) 

87 b = node_id(p1) 

88 if a == b: 

89 raise ValueError(f"tuple {i}: degenerate edge (p0==p1 within eps)") 

90 edges.append((a, b, int(n_seg), ev, i)) 

91 tup_names.append(name) 

92 

93 # adj[nid] = list of (other_node, edge_index), in registration order. 

94 adj = [[] for _ in nodes] 

95 for ei, (a, b, _, _, _) in enumerate(edges): 

96 adj[a].append((b, ei)) 

97 adj[b].append((a, ei)) 

98 

99 for nid, neigh in enumerate(adj): 

100 if len(neigh) == 0: 

101 raise ValueError(f"node {nid} is isolated") 

102 

103 # Polyline boundaries are exactly the non-degree-2 nodes: degree-1 

104 # ends an open polyline, degree>=3 is a junction that ends one 

105 # polyline and starts another. Walk every edge out of every boundary 

106 # node, threading through degree-2 nodes until the next boundary. 

107 is_boundary = [len(a) != 2 for a in adj] 

108 edge_seen = [False] * len(edges) 

109 

110 polylines = [] 

111 edge_segments = [] 

112 # junction_ends[node_id] -> list of (polyline_index, "start"|"end"). 

113 # Filled as we walk; only meaningful for degree>=3 nodes, but we 

114 # collect it for all boundary nodes and filter later. 

115 junction_ends = {nid: [] for nid in range(len(nodes)) if is_boundary[nid]} 

116 # tup_index -> (polyline_index, edge_index_within) 

117 edge_to_polyline = {} 

118 

119 def walk_from(start_nid, first_edge): 

120 path_nodes = [start_nid] 

121 path_edges = [] 

122 prev_edge = None 

123 cur = start_nid 

124 next_edge = first_edge 

125 while True: 

126 edge_seen[next_edge] = True 

127 path_edges.append(next_edge) 

128 a, b, _, _, _ = edges[next_edge] 

129 nxt = b if a == cur else a 

130 path_nodes.append(nxt) 

131 cur = nxt 

132 if is_boundary[cur]: 

133 return path_nodes, path_edges 

134 prev_edge = next_edge 

135 # Degree-2 interior: take the unique outgoing edge. 

136 next_edge = None 

137 for _nb, ei in adj[cur]: 

138 if ei != prev_edge: 

139 next_edge = ei 

140 break 

141 assert next_edge is not None, f"degree-2 node {cur} had no continuation" 

142 

143 for start in range(len(nodes)): 

144 if not is_boundary[start]: 

145 continue 

146 for _nb, ei in adj[start]: 

147 if edge_seen[ei]: 

148 continue 

149 path_nodes, path_edges = walk_from(start, ei) 

150 

151 polyline_idx = len(polylines) 

152 polylines.append(np.stack([nodes[n] for n in path_nodes], axis=0)) 

153 edge_segments.append([edges[e][2] for e in path_edges]) 

154 for k, e in enumerate(path_edges): 

155 edge_to_polyline[edges[e][4]] = (polyline_idx, k) 

156 junction_ends[path_nodes[0]].append((polyline_idx, "start")) 

157 junction_ends[path_nodes[-1]].append((polyline_idx, "end")) 

158 

159 # Pure-cycle components — every node degree 2, no boundary to start 

160 # the walk from — are left untouched by the loop above. Cut each at 

161 # its excited edge: the excited edge becomes one polyline (A→B), the 

162 # rest of the cycle becomes a second polyline walked B→A the long 

163 # way. The two cut nodes A and B are each registered as 2-entry 

164 # junctions so momwire's KCL enforces current continuity around the 

165 # loop. 

166 while not all(edge_seen): 

167 seed = next(i for i, seen in enumerate(edge_seen) if not seen) 

168 # Flood the component reachable from `seed` through unseen edges. 

169 comp_edges = [] 

170 stack = [seed] 

171 in_comp = {seed} 

172 while stack: 

173 ei = stack.pop() 

174 comp_edges.append(ei) 

175 edge_seen[ei] = True 

176 a, b, _, _, _ = edges[ei] 

177 for endpoint in (a, b): 

178 for _nb, eo in adj[endpoint]: 

179 if eo not in in_comp and not edge_seen[eo]: 

180 in_comp.add(eo) 

181 stack.append(eo) 

182 

183 # A "port edge" carries either a voltage (legacy build_tls path) or a 

184 # network-spec name. To open the cycle we cut ONE edge into its own 

185 # polyline and register the two cut nodes as junctions, so momwire's KCL 

186 # enforces current continuity around the loop. We PREFER to cut at a 

187 # port edge: it has to become its own polyline anyway, to host the 

188 # delta-gap feed. A PARASITIC loop has no port edge, so the cut point 

189 # is arbitrary (any edge breaks the cycle) -- we cut the first one, and 

190 # since it carries no voltage/name it simply stays a passive polyline 

191 # whose only role is to anchor the two cut-node junctions. Any extra 

192 # port edges (a feed + a termination, as in a terminated rhombic/T2FD) 

193 # stay inside the long-way polyline and are registered as feeds by 

194 # arclength below, so a multi-port loop is handled too. 

195 excited_in_comp = [ 

196 ei 

197 for ei in comp_edges 

198 if edges[ei][3] is not None or tup_names[edges[ei][4]] is not None 

199 ] 

200 cut_ei = excited_in_comp[0] if excited_in_comp else comp_edges[0] 

201 cut_a, cut_b, cut_n_seg, _, cut_tup_idx = edges[cut_ei] 

202 

203 # Polyline 0: the cut edge alone, A → B. It hosts the delta-gap feed 

204 # when the cut was a port edge; for a parasitic loop it is just passive. 

205 cut_pl_idx = len(polylines) 

206 polylines.append(np.stack([nodes[cut_a], nodes[cut_b]], axis=0)) 

207 edge_segments.append([cut_n_seg]) 

208 edge_to_polyline[cut_tup_idx] = (cut_pl_idx, 0) 

209 

210 # Polyline 1 (long way): walk B → ... → A via the remaining edges. 

211 # The cut nodes are now polyline boundaries; the walker stops there. 

212 is_boundary[cut_a] = True 

213 is_boundary[cut_b] = True 

214 junction_ends.setdefault(cut_a, []) 

215 junction_ends.setdefault(cut_b, []) 

216 

217 # Undo the flood-fill's seen marks on the cycle remainder so the 

218 # walker can traverse them. Keep the cut edge marked since it's 

219 # already become polyline 0. 

220 for ei in comp_edges: 

221 if ei != cut_ei: 

222 edge_seen[ei] = False 

223 

224 first = next( 

225 (eo for _nb, eo in adj[cut_b] if eo != cut_ei and not edge_seen[eo]), 

226 None, 

227 ) 

228 # In a pure cycle every node has degree 2, so there's exactly one 

229 # remaining edge at cut_b after consuming the cut edge. 

230 assert first is not None, "cycle cut left no continuation" 

231 path_nodes, path_edges = walk_from(cut_b, first) 

232 loop_pl_idx = len(polylines) 

233 polylines.append(np.stack([nodes[n] for n in path_nodes], axis=0)) 

234 edge_segments.append([edges[e][2] for e in path_edges]) 

235 for k, e in enumerate(path_edges): 

236 edge_to_polyline[edges[e][4]] = (loop_pl_idx, k) 

237 

238 # Register the cut endpoints as junctions: the cut polyline has 

239 # path [A, B] so its start=A, end=B; loop polyline was walked 

240 # B → A so its start=B, end=A. 

241 junction_ends[cut_a].append((cut_pl_idx, "start")) 

242 junction_ends[cut_a].append((loop_pl_idx, "end")) 

243 junction_ends[cut_b].append((cut_pl_idx, "end")) 

244 junction_ends[cut_b].append((loop_pl_idx, "start")) 

245 

246 # Junctions = nodes where >= 2 polylines meet. Single-end records 

247 # (degree-1 free ends) and lone polyline starts aren't junctions. 

248 junctions = [ends for ends in junction_ends.values() if len(ends) >= 2] 

249 

250 # Locate the excitation(s) and convert each to (polyline_idx, 

251 # arclength, voltage). PyNEC feeds at segment `(n_seg+1)//2` of the 

252 # excited tuple — the middle segment 1-indexed, i.e. the physical 

253 # midpoint of the wire. The excited tuple is one edge of its 

254 # polyline; feed at that edge's midpoint. 

255 # 

256 # A tuple is a feed if either (a) it has a non-None ex value (legacy 

257 # voltage-driven feed) or (b) it has a non-None name (network-port 

258 # placeholder, voltage gets set later by `build_network()`). 

259 feeds = [] 

260 feed_names = [] 

261 for tup_index, edge in enumerate(edges): 

262 voltage = edge[3] 

263 name = tup_names[tup_index] 

264 if voltage is None and name is None: 

265 continue 

266 feed_pl, feed_edge_idx = edge_to_polyline[tup_index] 

267 polyline = polylines[feed_pl] 

268 edge_lengths = np.linalg.norm(np.diff(polyline, axis=0), axis=1) 

269 feed_arclength = float( 

270 edge_lengths[:feed_edge_idx].sum() + 0.5 * edge_lengths[feed_edge_idx] 

271 ) 

272 feeds.append( 

273 (feed_pl, feed_arclength, complex(voltage if voltage is not None else 0)) 

274 ) 

275 feed_names.append(name) 

276 

277 if not feeds: 

278 raise ValueError("no excitation found in wire list") 

279 

280 return { 

281 "polylines": polylines, 

282 "edge_segments": edge_segments, 

283 "feeds": feeds, 

284 "feed_names": feed_names, 

285 # Back-compat scalars — first feed. 

286 "feed_wire_index": feeds[0][0], 

287 "feed_arclength": feeds[0][1], 

288 "feed_voltage": feeds[0][2], 

289 "junctions": junctions, 

290 }
