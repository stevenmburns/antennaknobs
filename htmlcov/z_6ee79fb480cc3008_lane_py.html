1"""Per-session single-lane solve scheduling (issue #382). 

2 

3One :class:`SolveLane` per client session serializes every solve-producing 

4compute — the live ``/ws`` solve, each ``/sweep`` chunk, each ``/converge`` 

5point, ``/norm_check``, ``/pattern``, ``/pattern_metrics``. Holding the lane 

6is the only way to compute, so "no two solver computations run concurrently 

7for one session" is structural, not a client-side heuristic. 

8 

9Turns are granted by ``(priority, arrival)`` — the live solve always 

10outranks background batches — and carry the session's *generation*: the 

11client's monotonic ``_seq``/``_gen`` counter. Newer generations supersede 

12older work (queued turns raise :class:`Superseded`; the running turn's 

13CancelToken is tripped so it aborts at the next momwire solver checkpoint). 

14A turn of the same kind supersedes its predecessor regardless of generation: 

15the client re-issued the job, so the old stream is already abandoned. 

16""" 

17 

18from __future__ import annotations 

19 

20import asyncio 

21import itertools 

22from contextlib import asynccontextmanager 

23 

24# Lane priority per job kind (lower wins). The live solve draws the heatmap 

25# the user is staring at; norm-check/pattern are single dwell-triggered 

26# solves; sweep/converge are long batches that should soak up whatever the 

27# lane has left. Unknown kinds sort last. 

28PRIORITY = { 

29 "live": 0, 

30 "norm_check": 1, 

31 "pattern": 1, 

32 "pattern_metrics": 1, 

33 "sweep": 2, 

34 "converge": 2, 

35} 

36_PRIORITY_DEFAULT = 9 

37 

38# Kinds where a new arrival supersedes an older one of the same kind: the 

39# client holds at most one of each (re-issuing means the old stream is 

40# abandoned). pattern_metrics is deliberately absent — the compare table 

41# legitimately runs one request per design row at once. 

42SAME_KIND_SUPERSEDES = frozenset({"live", "sweep", "converge", "norm_check", "pattern"}) 

43

45class Superseded(Exception): 

46 """Newer client intent overtook this turn before it ran; abandon the job. 

47 

48 Raised out of ``lane.turn(...)`` — never after the turn was granted (a 

49 running turn is preempted via its CancelToken instead, surfacing as 

50 ``momwire.SolveAborted`` from the solver). 

51 """ 

52 

53 

54def _default_token_factory(): 

55 # Deferred so the lane primitive (and its unit tests) don't pay the 

56 # momwire import; the server passes real CancelTokens into the solvers. 

57 import momwire 

58 

59 return momwire.CancelToken()

62class _Turn: 

63 __slots__ = ("kind", "gen", "order", "ready", "token") 

64 

65 def __init__(self, kind: str, gen: int | None, order: int, token) -> None: 

66 self.kind = kind 

67 self.gen = gen 

68 self.order = order 

69 self.token = token 

70 self.ready: asyncio.Future[None] = asyncio.get_running_loop().create_future() 

71 

72 @property 

73 def rank(self) -> tuple[int, int]: 

74 return (PRIORITY.get(self.kind, _PRIORITY_DEFAULT), self.order)

77class SolveLane: 

78 """One serialized compute lane: a priority-ordered, generation-aware 

79 turnstile. All bookkeeping happens in synchronous stretches on the event 

80 loop (no locks needed); the only awaits are on a turn's grant future.""" 

81 

82 def __init__(self, token_factory=None) -> None: 

83 self._token_factory = token_factory or _default_token_factory 

84 self.gen = 0 # highest client generation seen 

85 self._order = itertools.count() 

86 self._running: _Turn | None = None 

87 self._waiting: list[_Turn] = [] 

88 

89 @property 

90 def idle(self) -> bool: 

91 return self._running is None and not self._waiting 

92 

93 @asynccontextmanager 

94 async def turn(self, kind: str, gen: int | None = None): 

95 """Wait for exclusive compute rights; yield this turn's CancelToken. 

96 

97 Raises :class:`Superseded` (possibly immediately — a request carrying 

98 a generation older than the lane's is stale on arrival) if newer 

99 intent overtakes this turn while it is queued. 

100 """ 

101 t = _Turn(kind, gen, next(self._order), self._token_factory()) 

102 self._supersede_overtaken_by(t) 

103 if t.gen is not None and t.gen < self.gen: 

104 raise Superseded # stale on arrival: a newer generation exists 

105 self._waiting.append(t) 

106 self._grant_if_free() 

107 try: 

108 await t.ready 

109 except BaseException: 

110 # Superseded while queued, or the awaiting task itself was 

111 # cancelled (client torn down before the grant — or, in a narrow 

112 # race, just after it: release then too). 

113 if t in self._waiting: 

114 self._waiting.remove(t) 

115 self._release(t) 

116 raise 

117 try: 

118 yield t.token 

119 finally: 

120 self._release(t) 

121 

122 def advance(self, gen: int | None) -> None: 

123 """Fold newer client intent into the lane without taking a turn. 

124 

125 The /ws reader calls this the moment a newer request lands, so an 

126 older running compute (e.g. a benchmark-mesh sweep chunk) is 

127 preempted at its next checkpoint right away — the superseding live 

128 turn itself won't be admitted until the solver loop gets to it, and 

129 waiting for that would leave the stale chunk grinding meanwhile. 

130 """ 

131 if gen is None or gen <= self.gen: 

132 return 

133 self.gen = gen 

134 for w in list(self._waiting): 

135 if w.gen is not None and w.gen < self.gen: 

136 self._waiting.remove(w) 

137 w.ready.set_exception(Superseded()) 

138 r = self._running 

139 if r is not None and r.gen is not None and r.gen < self.gen: 

140 r.token.cancel() 

141 

142 def _supersede_overtaken_by(self, new: _Turn) -> None: 

143 if new.gen is not None and new.gen > self.gen: 

144 self.gen = new.gen 

145 for w in list(self._waiting): 

146 if self._overtaken(w, new): 

147 self._waiting.remove(w) 

148 w.ready.set_exception(Superseded()) 

149 r = self._running 

150 if r is not None and self._overtaken(r, new): 

151 # Preempt mid-compute: the solver raises SolveAborted at its next 

152 # checkpoint. (PyNEC solves have no checkpoints — they run out 

153 # their current call; admission bounds how long that can be.) 

154 r.token.cancel() 

155 

156 def _overtaken(self, old: _Turn, new: _Turn) -> bool: 

157 if old.kind == new.kind and old.kind in SAME_KIND_SUPERSEDES: 

158 return True # re-issued job: the old stream is abandoned 

159 return old.gen is not None and old.gen < self.gen 

160 

161 def _grant_if_free(self) -> None: 

162 if self._running is not None or not self._waiting: 

163 return 

164 t = min(self._waiting, key=lambda w: w.rank) 

165 self._waiting.remove(t) 

166 self._running = t 

167 t.ready.set_result(None) 

168 

169 def _release(self, t: _Turn) -> None: 

170 if self._running is t: 

171 self._running = None 

172 self._grant_if_free()

175class LaneRegistry: 

176 """SolveLanes keyed by client session id. 

177 

178 Sessions are minted client-side (one UUID per workbench tab) and stamped 

179 on every request as ``_session``. A request without one (curl, scripts, 

180 old clients) gets a private throwaway lane: admission still applies, 

181 serialization doesn't — cross-session coupling is explicitly not this 

182 module's job. Idle lanes are dropped on turn exit, so disconnect-only 

183 traffic can't leak entries. 

184 """ 

185 

186 def __init__(self, token_factory=None) -> None: 

187 self._token_factory = token_factory 

188 self._lanes: dict[str, SolveLane] = {} 

189 

190 def lane(self, session: str | None) -> SolveLane: 

191 if session is None: 

192 return SolveLane(self._token_factory) 

193 found = self._lanes.get(session) 

194 if found is None: 

195 found = self._lanes[session] = SolveLane(self._token_factory) 

196 return found 

197 

198 def advance(self, session: str | None, gen: int | None) -> None: 

199 """See :meth:`SolveLane.advance`. No-op for unknown/sessionless keys 

200 (nothing to preempt — a lane only exists while work is in it).""" 

201 if session is None: 

202 return 

203 found = self._lanes.get(session) 

204 if found is not None: 

205 found.advance(gen) 

206 

207 @asynccontextmanager 

208 async def turn(self, session: str | None, kind: str, gen: int | None = None): 

209 # No await between the lookup and the waiter registering inside 

210 # lane.turn's entry, so a lane can't be reaped out from under a 

211 # not-yet-registered waiter. 

212 found = self.lane(session) 

213 try: 

214 async with found.turn(kind, gen) as token: 

215 yield token 

216 finally: 

217 if session is not None and found.idle: 

218 self._lanes.pop(session, None) 

219 

220 def __len__(self) -> int: # observability + tests 

221 return len(self._lanes) 

222

224@asynccontextmanager 

225async def cancel_on_disconnect(request, token, interval: float = 0.25): 

226 """Trip ``token`` if the HTTP client goes away mid-compute. 

227 

228 The streaming endpoints only used to notice a disconnect *between* 

229 chunks — useless when one benchmark-mesh chunk takes minutes. This 

230 watcher polls ``request.is_disconnected()`` for the duration of a turn, 

231 so an abandoned tab stops the burn at the next solver checkpoint. 

232 """ 

233 

234 async def _watch() -> None: 

235 while not await request.is_disconnected(): 

236 await asyncio.sleep(interval) 

237 token.cancel() 

238 

239 task = asyncio.create_task(_watch()) 

240 try: 

241 yield 

242 finally: 

243 task.cancel()
