1"""Standard library of station building blocks (issue #489): matchboxes, 

2transformers, and pass-throughs as reusable `Composite` components. 

3 

4Each factory is an ordinary Python function returning a `Composite` — the 

5function arguments ARE the component's parameter list ("generators are 

6code, modules are data"). Designs instantiate them by name with a 

7formal/actual port map: 

8 

9 from antennaknobs.network import Instance 

10 from antennaknobs.station import t_network_tuner, bypass 

11 

12 branches = [ 

13 Instance("tuner", t_network_tuner(c1_pF=..., c2_pF=..., l_uH=...), 

14 rig="rig", out="li"), 

15 TL.from_cable("openwire-600", "li", "feed", 20.0), 

16 ] 

17 

18Swapping a box for `bypass()` (same two-port interface, wires straight 

19through) turns any "with/without the matchbox" comparison into a one-line 

20change. 

21 

22Units are radio-work units — picofarads and microhenries, matching the 

23design-knob conventions (`series_c1_pF`, `lmag_uH`, …) — converted to the 

24branch classes' SI at construction. Ohms and Q are dimensionless-as-usual. 

25""" 

26

32def bypass() -> Composite: 

33 """A two-port that wires its input straight to its output — the 

34 pass-through with a matchbox's interface (formals ``a``/``b``). 

35 Implemented as a pure alias (node merge), not a 0 Ω element: no extra 

36 MNA unknown, no budget row, electrically *identical* to not being 

37 there. Use it to A/B a station with and without its tuner/balun 

38 without touching anything else.""" 

39 return Composite(ports=("a", "b"), aliases=(("a", "b"),))

42def t_network_tuner( 

43 c1_pF: float, c2_pF: float, l_uH: float, ql: float | None = None 

44) -> Composite: 

45 """The classic T-network ("high-pass tee") antenna tuner: series C1 

46 from ``rig`` to an internal tee midpoint, shunt L to common at the 

47 midpoint, series C2 on to ``out``. `ql` gives the coil a finite Q 

48 (R = ωL/Q, issue #298) — the coil is where a real T-network burns 

49 its watts. Formals: ``rig`` (transmitter side), ``out`` (line side).""" 

50 return Composite( 

51 ports=("rig", "out"), 

52 branches=( 

53 TwoPort(a="rig", b="m", c=c1_pF * 1e-12), 

54 Shunt(port="m", l=l_uH * 1e-6, ql=ql), 

55 TwoPort(a="m", b="out", c=c2_pF * 1e-12), 

56 ), 

57 )

60def l_network_tuner( 

61 series_l_uH: float, shunt_c_pF: float, ql: float | None = None 

62) -> Composite: 

63 """L-match: series L from ``rig`` to ``out``, shunt C across ``out`` 

64 (the load side — the arrangement that steps a higher load R down to 

65 the rig). Degenerate values are physics, not errors (issue #285): a 

66 0 H series arm is an ideal short and a 0 F shunt is an open, so both 

67 arms at zero make this an inert pass-through. Formals: ``rig``, 

68 ``out``.""" 

69 return Composite( 

70 ports=("rig", "out"), 

71 branches=( 

72 TwoPort(a="rig", b="out", l=series_l_uH * 1e-6, ql=ql), 

73 Shunt(port="out", c=shunt_c_pF * 1e-12), 

74 ), 

75 )

78def unun( 

79 turns: float, 

80 lmag_uH: float | None = None, 

81 qlmag: float | None = None, 

82 comp_c_pF: float | None = None, 

83) -> Composite: 

84 """Step-down unun (the EFHW / OCF box): an ideal ``turns``:1 

85 transformer — the ``line`` side sees Z_ant/turns² — with the minimal 

86 loss model of `Transformer` (magnetizing inductance ``lmag_uH`` shunted 

87 across the line side, finite-Q core loss ``qlmag``), plus the optional 

88 compensation capacitor ``comp_c_pF`` across the line-side terminals 

89 that commercial 49:1 builds carry. Formals: ``line`` (rig/feedline 

90 side), ``ant`` (high-Z antenna side).""" 

91 lmag = lmag_uH * 1e-6 if lmag_uH is not None else None 

92 branches: tuple = ( 

93 Transformer(a="line", b="ant", n=1.0 / turns, lmag=lmag, qlmag=qlmag), 

94 ) 

95 if comp_c_pF: 

96 branches += (Shunt(port="line", c=comp_c_pF * 1e-12),) 

97 return Composite(ports=("line", "ant"), branches=branches)

100def balun( 

101 n: float, lmag_uH: float | None = None, qlmag: float | None = None 

102) -> Composite: 

103 """Balun as an ideal ``a:b`` ratio transformer with the minimal loss 

104 model (magnetizing branch on the ``line`` side): ``n`` is the 

105 line:antenna voltage ratio, so a 4:1 impedance balun stepping a 

106 ~300 Ω feed down to ~75 Ω line is ``balun(n=0.5)`` — same convention 

107 as `Transformer` itself. Formals: ``line``, ``ant``.""" 

108 lmag = lmag_uH * 1e-6 if lmag_uH is not None else None 

109 return Composite( 

110 ports=("line", "ant"), 

111 branches=(Transformer(a="line", b="ant", n=n, lmag=lmag, qlmag=qlmag),), 

112 )
