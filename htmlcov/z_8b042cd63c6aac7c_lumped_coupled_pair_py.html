1"""Lumped-coupled dipole pair — the TwoPort-branch showcase for issue #65. 

2 

3Two parallel half-wave dipoles a fraction of a wavelength apart. Only the 

4front one is fed; the rear one is a parasite reachable *only* through a lumped 

5series R+jωL element bridging the two feed segments (a `TwoPort` branch). By 

6tuning that coupling impedance you steer current onto the parasite with a 

7chosen magnitude and phase, turning the pair into a small end-fire beam — 

8the lumped analogue of a reactance-loaded parasitic element. 

9 

10This is the smallest geometry that exercises a `TwoPort` as a genuine 2-port 

11between two distinct antenna segments (not a self-load on one segment, which 

12is what `Load` covers). It is the cross-engine cross-check for issue #65 

13piece (B): MomwireEngine and PyNECEngine both reduce the branch through the 

14shared `NetworkReducer` stamp, and PyNECEngine with ``native_nt=True`` instead 

15bakes a real NEC2 `nt_card` and solves it simultaneously with the MoM currents 

16— an independent oracle, exactly as `build_tls()` → native `tl_card` is the 

17oracle for the TL branch. All three agree on impedance and far field. 

18 

19`coupling_r_ohm` / `coupling_l_uH` set the bridge impedance; large R (≳1 kΩ) 

20opens the branch and recovers the isolated driven dipole (the base case the 

21cross-check pins first). 

22""" 

23 

24from ... import AntennaBuilder 

25from ...network import Driven, Network, PortOnWire, TwoPort 

26 

27from types import MappingProxyType 

28 

29 

30class Builder(AntennaBuilder): 

31 default_params = MappingProxyType( 

32 { 

33 "design_freq": 28.0, 

34 "freq": 28.0, 

35 # Dipole half-arm as a fraction of λ/4 (1.0 = resonant half-wave). 

36 "length_factor": 0.95, 

37 # Front-to-rear spacing as a fraction of λ. 

38 "spacing_factor": 0.13, 

39 # Lumped series R+jωL bridging the two feed segments. 

40 "coupling_r_ohm": 20.0, 

41 "coupling_l_uH": 0.10, 

42 # Height above the origin plane (kept out of any ground image). 

43 "base": 7.0, 

44 } 

45 ) 

46 

47 def build_wires(self): 

48 wavelength = 299.792458 / self.design_freq 

49 half_arm = 0.25 * wavelength * self.length_factor 

50 spacing = self.spacing_factor * wavelength 

51 z = self.base 

52 tups = [] 

53 for x, name in ((0.0, "front"), (-spacing, "rear")): 

54 # One straight wire per dipole, feed edge at the centre. No `ev` — 

55 # the Network supplies the source (front) and the coupling terminal 

56 # (rear); `name` marks each centre segment for PortOnWire lookup. 

57 tups.append(((x, -half_arm, z), (x, half_arm, z), 21, None, name)) 

58 return tups 

59 

60 def build_network(self): 

61 return Network( 

62 ports={ 

63 "front": PortOnWire("front"), 

64 "rear": PortOnWire("rear"), 

65 }, 

66 branches=[ 

67 TwoPort( 

68 a="front", 

69 b="rear", 

70 r=self.coupling_r_ohm, 

71 l=self.coupling_l_uH * 1e-6, 

72 ), 

73 ], 

74 sources=[Driven(port="front", voltage=1 + 0j)], 

75 )
