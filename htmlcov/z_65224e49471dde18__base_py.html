1"""Base types for the antenna-example registry. 

2 

3An `AntennaExample` bundles everything the web layer needs to serve one 

4geometry: parameter schema, momwire solve/sweep, and pynec build/solve. Any 

5callable left as None signals the backend doesn't support that operation 

6for this geometry. 

7 

8Keeping these as plain callables (rather than a class hierarchy) means 

9each example module is a flat file of functions plus one EXAMPLE = ... 

10assignment at the bottom — easy to read, easy to delete. 

11""" 

12 

13from __future__ import annotations

15from dataclasses import dataclass, field 

16from typing import Any, Callable, Optional, Tuple, Union

18# A pynec "build" returns the dict-of-context-plus-derived-geom that both 

19# solve() and pattern() consume — see web/pynec_backend.py for the shape. 

20PynecBuildFn = Callable[[dict], dict] 

21 

22# momwire solve / pynec solve: take a request dict, return the response dict 

23# the frontend renders. 

24SolveFn = Callable[[dict], dict] 

25 

26# momwire sweep: take a request dict + frequency list. Single-feed geometries 

27# return (re, im) — two parallel lists of input impedance. Multi-feed 

28# geometries (bowtie 1×2) return (primary_re, primary_im, feeds_re, 

29# feeds_im) where feeds_* is (n_freqs × n_feeds). The sweep endpoint 

30# detects the tuple length to know which shape it got. 

31SweepResult = Union[ 

32 Tuple[list[float], list[float]], 

33 Tuple[list[float], list[float], list[list[float]], list[list[float]]], 

34] 

35SweepFn = Callable[[dict, list[float]], SweepResult] 

36 

37# pynec pattern excitation: drive the NEC context using the build dict's 

38# feed metadata. Single-feed default lives in web.pynec_backend.pattern(); 

39# multi-feed examples (bowtie) supply their own (e.g. two ex_card calls). 

40PynecPatternExciteFn = Callable[[dict, float], None] # (build, freq_mhz)

43@dataclass(frozen=True) 

44class ParamSpec: 

45 """One UI-exposed parameter for a geometry. 

46 

47 Served from `GET /examples` so the frontend can render parameter 

48 controls generically — one ParamForm component reads the schema and 

49 builds the matching <input> elements without per-antenna JSX. 

50 

51 `visible_when` makes a parameter conditional on another's value: 

52 e.g. yagi's director_spacing_wavelengths slider only appears when 

53 n_directors > 0. The dict is shaped {name, op, value}; ops include 

54 "eq", "ne", "gt", "ge", "lt", "le". 

55 

56 `kind` switches the input type: 

57 - "float": continuous slider (use min/max/step) 

58 - "int": integer slider (use min/max/step=1) 

59 - "bool": checkbox 

60 - "enum": <select> dropdown; supply `enum_options` 

61 Bowtie's phase_lr_deg uses a signed range — kind="float" with min<0. 

62 

63 For "enum" params, `enum_options` is a tuple of free-form dicts. Each 

64 dict must have at least {"value", "label"}; antennas can attach 

65 arbitrary extra keys to drive related controls. Fan_dipole's per-band 

66 `band_id` carries {"value": "20m", "label": "20m", "freq_min": 14.0, 

67 "freq_max": 14.35, "freq_default": 14.175} so a sibling freq slider 

68 can read its bounds + default off the active band. 

69 

70 `range_from_enum_option` lets a slider's min/max/step come from the 

71 currently-selected enum value of a sibling param. Shape: 

72 {"param": <enum_param_name>, "min_key": <key>, "max_key": <key>}. 

73 

74 `on_change_set` is a side-effect rule: when this enum changes value, 

75 write a sibling param using a key from the new enum option. Shape: 

76 {"set": <sibling_name>, "from_enum_key": <key>}. Fan_dipole's 

77 band_id uses {"set": "freq", "from_enum_key": "freq_default"} so 

78 flipping the band pulldown snaps the freq slider to that band's 

79 centre. 

80 

81 `linked_to_design_freq` means this param's value also drives the 

82 global designFreq state on the frontend. Fan_dipole uses it on 

83 `bands[0].freq` so the first band's frequency stays the antenna's 

84 design frequency without a separate global slider. 

85 

86 `link_meas_freq_to_param` is the flat-schema sibling of 

87 ParamGroupSpec.link_meas_freq_to_param: when this scalar slider 

88 moves, push the current value of the named sibling param into the 

89 global measFreq state (gated by the frontend's linkMeas toggle). 

90 Used by multi-band antennas with parallel `length_NN` / `freq_NN` 

91 params so dragging length_20 jumps the measurement-freq slider to 

92 the 20m band's freq. A self-reference (param links to itself, e.g. 

93 freq_20 → freq_20) is the natural way to mark "this slider's own 

94 value is a measurement-frequency anchor". 

95 """ 

96 

97 name: str 

98 label: str 

99 default: Any 

100 kind: str = "float" # float | int | bool | enum 

101 min: Optional[float] = None 

102 max: Optional[float] = None 

103 step: Optional[float] = None 

104 precision: int = 3 # decimal places shown in the value readout 

105 unit: Optional[str] = None # rendered next to the value: "°", "m", "λ", etc. 

106 visible_when: Optional[dict] = None # {"name": ..., "op": ..., "value": ...} 

107 enum_options: Optional[tuple[dict, ...]] = None 

108 range_from_enum_option: Optional[dict] = None 

109 on_change_set: Optional[dict] = None 

110 linked_to_design_freq: bool = False 

111 link_meas_freq_to_param: Optional[str] = None 

112 sweepable: bool = False 

113 # Optional explicit placement in the param grid, opting this knob out 

114 # of the default auto-flow packing. Shape: {"row", "col", "row_span", 

115 # "col_span"} — all optional ints, 1-indexed (CSS grid lines). The 

116 # frontend maps them straight onto grid-row / grid-column so a design 

117 # can lay its knobs out in a deliberate row/column arrangement. Pairs 

118 # with AntennaExample.layout["columns"], which fixes the grid's column 

119 # count so these positions are stable rather than width-dependent. 

120 # Authored under ui_params[<param>]["layout"]; None keeps auto-flow. 

121 layout: Optional[dict] = None

124@dataclass(frozen=True) 

125class ParamGroupSpec: 

126 """A repeating section of params, count-controlled by another param. 

127 

128 Used for multi-band / multi-element antennas where each band (or 

129 element) carries its own set of knobs. Fan_dipole is the first user 

130 (one group per band: band_id + freq + length_factor). The pattern 

131 generalises: a trapped multi-bander would have a group with 

132 {band_id, trap_freq, trap_q}; a multi-band Yagi would have a group 

133 with {driver_factor, reflector_factor, n_directors, ...} possibly 

134 containing its own nested director sub-group. 

135 

136 The schema currently allows nested groups (params is a union type) 

137 but the frontend only renders one level of nesting in the first 

138 cut. Adding deeper nesting is a frontend-only change. 

139 

140 `label_template` is the heading shown above each instance with 

141 {i} substituted to the 0-indexed instance number, e.g. "band {i}". 

142 

143 `repeat_count` names a scalar param (kind="int") elsewhere in the 

144 same schema whose value determines how many instances render. 

145 `max_repeats` is the absolute cap used for state pre-allocation — 

146 matching the count param's `max`. 

147 """ 

148 

149 name: str # key under which the frontend stores per-instance values 

150 label_template: str # e.g. "band {i}" — {i} is the 0-indexed instance 

151 repeat_count: str # name of the scalar param that controls how many 

152 max_repeats: int 

153 params: tuple[Any, ...] # tuple[Union[ParamSpec, ParamGroupSpec], ...] 

154 # Optional per-instance default overrides. Index i applies to 

155 # instance i; each dict maps {param_name: default_value} and beats 

156 # the ParamSpec.default for that one instance. Used by fan_dipole 

157 # so the 5 max bands seed to (20m, 10m, 17m, 12m, 15m) instead of 

158 # all defaulting to the same band. Tuple length doesn't need to 

159 # match max_repeats; missing indices fall back to the ParamSpec 

160 # default. 

161 default_overrides: tuple[dict, ...] = () 

162 # When set, names a param inside `params` (e.g. "freq") whose 

163 # current per-instance value drives the global measFreq state 

164 # *every time any leaf in the touched instance changes*. UX 

165 # rationale: tuning any knob of band i (length factor, t0 factor, 

166 # ...) signals the user is focused on band i, so the live 

167 # simulation should track that band's freq. Distinct from 

168 # ParamSpec.linked_to_design_freq, which is a steady-state link 

169 # to a single fixed instance (always band 0). This is a 

170 # per-interaction "follow the user's focus" link, fan_dipole and 

171 # hexbeam_5band both use "freq". 

172 # Gated by the frontend's linkMeas toggle. 

173 link_meas_freq_to_param: Optional[str] = None

194@dataclass(frozen=True) 

195class SweepPolicy: 

196 """Where to centre the freq sweep and how wide to make it. 

197 

198 `anchor` picks which scalar the sweep range is anchored to: 

199 - "design_freq": sweep around the antenna's design frequency 

200 (the wider out-of-band picture; default for single-band antennas). 

201 - "meas_freq": sweep around the current measurement frequency 

202 (multi-band antennas like fan_dipole — keeps the trace focused 

203 on the band the user is currently tuning, since the design 

204 frequency stays pinned to band 0). 

205 

206 `lo_factor` / `hi_factor` are multiplicative bounds applied to the 

207 anchor. Defaults give a broad resonance/out-of-band view; multi-band 

208 antennas narrow to ±5% so the trace doesn't cross into neighbouring 

209 bands the user isn't tuning. 

210 """ 

211 

212 anchor: str = "design_freq" # "design_freq" | "meas_freq" 

213 lo_factor: float = 0.8 

214 hi_factor: float = 1.25 

215 # When True, the frontend snaps the sweep range to the [min_mhz, 

216 # max_mhz] of the band whose range contains the current anchor 

217 # frequency (typically measFreq for multi-band antennas). Falls 

218 # back to lo_factor/hi_factor multiplicatively if the anchor sits 

219 # outside every band. Used by fandipole so the sweep trace stays 

220 # inside the amateur band the user is currently tuning instead of 

221 # bleeding into the adjacent bands. 

222 band_locked: bool = False 

223 

224 

225DEFAULT_SWEEP_POLICY = SweepPolicy() 

226

262class ResultGroupSpec: 

263 """A repeating result row, one instance per element of a top-level 

264 array field on the solve response. 

265 

266 The number of instances comes from the length of the first inner

268

286@dataclass(frozen=True) 

287class AntennaExample: 

288 name: str 

289 label: str 

290 momwire_solve: SolveFn 

291 momwire_sweep: SweepFn 

292 # Geometry-only snapshot (wires, feed marker; no solve, no currents) for a 

293 # fast antenna-shape preview while the real solve runs. Optional so an 

294 # example without one degrades to "no preview". 

295 momwire_geometry: Optional[SolveFn] = None 

296 # Cheap geometry-only count of the wire segments (≈ MoM basis functions / 

297 # matrix dimension N) a solve request would build, used to reject solves 

298 # that would allocate an N×N matrix too large/slow for the live engine. None 

299 # if the geometry can't be built (the real solve surfaces that error). Takes 

300 # the same req dict as momwire_solve. 

301 count_basis: Optional[Callable[[dict], Optional[int]]] = None 

302 # Recommended default solver backend for this design (e.g. "arrayblock" for 

303 # grid arrays, where it's far faster than the dense default). None lets the 

304 # UI keep its own default. Surfaced in the /examples schema; the frontend 

305 # seeds the active slot's backend from it on antenna selection. 

306 default_backend: Optional[str] = None 

307 pynec_build: Optional[PynecBuildFn] = None 

308 pynec_solve: Optional[SolveFn] = None 

309 # Render the geometry as a NEC2 .nec card deck (str) for the current 

310 # request (params/variant/freq/ground). None when the design has no 

311 # faithful native-NEC representation (TL/DiffTL/virtual-driver networks). 

312 nec_export: Optional[Callable[[dict], str]] = None 

313 # Serialise the request's current knob values back to a paste-ready 

314 # Python `default_params = {...}` (or named-variant) block, so the UI can 

315 # offer a "Copy Python" action that drops straight into a design file. 

316 # None when the design can't be serialised. 

317 params_source: Optional[Callable[[dict], str]] = None 

318 # Compute scalar far-field metrics (peak gain, takeoff, F/B, beamwidths) 

319 # for the request's antenna, so the UI can show a side-by-side table when 

320 # comparing pinned patterns. Always the momwire engine (matches the 

321 # client-derived lobe on screen). None when the design can't be evaluated. 

322 far_field_metrics: Optional[Callable[[dict], dict]] = None 

323 # When set, pattern() calls this to excite the NEC context (e.g. 

324 # multi-source drive for bowtie's 1×2 array). When None, pattern() 

325 # uses the default single-feed excitation reading b["feed_seg"] / 

326 # b["feed_tag"] / b["n_per_wire"] from the build dict. 

327 pynec_pattern_excite: Optional[PynecPatternExciteFn] = None 

328 # Sweep returns the multi-feed 4-tuple instead of the single-feed 

329 # 2-tuple. Sweep endpoint uses this to dispatch the two response 

330 # streaming shapes. 

331 multi_feed: bool = False 

332 # Mixed sequence of ParamSpec (scalar) and ParamGroupSpec (repeat). 

333 # The Any erases the union; runtime discrimination is by presence of 

334 # `params` (groups have it, scalars don't). 

335 param_schema: tuple[Any, ...] = field(default_factory=tuple) 

336 # Mixed sequence of ResultFieldSpec (scalar row) and ResultGroupSpec 

337 # (repeat group). Runtime discrimination on presence of `fields`. 

338 result_schema: tuple[Any, ...] = field(default_factory=tuple) 

339 # Design-frequency band tabs offered by the UI. Defaults to the HF 

340 # amateur set; multi-band examples (fan_dipole) set this to () to 

341 # suppress the row. 

342 bands: tuple[BandSpec, ...] = DEFAULT_HF_BANDS 

343 # Optional override for the measurement-freq slider span. When None, 

344 # the UI uses a generic ±20%/+25% window around the design freq. 

345 # Multi-band examples that span the whole HF range set this to e.g. 

346 # (13.5, 30.2) so the slider can reach every band. 

347 meas_freq_range_mhz: Optional[tuple[float, float]] = None 

348 # Sweep range + anchor for the live freq sweep. Defaults to 

349 # design-freq-anchored ±20%/+25%; multi-band examples override 

350 # to meas-freq-anchored ±5% so the trace stays on the band the 

351 # user is tuning. 

352 sweep_policy: SweepPolicy = DEFAULT_SWEEP_POLICY 

353 # Initial 2D-view projection the wire-render canvas picks when the 

354 # user first selects this example. "xy" = top-down (beam-in-xy 

355 # antennas like yagi/moxon/hexbeam); "yz" = side (antennas whose 

356 # arms run along y and droop in z, like inverted_v / fan_dipole; 

357 # also the vertical-loop hentenna). The user can still override 

358 # via the projection buttons; this just sets the starting view. 

359 # None for a deferred (user) design with no override: the real view is 

360 # auto-detected and arrives with the first geometry/solve response, so the 

361 # frontend holds the current camera until then instead of snapping to a 

362 # provisional value. 

363 default_view: Optional[str] = None 

364 # The frequency this antenna is naturally designed for, in MHz. 

365 # Used by the frontend to snap the band selector + design-freq / 

366 # meas-freq sliders to the right starting value when the example 

367 # first loads, without exposing it as a redundant schema slider. 

368 # None means "no preferred freq" — frontend falls back to bands[0] 

369 # or its built-in default. 

370 default_freq_mhz: Optional[float] = None 

371 # True when the Builder has a `design_freq` parameter that scales 

372 # geometry (design_freq-sized designs). The frontend uses this to show 

373 # or hide the design-freq band-tab row: hand-tuned absolute 

374 # designs (top-level designs/) ignore design_freq, so the row 

375 # would just be a noop slider for them. 

376 has_design_freq: bool = False 

377 # Alternate seed dicts available on the Builder — class-level 

378 # attributes named `<variant>_params` (e.g. `default`, `opt`, 

379 # `original`, `z50`, `z100`). The bare name is what the frontend 

380 # sends back in the `variant` field of solve / sweep requests. 

381 # `default` is always first when present so the UI lists it as 

382 # the canonical starting point. Single-variant Builders ship 

383 # ('default',) — the frontend can suppress the selector then. 

384 variants: tuple[str, ...] = ("default",) 

385 # Per-variant param values, keyed by variant name → {param_name: 

386 # value}. The frontend uses this to reset schema sliders + design 

387 # freq when the user picks a different variant. Complex-valued 

388 # params are encoded as {"re": ..., "im": ...}; ui_params is 

389 # stripped. Single-variant designs ship {} since there's nothing 

390 # to switch between. 

391 variant_values: dict = field(default_factory=dict) 

392 # Per-variant UI-hint overrides, keyed by variant name → {hint_name: 

393 # value}. Today the only hint is `sweep_policy` (a SweepPolicy), emitted 

394 # only for variants whose derived policy differs from the design-level 

395 # `sweep_policy` above; the frontend falls back to that top-level value 

396 # for any variant absent here. Kept as an open map so more ui hints can 

397 # move per-variant later without another /examples contract change. 

398 variant_ui: dict = field(default_factory=dict) 

399 # Optional grid-level layout config for the top-level knob rail. 

400 # Recognised key today: {"columns": int} — pins the param grid to a 

401 # fixed column count so per-ParamSpec.layout col positions land 

402 # predictably (the default auto-flow grid varies its column count with 

403 # the rail width). None keeps the responsive auto-fill packing. 

404 # Authored under the reserved ui_params["layout"] key. 

405 layout: Optional[dict] = None
