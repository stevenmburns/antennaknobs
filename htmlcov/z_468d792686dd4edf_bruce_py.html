1"""Bruce array: a series-fed vertically-polarised curtain (L. B. Cebik, W4RNL). 

2 

3One continuous wire folded into a square wave: quarter-wave VERTICAL risers 

4joined by quarter-wave HORIZONTAL jogs that alternate top and bottom. The trick 

5of the Bruce is the phasing it forces for free. Going around the meander, the 

6standing wave puts a current MAXIMUM on every vertical riser all pointing the 

7same way, while the horizontal jogs sit at the current minima and carry equal, 

8opposed currents that cancel in the far field. The result is a row of co-phased 

9vertical radiators on ~half-wave centres -- a VERTICALLY POLARISED broadside 

10curtain (bidirectional off +/-x), gain growing with the number of risers 

11(~6.5 dBi at five risers here, climbing past 7 with more). 

12 

13This is the catalog's first SERIES-fed curtain. The half-square/bobtail/bisquare 

14VP family are parallel/standing-wave structures fed at one element; the Bruce 

15threads a single standing wave through the entire meander, so the feedpoint 

16location sets the phase progression of the whole array (moving the feed to the 

17centre, as one can on the bobtail, instead DETUNES the Bruce). Geometrically it 

18is a long chain of right-angle junctions, a good stress on the engines' 

19junction handling. 

20 

21FEEDPOINT. Like the classic Bruce, the natural feed is near the bottom of an 

22END riser, in the region of a current MINIMUM: the driving point is HIGH and 

23strongly REACTIVE (here a few hundred ohms real over ~ -1.2 kohm). A real Bruce 

24resonates that out with a tuned matching network, NOT a coax-direct connection; 

25unlike the bobtail there is no clean current-maximum tap that keeps the phasing 

26(moving the feed detunes the array), so we model the honest high-Z feed and 

27treat gain/pattern -- not a 50 ohm match -- as the design target (cf. 

28`bisquare`, `lazy_h`). Note this is NOT as pathological as the bobtail's 

29ORIGINAL base feed: because the tap sits a little way up the riser 

30(`feed_height_frac`), off the exact current null, the four solver bases agree on 

31it to within a few percent -- high-Z but well-conditioned, not the basis- 

32dependent runaway a feed placed exactly on a current zero produces. 

33`feed_height_frac` slides the tap up the end riser to trade feed resistance 

34against a small pattern change. 

35 

36Geometry, in the framework's (x, y, z) convention: 

37 - y : the long axis of the curtain (risers march along y) 

38 - z : riser height; the square wave oscillates between `base` and `base+vert` 

39 - x : firing axis; VP radiation broadside off +/- x (planar in x = 0) 

40 

41 __ __ __ z = base + vert (top jogs) 

42 | | | | | | 

43 | | | | | | five quarter-wave risers (current maxima) 

44 F |__| |__| | z = base (bottom jogs); F = end feed 

45""" 

46 

47from ... import AntennaBuilder 

48import math 

49from types import MappingProxyType

52class Builder(AntennaBuilder): 

53 default_params = MappingProxyType( 

54 { 

55 "design_freq": 28.57, 

56 "freq": 28.57, 

57 # Height of the bottom jogs above ground. 

58 "base": 4.0, 

59 # Vertical riser length as a fraction of a wavelength (~0.25 wl). 

60 "vert_frac": 0.25, 

61 # Horizontal jog length as a fraction of a wavelength (~0.25 wl); 

62 # successive risers end up on ~half-wave centres. 

63 "horiz_frac": 0.25, 

64 # Number of vertical risers (radiators). 

65 "n_vert": 5, 

66 # Tap height of the feed on the END riser, as a fraction of its 

67 # length from the bottom. Small values sit near the current-minimum 

68 # end (very high reactance); larger values raise feed R but nudge 

69 # the pattern. ~0.3 is a usable compromise. 

70 "feed_height_frac": 0.3, 

71 # Overall scale knob (gain/pattern is the design target; the feed 

72 # is a high-Z current-minimum point, so this does not resonate it). 

73 "length_factor": 1.0, 

74 "ui_params": MappingProxyType( 

75 { 

76 # Series-fed current-minimum feed -> high, reactive Z; a 

77 # tuned matching network feeds it in practice. 

78 "target_z0": 600.0, 

79 "default_view": "yz", 

80 "n_vert": { 

81 "min": 3, 

82 "max": 9, 

83 "step": 1, 

84 "precision": 0, 

85 }, 

86 "feed_height_frac": { 

87 "min": 0.1, 

88 "max": 0.7, 

89 }, 

90 "length_factor": { 

91 "min": 0.9, 

92 "max": 1.1, 

93 }, 

94 }

96 }

99 def build_wires(self): 

100 eps = 0.05 

101 wavelength = 299.792458 / self.design_freq 

102 quarter = 0.25 * wavelength 

103 

104 vert = self.vert_frac * wavelength * self.length_factor 

105 horiz = self.horiz_frac * wavelength * self.length_factor 

106 z0 = self.base 

107 M = int(self.n_vert) 

108 

109 # Walk the square wave, recording the (y, z) node path. Each riser 

110 # alternates direction; horizontal jogs connect successive risers. 

111 nodes = [(0.0, z0)] 

112 riser_edge0 = [] # index of the lower node of each riser 

113 up = True 

114 for i in range(M): 

115 ycur, zcur = nodes[-1] 

116 riser_edge0.append(len(nodes) - 1) 

117 znew = zcur + vert if up else zcur - vert 

118 nodes.append((ycur, znew)) 

119 if i < M - 1: 

120 nodes.append((ycur + horiz, znew)) 

121 up = not up 

122 

123 # Feed sits on the first (end) riser at feed_height_frac up from its 

124 # bottom. The first riser always runs upward from z0. 

125 fa = riser_edge0[0] 

126 y_feed, z_riser_bot = nodes[fa] 

127 zf = z_riser_bot + self.feed_height_frac * vert 

128 

129 tups = [] 

130 for k, ((ya, za), (yb, zb)) in enumerate(zip(nodes[:-1], nodes[1:])): 

131 seg = math.hypot(yb - ya, zb - za) 

132 if k == fa: 

133 # Split the end riser: passive below, 1-seg driven gap, passive 

134 # above (a current-maximum-style tap, but on a current minimum). 

135 tups.append( 

136 ( 

137 (0.0, ya, za), 

138 (0.0, ya, zf), 

139 self.segs_for(zf - za, quarter), 

140 None, 

141 ) 

142 ) 

143 tups.append(((0.0, ya, zf), (0.0, ya, zf + 2 * eps), 1, 1 + 0j)) 

144 tups.append( 

145 ( 

146 (0.0, ya, zf + 2 * eps), 

147 (0.0, yb, zb), 

148 self.segs_for(zb - (zf + 2 * eps), quarter), 

149 None, 

150 ) 

151 ) 

152 else: 

153 tups.append( 

154 ((0.0, ya, za), (0.0, yb, zb), self.segs_for(seg, quarter), None) 

155 ) 

156 

157 return tups
