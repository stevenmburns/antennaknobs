1"""Terminated rhombic: a traveling-wave directional long-wire (L. B. Cebik, 

2W4RNL, "Long-Wire Antennas" / "The Terminated Vee-Beam and Rhombic"). 

3 

4Four long wires (each 1-4 wavelengths) arranged as a horizontal diamond. The 

5antenna is fed at one acute apex and TERMINATED at the opposite apex with a 

6non-inductive resistor (~600-800 ohm). The resistor absorbs the forward 

7traveling wave when it reaches the far apex, so almost no wave reflects back: 

8the current is a progressive (traveling) wave rather than a standing wave, and 

9the pattern is UNIDIRECTIONAL toward the terminated end. This is the catalog's 

10first non-resonant antenna -- its behaviour is set by wire length and the 

11termination, not by resonance, so it is inherently broadband (the driving-point 

12impedance stays near the termination value across a wide band). 

13 

14The legs each make a small "tilt" angle with the main axis so that the 

15long-wire radiation lobes of the four legs add along the axis. Gain runs a few 

16dB over a dipole with a 10-15 dB front-to-back ratio; roughly half the input 

17power is dissipated in the terminating resistor (the price of the clean 

18unidirectional, broadband pattern). 

19 

20Geometry, in the framework's (x, y, z) convention: 

21 - x : main axis; feed apex at x=0, terminated apex at x=2d; beam --> +x 

22 - y : transverse; the side apexes sit at (d, +/-w) 

23 - z : constant height `base` (a horizontal antenna over ground) 

24with d = L*cos(tilt), w = L*sin(tilt), L the leg length. Horizontally 

25polarised. 

26 

27 (d, +w) 

28 / \\ 

29 feed --> 0 2d --> [R termination], beam --> +x 

30 \\ / 

31 (d, -w) 

32""" 

33 

34from antennaknobs import AntennaBuilder 

35from antennaknobs.network import Driven, Load, Network, PortOnWire 

36import math 

37from types import MappingProxyType 

38 

39 

40class Builder(AntennaBuilder): 

41 default_params = MappingProxyType( 

42 { 

43 "design_freq": 28.57, 

44 "freq": 28.57, 

45 # Height above ground (a horizontal traveling-wave antenna). 

46 "base": 10.0, 

47 # Leg length as a fraction of a wavelength (each of the 4 legs). 

48 "leg_factor": 3.0, 

49 # Tilt angle (leg to main axis), degrees. Small -> long, narrow 

50 # diamond; tuned so the long-wire lobes add along the axis. 

51 "tilt_deg": 18.0, 

52 # Terminating resistance (ohm) at the far apex; absorbs the 

53 # forward wave to make the pattern unidirectional. 

54 "term_r": 700.0, 

55 "ui_params": MappingProxyType( 

56 { 

57 # Driving-point impedance tracks the termination (~700 ohm), 

58 # fed via open-wire line; reference SWR there. 

59 "target_z0": 700.0, 

60 "default_view": "xy", 

61 "leg_factor": { 

62 "min": 1.0, 

63 "max": 5.0, 

64 "step": 0.05, 

65 "precision": 3, 

66 }, 

67 "tilt_deg": { 

68 "min": 8.0, 

69 "max": 35.0, 

70 "step": 0.5, 

71 "precision": 2, 

72 }, 

73 "term_r": {"min": 400.0, "max": 1000.0, "step": 10.0}, 

74 } 

75 ), 

76 } 

77 ) 

78 

79 def build_wires(self): 

80 eps = 0.05 # half-gap at the feed / termination apexes 

81 wavelength = 299.792458 / self.design_freq 

82 quarter = 0.25 * wavelength 

83 

84 L = self.leg_factor * wavelength 

85 tilt = math.radians(self.tilt_deg) 

86 d = L * math.cos(tilt) 

87 w = L * math.sin(tilt) 

88 z = self.base 

89 

90 leg = self.segs_for(L, quarter) 

91 FU = (0.0, eps, z) # feed apex, upper terminal 

92 FL = (0.0, -eps, z) # feed apex, lower terminal 

93 SU = (d, w, z) # upper side apex 

94 SD = (d, -w, z) # lower side apex 

95 TU = (2 * d, eps, z) # terminated apex, upper terminal 

96 TL = (2 * d, -eps, z) # terminated apex, lower terminal 

97 

98 return [ 

99 # feed gap at the rear apex (driven via build_network) 

100 (FU, FL, self.segs_for(2 * eps, quarter), None, "feed"), 

101 # four legs 

102 (FU, SU, leg, None, None), # rear upper 

103 (FL, SD, leg, None, None), # rear lower 

104 (SU, TU, leg, None, None), # front upper 

105 (SD, TL, leg, None, None), # front lower 

106 # termination gap at the far apex (resistor via build_network) 

107 (TU, TL, self.segs_for(2 * eps, quarter), None, "term"), 

108 ] 

109 

110 def build_network(self): 

111 return Network( 

112 ports={"feed": PortOnWire("feed"), "term": PortOnWire("term")}, 

113 branches=[Load(port="term", r=self.term_r)], 

114 sources=[Driven(port="feed", voltage=1 + 0j)], 

115 )
