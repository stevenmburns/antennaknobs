1"""A delta loop built top-down, with the feed terminal *discovered* by flying a 

2slant into a plane rather than computed. 

3 

4The other drone-built variants fix the bottom feed gap and grow the triangle 

5upward (``delta_loop_marked`` flies each slant by its length; ``delta_loop_ 

6reflected`` flies pen-up to read off the top corner, then reflects). This one 

7inverts that: it starts at the **top centre**, flies right to the top corner, 

8turns down onto the slant, and lets ``Drone.forward_to_plane`` fly until it 

9crosses the feed plane ``y = eps`` -- the plane ``eps`` from the loop's centre 

10line. Neither the slant length nor the feed's height is computed in the script; 

11both fall out of where the slant meets that plane. From there it borrows the 

12*last two* variants' techniques wholesale: the two right-side vertices become 

13**named pins** (``mark``/``line_to``) and their **reflections** across the 

14``y = 0`` plane (``ry``) give the left side, so the four-edge loop closes with 

15no trig at all in this module. 

16 

17The loop is vertical, in the plane x = 0, fed by a short driven gap at the 

18bottom centre and opening upward: 

19 

20 R-----------L top, flown corner-to-corner from the centre start 

21 \\ / 

22 \\ / two equal slants; the right one is flown to the 

23 \\ / feed plane, the left is its reflection 

24 T~~~~S short driven feed gap straddling y = 0 (T = ry(S)) 

25 

26Because the build is anchored at the *top* (param ``top``), the feed sits 

27wherever the slant lands -- a little below ``top`` -- rather than at a fixed 

28height; with the defaults it lands near 7 m, matching the sibling loops' base. 

29""" 

30 

31from types import MappingProxyType 

32 

33from ... import AntennaBuilder, Drone 

34 

35 

36class Builder(AntennaBuilder): 

37 default_params = MappingProxyType( 

38 { 

39 "design_freq": 28.47, 

40 "freq": 28.47, 

41 # Height of the top edge -- the centre point the build starts from. 

42 # The feed height is NOT set here; it is whatever height the slant 

43 # has dropped to when it crosses the feed plane (see build_wires). 

44 "top": 10.0, 

45 # Scales the "go right" distance (half the top width), so it tunes 

46 # the loop size for resonance. The slant length follows from it and 

47 # the tilt angle via forward_to_plane. 

48 "length_factor": 1.0, 

49 # Tilt of each slant from vertical. 30 deg -> a 60 deg apex, the 

50 # classic near-equilateral delta loop. The downward turn at the top 

51 # corner is -(90 + angle_deg) deg (~ -120, i.e. a ~240/300 deg 

52 # swing) so the nose points down the slant toward the feed plane. 

53 "angle_deg": 30.0, 

54 "ui_params": MappingProxyType( 

55 { 

56 "target_z0": 100.0, 

57 "default_view": "yz", # loop lies in the x = 0 plane 

58 "length_factor": {"min": 0.85, "max": 1.15}, 

59 "angle_deg": {"min": 10.0, "max": 60.0}, 

60 } 

61 ), 

62 } 

63 ) 

64 

65 def build_wires(self): 

66 eps = 0.05 

67 wavelength = 299.792458 / self.design_freq 

68 quarter = 0.25 * wavelength 

69 # "Go right" distance: half the top width, wl-scaled so length_factor 

70 # tunes the loop. The slant length and the feed height stay unknown to 

71 # this script -- forward_to_plane discovers where the slant crosses the 

72 # feed plane. 

73 half_top = (wavelength / 6.0) * self.length_factor 

74 

75 n_seg0 = self.nominal_nsegs 

76 n_seg1 = max(3, self.nominal_nsegs // 7) 

77 

78 def ry(p): 

79 return p[0], -p[1], p[2] 

80 

81 drone = Drone(nominal_nsegs=self.nominal_nsegs, ref=quarter) 

82 

83 # Fly the right half pen-up to FIND its two vertices, no trig: 

84 # - start at the top centre, nose +y ("right"), up +x so every turn 

85 # stays in the x = 0 plane; 

86 # - fly half the top width to the top-right corner R; 

87 # - turn down onto the slant and fly until we cross the feed plane 

88 # y = eps (the plane eps from the centre line) to land on the 

89 # feed terminal S. forward_to_plane solves the slant length for us. 

90 drone.move_to((0.0, 0.0, self.top)) 

91 drone.face(heading=(0.0, 1.0, 0.0), up=(1.0, 0.0, 0.0)) 

92 R = drone.forward(half_top).position 

93 drone.yaw(-(90.0 + self.angle_deg)) 

94 S = drone.forward_to_plane((0.0, 1.0, 0.0, eps)).position 

95 

96 # The left half is the right half reflected across the y = 0 plane. 

97 L, T = ry(R), ry(S) 

98 

99 # Complete the loop with named pins (the marked variant's idiom): pin 

100 # all four corners, then stitch the perimeter S -> R -> L -> T with the 

101 # pen down (the drone works out each segment's direction), and finally 

102 # the short driven gap T -> S. 

103 drone.cut().move_to(S).mark("S") 

104 drone.move_to(R).mark("R") 

105 drone.move_to(L).mark("L") 

106 drone.move_to(T).mark("T") 

107 

108 drone.move_to(S).pay_out() 

109 drone.line_to("R", nsegs=n_seg0) # up the right slant 

110 drone.line_to("L", nsegs=n_seg0) # across the top 

111 drone.line_to("T", nsegs=n_seg0) # down the left slant 

112 drone.cut().move_to(T).feed(1 + 0j).line_to("S", nsegs=n_seg1) # feed gap 

113 

114 return drone.wires()
