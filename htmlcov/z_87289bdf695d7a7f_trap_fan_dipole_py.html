1"""Four-band trapped fan dipole — combines `fandipole` geometry with the 

2`trap_dipole` Load(parallel=True) idiom. 

3 

4Two physical fan-dipole spokes (per arm), each broken by a parallel-LC 

5trap segment partway out. At the trap's resonant frequency, the tank is 

6high-Z and interrupts the spoke's current — only the inner stub radiates, 

7behaving as a shorter dipole. Well below the trap, the tank looks 

8inductive and electrically lengthens the spoke so the full physical 

9length sets the low-band tuning. 

10 

11Default tuning: 

12 

13 spoke 0 (long) — full length resonant near 17m (18.16 MHz); 

14 trap at 12m (24.97 MHz) shortens it to a 12m dipole. 

15 spoke 1 (short) — full length resonant near 15m (21.38 MHz); 

16 trap at 10m (28.47 MHz) shortens it to a 10m dipole. 

17 

18→ four bands (10m / 12m / 15m / 17m) from a single shared feed. 

19 

20The cone geometry is the same fan layout as `fandipole.py`: each spoke 

21shares a common droop-angle direction (0, Zc, −Zs) from the cone apex 

22outward. 

23Each arm of each spoke is broken into 

24 

25 S → A[i] (cone segment) 

26 A[i] → trap_inner (inner outer segment) 

27 trap_inner → trap_outer (one segment, named "trap_<sign>_b<i>") 

28 trap_outer → tip (outer segment) 

29 

30with two arms per spoke (mirrored via ry()), so a 2-spoke design has 

31four traps total. The trap segment is wire-interior, so its basis 

32function carries the actual arm current and Load(parallel=True) acts 

33exactly as NEC2's ld_card type-1 would. 

34"""

46# Defaults: 5 µH traps, C chosen so each trap LC-resonates at its trap_freq. 

47# length_factor ≈ 0.49 captures the usual end-effect shortening on a fan 

48# spoke; tweak per band if a sweep shows the resonance off-target. 

49# All four bands tuned by `cli optimize --resonance`. Both traps have 

50# their L/C decoupled from the usual L↔C lockstep at ω₀ = trap_freq, in 

51# opposite directions: 

52# - Band 1: L pushed DOWN to ~0.2 µH (with C → ~159 pF). The 10m trap 

53# is only ~7 MHz above 15m, so the stock 5 µH puts ~+j1.4 kΩ on the 

54# spoke at 15m — enough to smother every series resonance. Cutting L 

55# weakens that loading; the 15m spoke can then find resonance at its 

56# physical length, while the tank still LC-resonates at 28.47 to 

57# interrupt the spoke at 10m. 

58# - Band 0: L pushed UP to ~11.85 µH (with C → ~3.4 pF). Heavier 

59# loading at 17m concentrates current toward the feed and lowers the 

60# resonant Re(Z) toward 50 Ω, shaving max SWR50 from 1.31 to 1.27. 

61# Beyond ~11.9 µH the 17m series resonance merges with the adjacent 

62# parallel-resonance pole and disappears. 

63# A droop angle ≈34.2° (instead of the original ~26.6°) drops the 

64# resistances uniformly so they straddle 50 Ω: 17m and 10m sit at the 

65# extremes with 

66# matching SWR50≈1.21, and the in-band bands (15m/12m) come in lower. 

67# A small `freq_shift` = 0.98 on band 1 nudges 10m into a different mode 

68# where Re(Z) ≈ 50 Ω (the trap doesn't fully open at 10m, the outer 

69# extension joins in, and the inner settles at ~0.40·λ/2 instead of 

70# ~0.50). 10m's SWR50 drops from 1.20 to 1.02 with the other three 

71# bands largely unchanged. (Band-0 shift has near-zero leverage on 

72# Re17 — kept at 1.0.) 

73# Length factors tuned against MomwireEngine with BSplineSolver(degree=2) 

74# at nominal_nsegs=41. After moving to adaptive per-wire segmentation 

75# (target_seg_len = max_wire / nominal_nsegs, see build_wires), both Bs2 

76# and the (since retired) Triangular basis stay essentially flat across N=21..81 — drift ≤ 0.22 Ω 

77# on every band — and agree with each other to ~1 Ω at the converged 

78# limit. Sinusoidal still wanders 2–8 Ω over the same N range (basis- 

79# family issue, not segmentation). PyNEC sits ~10 Ω above the momwire 

80# cluster on every band — a systematic offset — and on 10m / 12m 

81# actually drifts UP with refinement (57.7 → 67.2 Ω at 10m from N=21 to 

82# N=81), which is its own discretization story. 

83# Final SWR50 at target freqs (Bs2 @ N=41): 17m=1.04, 15m=1.02, 12m=1.19, 10m=1.13. 

84# Plan to verify by measurement after build. 

85_BAND_17_12 = { 

86 "full_freq": 18.1575, 

87 "trap_freq": 24.97, 

88 "full_length_factor": 0.442400, 

89 "inner_length_factor": 0.498100, 

90 # L=3 µH well below the parallel-resonance "cliff" (the regime 

91 # where the 17m series resonance merges with the adjacent parallel- 

92 # resonance pole at L ≳ 11.9 µH). The high-L cliff gives a marginally 

93 # better ideal SWR50 (1.05 vs 1.07) but the resonance Q is so high 

94 # there that ±1 cm length error sends SWR to 3-5. At L=3 the ±1 cm 

95 # tolerance is ≤ 1.17 SWR — buildable. C tracks at LC-resonance for 

96 # 24.97 MHz given this L. 

97 "trap_L_uH": 3.0, 

98 # Per-band trap resonant-freq multiplier (dimensionless). The tank's 

99 # C is computed from L and (trap_freq × freq_shift) at network-build 

100 # time, so freq_shift detunes ω₀ without an explicit C knob. Band-0's 

101 # shift has near-zero leverage on Re(Z) at 17m so it stays at 1.0. 

102 "freq_shift": 1.0, 

103}

112 # are hard to build reproducibly). C tracks at LC-resonance for

231 def build_wires(self): 

232 eps = 0.01 

233 radius = 0.12 

234 

235 n_bands = int(self.n_bands) 

236 if n_bands != 2: 

237 raise ValueError( 

238 f"trap_fan_dipole is a 2-spoke design; got n_bands={n_bands}" 

239 ) 

240 bands = tuple(self.bands)[:n_bands] 

241 

242 # Zc, Zs are the cos/sin of the droop angle — the unit fan-spoke 

243 # direction (0, Zc, -Zs) shared by every spoke beyond the cone. 

244 theta = math.radians(self.angle_deg) 

245 Zc = math.cos(theta) 

246 Zs = math.sin(theta) 

247 

248 def ry(p): 

249 return p[0], -p[1], p[2] 

250 

251 S = (0, eps, 0) 

252 T = ry(S) 

253 

254 # Each band's spoke originates from a short horizontal pigtail at 

255 # ±radius in x from the feed. The pigtail puts band 0 and band 1 

256 # on opposite sides of the feed (separated by 2·radius along x); 

257 # beyond the pigtail each spoke extends in the shared inverted-vee 

258 # direction (0, Zc, −Zs). The two spokes are then parallel 

259 # everywhere except the small horizontal segment near the feed, 

260 # which keeps the `angle_deg` parameter from also controlling the 

261 # band-spacing direction (a problem the original cone-apex layout 

262 # had — the droop angle and band spacing were entangled near the 

263 # feed). 

264 A = [ 

265 (+radius, S[1], S[2]), # band 0 on the +x side 

266 (-radius, S[1], S[2]), # band 1 on the −x side 

267 ] 

268 

269 def dist(p0, p1): 

270 return math.sqrt(sum((x0 - x1) ** 2 for x0, x1 in zip(p0, p1))) 

271 

272 # Outward direction shared by every spoke beyond the cone (the 

273 # "fan" direction): pure (0, Zc, −Zs). 

274 def offset_outward(p, q): 

275 return (p[0], p[1] + q * Zc, p[2] - q * Zs) 

276 

277 trap_seg = float(self.trap_seg_m) 

278 

279 # For each spoke, derive the three outer waypoints along the 

280 # shared outward direction: 

281 # trap_inner = A + q1 

282 # trap_outer = A + q1 + trap_seg 

283 # tip = A + q1 + trap_seg + q2 

284 # with q1 chosen so dist(S,A) + q1 = inner half-length, and the 

285 # full half-length (q1 + trap_seg + q2) set by the low-band freq. 

286 spokes = [] 

287 for i, b in enumerate(bands): 

288 full_half = float(b["full_length_factor"]) * ( 

289 0.5 * C_LIGHT_MHZ_M / float(b["full_freq"]) 

290 ) 

291 inner_half = float(b["inner_length_factor"]) * ( 

292 0.5 * C_LIGHT_MHZ_M / float(b["trap_freq"]) 

293 ) 

294 

295 q1 = inner_half - dist(S, A[i]) 

296 q2 = full_half - inner_half - trap_seg 

297 if q1 <= 0: 

298 raise ValueError( 

299 f"band {i}: inner half-length {inner_half:.3f} m is " 

300 f"shorter than the cone segment {dist(S, A[i]):.3f} m — " 

301 "shorten the cone or pick a lower trap_freq" 

302 ) 

303 if q2 <= 0: 

304 raise ValueError( 

305 f"band {i}: full half-length {full_half:.3f} m leaves no " 

306 f"room past the trap (inner {inner_half:.3f} + trap " 

307 f"{trap_seg:.3f}) — pick a lower full_freq or shorter inner" 

308 ) 

309 

310 trap_in = offset_outward(A[i], q1) 

311 trap_out = offset_outward(A[i], q1 + trap_seg) 

312 tip = offset_outward(A[i], q1 + trap_seg + q2) 

313 spokes.append((trap_in, trap_out, tip)) 

314 

315 # Adaptive segmentation: size each variable-length wire (cone + 

316 # inner outer + outer) so segment length is near-constant across 

317 # the antenna. nominal_nsegs sets the count for the longest such 

318 # wire; everything else scales proportionally. Trap segments are 

319 # pinned to 1 (load-port convention — the named port lives on a 

320 # single basis function). The feed bridge meshes via n_for like 

321 # the rest (1 segment at the default mesh, which is fine for the 

322 # BSpline d=2 basis this design is tuned against; the retired 

323 # triangular basis couldn't drive a 1-segment feed gap). 

324 adaptive_lengths = [] 

325 for i, (trap_in, trap_out, tip) in enumerate(spokes): 

326 adaptive_lengths.append(dist(S, A[i])) 

327 adaptive_lengths.append(dist(A[i], trap_in)) 

328 adaptive_lengths.append(dist(trap_out, tip)) 

329 target_seg_len = max(adaptive_lengths) / self.nominal_nsegs 

330 

331 def n_for(length): 

332 return max(1, round(length / target_seg_len)) 

333 

334 tups = [] 

335 for i, (trap_in, trap_out, tip) in enumerate(spokes): 

336 # +y arm 

337 tups.append((S, A[i], n_for(dist(S, A[i])), None)) 

338 tups.append((A[i], trap_in, n_for(dist(A[i], trap_in)), None)) 

339 tups.append((trap_in, trap_out, 1, None, f"trap_p_b{i}")) 

340 tups.append((trap_out, tip, n_for(dist(trap_out, tip)), None)) 

341 

342 # −y arm (mirror via ry) 

343 Ay = ry(A[i]) 

344 tin_y = ry(trap_in) 

345 tout_y = ry(trap_out) 

346 tip_y = ry(tip) 

347 tups.append((T, Ay, n_for(dist(T, Ay)), None)) 

348 tups.append((Ay, tin_y, n_for(dist(Ay, tin_y)), None)) 

349 tups.append((tin_y, tout_y, 1, None, f"trap_n_b{i}")) 

350 tups.append((tout_y, tip_y, n_for(dist(tout_y, tip_y)), None)) 

351 

352 # Feed wire — named "feed", source supplied by build_network(). 

353 # Meshed via n_for like the rest (1 segment at the default mesh); 

354 # the BSpline d=2 basis handles a 1-segment feed cleanly. 

355 tups.append((T, S, n_for(dist(T, S)), None, "feed")) 

356 

357 # Lift to base height. 

358 zoff = self.base 

359 lifted = [] 

360 for t in tups: 

361 (x0, y0, z0), (x1, y1, z1) = t[0], t[1] 

362 lifted.append(((x0, y0, z0 + zoff), (x1, y1, z1 + zoff), *t[2:])) 

363 return lifted 

364 

365 def build_network(self): 

366 bands = tuple(self.bands) 

367 ports = {"feed": PortOnWire("feed")} 

368 branches = [] 

369 for i, b in enumerate(bands): 

370 L = float(b["trap_L_uH"]) * 1e-6 

371 # Tank C is whatever LC-resonates at (trap_freq × freq_shift) 

372 # given L. freq_shift = 1 ⇒ tank resonates exactly at 

373 # trap_freq; values <1 push ω₀ down, >1 push it up. 

374 shift = float(b.get("freq_shift", 1.0)) 

375 omega0 = 2 * math.pi * float(b["trap_freq"]) * 1e6 * shift 

376 C = 1.0 / (omega0**2 * L) 

377 for sign in ("p", "n"): 

378 name = f"trap_{sign}_b{i}" 

379 ports[name] = PortOnWire(name) 

380 branches.append(Load(port=name, l=L, c=C, parallel=True)) 

381 return Network( 

382 ports=ports, 

383 branches=branches, 

384 sources=[Driven(port="feed", voltage=1 + 0j)], 

385 )
