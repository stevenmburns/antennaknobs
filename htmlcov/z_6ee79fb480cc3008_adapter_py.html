1"""Bridge antennaknobs's Builder idiom into momwire's web AntennaExample. 

2 

3Each `designs/<name>.py` exposes a `Builder` class with `default_params` 

4(a MappingProxyType of physics knobs). We walk that registry, derive a 

5`ParamSpec` schema from `default_params` (with optional per-design 

6overrides under the reserved `ui_params` key), and register one 

7`AntennaExample` per design so the existing momwire web frontend can 

8drive it without per-design glue. 

9 

10Reserved keys inside `ui_params`: 

11 default_view : "xy" | "yz" | "xz" — initial 2D projection 

12 target_z0 : float — reference impedance for SWR (default 50) 

13 meas_freq_range : (lo, hi) — measurement-freq slider span override 

14 bands : tuple[BandSpec] — band tabs (default HF amateur set) 

15 sweep_policy : (anchor, lo_factor, hi_factor) 

16 multi_feed : bool — declare multi-feed response shape 

17 layout : dict {columns: int} — pin the knob grid to a fixed 

18 column count so per-param `layout` col positions are 

19 stable (default: responsive auto-flow packing) 

20 <param_name> : dict of {min, max, step, unit, label, precision, 

21 kind, sweepable, enum_options, layout, hidden} 

22 — slider-bounds + metadata overrides for one param. 

23 `layout` is {row, col, row_span, col_span} (1-indexed 

24 CSS grid lines, all optional) to place this knob 

25 explicitly. `hidden: True` suppresses the control 

26 entirely (the param stays pinned at its default value 

27 through solves) — for a knob that's degenerate with 

28 another. Anything missing falls back to auto-derived 

29 defaults. 

30 

31Everything else in `default_params` becomes a `ParamSpec`. Numeric 

32defaults become float sliders with auto bounds (±50% around default); 

33ints become int sliders; bools become checkboxes; complex defaults are 

34skipped (no UI yet — the request can still override via re/im dict).

133def _nice_step(raw: float) -> float: 

134 """Snap a raw step to the 1-2-5 series (… 0.001, 0.002, 0.005, 0.01 …) 

135 so auto-derived sliders advance in familiar increments instead of odd 

136 grids like 0.003 or 0.0007. Picks the nearest 1/2/5·10ⁿ using the 

137 conventional log-spaced thresholds (1.5, 3, 7).""" 

138 if raw <= 0.0: 

139 return raw 

140 exp = math.floor(math.log10(raw)) 

141 mant = raw / 10.0**exp # in [1, 10) 

142 nice = 1.0 if mant < 1.5 else 2.0 if mant < 3.0 else 5.0 if mant < 7.0 else 10.0 

143 return nice * 10.0**exp

146def _precision_for_step(step: float) -> int: 

147 """Decimal places to display a value stepped by `step`, with one digit 

148 of headroom so an off-grid default still reads meaningfully. Capped at 

149 6 to keep labels sane for very small factors.""" 

150 if step <= 0.0: 

151 return 3 

152 return min(6, max(0, -math.floor(math.log10(step))) + 1)

172def _auto_paramspec(name: str, default: Any, override: dict | None) -> ParamSpec | None: 

173 """Build a ParamSpec from a default value plus optional UI overrides. 

174 

175 Returns None when the value type has no UI representation (complex, 

176 string-non-enum, etc.) and no override was supplied — the param is 

177 still settable via the API, it just doesn't appear in the UI. 

178 """ 

179 override = dict(override or {}) 

180 label = override.pop("label", _display_label(name)) 

181 unit = override.pop("unit", None) 

182 # Optional explicit grid placement for this knob (row/col/spans). Only a 

183 # dict is meaningful; anything else is ignored so a typo can't crash the 

184 # registry. Passed verbatim to ParamSpec.layout for every kind. 

185 layout_raw = override.pop("layout", None) 

186 layout = dict(layout_raw) if isinstance(layout_raw, dict) else None 

187 # Precision (decimal places shown on the slider label) defaults to None 

188 # here so the numeric branch can derive it from the resolved step. Any 

189 # non-numeric path falls back to 3, matching the historical default. 

190 explicit_precision = override.pop("precision", None) 

191 precision = 3 if explicit_precision is None else int(explicit_precision) 

192 sweepable = bool(override.pop("sweepable", name == "freq")) 

193 

194 if isinstance(default, bool): 

195 kind = override.pop("kind", "bool") 

196 return ParamSpec( 

197 name=name, 

198 label=label, 

199 default=default, 

200 kind=kind, 

201 unit=unit, 

202 precision=precision, 

203 layout=layout, 

204 ) 

205 

206 if isinstance(default, (int, float)) and not isinstance(default, bool): 

207 is_int = isinstance(default, int) and override.get("kind") != "float" 

208 kind = override.pop("kind", "int" if is_int else "float") 

209 d = float(default) 

210 # Auto bounds: a generous ±50% window. The step gives 0.1% 

211 # *relative* resolution (window / 1000) so any scaling factor, 

212 # fraction, length, or angle is fine-tunable by hand regardless of 

213 # its magnitude — a flat absolute step would be too coarse for 

214 # sub-unity fractions and needlessly fine for large values. Snapped 

215 # to the 1-2-5 series for clean slider stops. For an int 

216 # default of 0 the multiplicative window collapses, so fall back to 

217 # a small absolute range. 

218 if d == 0.0: 

219 lo, hi, step = -1.0, 1.0, 0.1 

220 else: 

221 lo = d * 0.5 if d > 0 else d * 1.5 

222 hi = d * 1.5 if d > 0 else d * 0.5 

223 step = _nice_step(max((hi - lo) / 1000.0, 1e-9)) 

224 if kind == "int": 

225 lo = float(int(round(lo))) 

226 hi = float(int(round(hi))) 

227 step = 1.0 

228 # Phase params (phase_lr, phase_tb, ...) are degrees, converted 

229 # to a phasor by the array builders via exp(j π · phase / 180). 

230 # ±180° covers the full unit circle; signed range puts the 

231 # zero-phase reference at slider centre with positive = lead, 

232 # negative = lag. The auto-derived (-1, 1) fallback for 

233 # default=0 would otherwise give a useless 2° span. 

234 if name.startswith("phase_"): 

235 lo, hi, step = -180.0, 180.0, 1.0 

236 unit = unit or "°" 

237 # `design_freq` is the geometry-sizing frequency for 

238 # geometry-from-design_freq designs (wavelength = c / design_freq, then 

239 # dimensions are wavelength × factors). Wire it into the 

240 # global designFreq state on the frontend so the slider 

241 # actually retunes the geometry AND the meas-freq slider 

242 # follows when linkMeas is on. Top-level designs don't have a 

243 # design_freq param — their geometry is hand-tuned in absolute 

244 # meters and the measurement freq slider (at the top of the 

245 # UI) is the only thing that needs to move per solve. 

246 if name == "design_freq": 

247 unit = unit or "MHz" 

248 override["linked_to_design_freq"] = True # keep around 

249 # Angle params read in degrees; show the ° unit on the slider so the 

250 # label can drop the redundant `_deg` token (see _display_label), and 

251 # default to a 0.5° step (finer than the auto 1-2-5 step is overkill, 

252 # coarser loses the half-degree tuning hams expect). A design's 

253 # ui_params `step` still overrides; int-typed angles keep whole steps. 

254 if _is_degree_param(name): 

255 unit = unit or "°" 

256 if kind != "int": 

257 step = 0.5 

258 final_step = float(override.pop("step", step)) 

259 # Derive display precision from the resolved step (matching its 

260 # decimals plus one digit of headroom), unless the design pinned a 

261 # precision or this is an int / phase param fixed to whole units. 

262 if explicit_precision is not None: 

263 resolved_precision = int(explicit_precision) 

264 elif kind == "int" or name.startswith("phase_"): 

265 resolved_precision = 0 

266 else: 

267 resolved_precision = _precision_for_step(final_step) 

268 spec_kwargs = dict( 

269 name=name, 

270 label=label, 

271 default=int(d) if kind == "int" else d, 

272 kind=kind, 

273 min=float(override.pop("min", lo)), 

274 max=float(override.pop("max", hi)), 

275 step=final_step, 

276 precision=resolved_precision, 

277 unit=unit, 

278 sweepable=sweepable, 

279 layout=layout, 

280 ) 

281 if "linked_to_design_freq" in override: 

282 spec_kwargs["linked_to_design_freq"] = bool( 

283 override.pop("linked_to_design_freq") 

284 ) 

285 if "link_meas_freq_to_param" in override: 

286 spec_kwargs["link_meas_freq_to_param"] = str( 

287 override.pop("link_meas_freq_to_param") 

288 ) 

289 return ParamSpec(**spec_kwargs) 

290 

291 if isinstance(default, str): 

292 opts = override.pop("enum_options", None) 

293 if opts is None: 

294 return None 

295 return ParamSpec( 

296 name=name, 

297 label=label, 

298 default=default, 

299 kind="enum", 

300 enum_options=tuple(opts), 

301 precision=precision, 

302 unit=unit, 

303 layout=layout, 

304 ) 

305 

306 # complex, None, or anything exotic — skip the auto-UI; the request 

307 # body can still override via {"re": ..., "im": ...}. 

308 return None

311def _group_spec_from_default( 

312 name: str, 

313 default_value: tuple | list, 

314 ui_override: dict, 

315 all_default_params: dict, 

316) -> ParamGroupSpec | None: 

317 """Build a ParamGroupSpec from a tuple/list-of-dicts default value. 

318 

319 The default value's length seeds default_overrides for the group's 

320 instances; the inner ParamSpecs come from auto-deriving each key of 

321 the first instance dict (with optional per-leaf overrides supplied 

322 under the same ui_override dict, keyed by leaf name). 

323 

324 `ui_override` is the dict stored under `ui_params[<group_name>]`. 

325 Recognised keys: label_template, repeat_count, max_repeats, 

326 link_meas_freq_to_param, plus any leaf-name → override-dict pairs. 

327 Falls back to sensible defaults when missing. 

328 """ 

329 if not default_value or not all(isinstance(d, dict) for d in default_value): 

330 return None 

331 template = default_value[0] 

332 if not template: 

333 return None 

334 

335 repeat_count = ui_override.get("repeat_count") 

336 if repeat_count is None: 

337 # Heuristic: prefer n_<name> (n_bands for bands), then n_<singular>. 

338 for cand in (f"n_{name}", f"n_{name.rstrip('s')}"): 

339 if cand in all_default_params: 

340 repeat_count = cand 

341 break 

342 if not isinstance(repeat_count, str): 

343 # No count param → can't render a repeating group. 

344 return None 

345 

346 max_repeats = int(ui_override.get("max_repeats", len(default_value))) 

347 label_template = str(ui_override.get("label_template", f"{name} {{i}}")) 

348 link = ui_override.get("link_meas_freq_to_param") 

349 

350 inner_params: list[ParamSpec] = [] 

351 for leaf_name, leaf_default in template.items(): 

352 leaf_override = ui_override.get(leaf_name) 

353 if leaf_override is None or not isinstance(leaf_override, dict): 

354 leaf_override = {} 

355 spec = _auto_paramspec(leaf_name, leaf_default, dict(leaf_override)) 

356 if spec is not None: 

357 inner_params.append(spec) 

358 if not inner_params: 

359 return None 

360 

361 default_overrides = tuple(dict(d) for d in default_value) 

362 

363 return ParamGroupSpec( 

364 name=name, 

365 label_template=label_template, 

366 repeat_count=repeat_count, 

367 max_repeats=max_repeats, 

368 params=tuple(inner_params), 

369 default_overrides=default_overrides, 

370 link_meas_freq_to_param=str(link) if isinstance(link, str) else None, 

371 )

374def _derive_schema(default_params: dict) -> tuple: 

375 ui = dict(default_params.get("ui_params") or {}) 

376 specs: list[ParamSpec] = [] 

377 for key, default in default_params.items(): 

378 if key == "ui_params": 

379 continue 

380 # `freq` is measurement frequency only — driven by the dedicated 

381 # meas-freq slider at the top of the UI, never by a schema 

382 # slider. The Builder's default_params['freq'] value is still 

383 # used as the initial measurement freq when the example loads; 

384 # the adapter just doesn't expose a redundant slider for it. 

385 # 

386 # `design_freq` is the geometry-sizing frequency for 

387 # design_freq-sized designs, driven by the "design freq" band-tab 

388 # row + slider in the UI (which sends design_freq_mhz on the 

389 # request). Skipping it here too prevents the auto-derived 

390 # schema slider from duplicating that control. 

391 if key in ("freq", "design_freq"): 

392 continue 

393 # `hidden`: the design pins this param at its default and suppresses its 

394 # control. The value still flows through every solve (it's in 

395 # default_params, which _build_builder seeds from), so this is a 

396 # display-only override — used to drop a knob that's degenerate with 

397 # another (e.g. a `_frac` that only ever multiplies `length_factor`). 

398 # Checked before the group/scalar branches so it applies to any kind. 

399 override_raw = ui.get(key) 

400 if isinstance(override_raw, dict) and override_raw.get("hidden"): 

401 continue 

402 # Repeating-group default: tuple/list of dicts → ParamGroupSpec. 

403 # The ui_params override (if any) carries the group-level 

404 # config (label_template, repeat_count, max_repeats, 

405 # link_meas_freq_to_param) plus per-leaf override dicts. 

406 if ( 

407 isinstance(default, (tuple, list)) 

408 and default 

409 and all(isinstance(x, dict) for x in default) 

410 ): 

411 group_override = ui.get(key) 

412 if not isinstance(group_override, dict): 

413 group_override = {} 

414 group_spec = _group_spec_from_default( 

415 key, default, group_override, default_params 

416 ) 

417 if group_spec is not None: 

418 specs.append(group_spec) 

419 continue 

420 override = ui.get(key) 

421 if override is not None and not isinstance(override, dict): 

422 # Reserved scalar (e.g. `target_z0`) — not a per-param spec. 

423 continue 

424 spec = _auto_paramspec(key, default, override) 

425 if spec is not None: 

426 specs.append(spec) 

427 return tuple(specs)

447def _build_builder(cls, req: dict): 

448 """Construct a Builder from default_params overlaid with request fields. 

449 

450 The momwire frontend assembles its solve request by Object.assign'ing 

451 every live slider value as a *top-level* key on the request dict 

452 (App.tsx:buildRequest), so we read each Builder param off the request 

453 directly. A nested `params` dict is also accepted as a fallback for 

454 other clients. 

455 """ 

456 # Seed from the named variant (e.g. `opt_params`, `z50_params`). 

457 # Unrecognised / absent → fall back to default_params. 

458 base = _strip_ui(_variant_params(cls, req.get("variant"))) 

459 nested = req.get("params") or {} 

460 for k in list(base.keys()): 

461 if k in req: 

462 base[k] = _rehydrate_param(base[k], req[k]) 

463 elif k in nested: 

464 base[k] = _rehydrate_param(base[k], nested[k]) 

465 builder = cls(params=base) 

466 # n_per_wire drives the per-Builder nominal_nsegs (the convergence 

467 # sweep at /converge overrides this value per N). Each generator 

468 # decides which per-edge segment counts scale with it and which stay 

469 # fixed (feed gaps). See AntennaBuilder.FRAMEWORK_PARAMS. 

470 n_per_wire = req.get("n_per_wire") 

471 if n_per_wire is not None: 

472 builder.nominal_nsegs = int(n_per_wire) 

473 return builder

490def _ground_for_engine(req: dict, ground_z: float): 

491 """Map the frontend's ground knobs to MomwireEngine's ground spec — 

492 same three-way model as `_pynec_ground_spec`, one shared selector 

493 describing the GROUND; each engine approximates it as best it can. 

494 "pec" → the PEC image; "sommerfeld" → ("finite", ...), which momwire 

495 solves as the TRUE Sommerfeld ground on every solver (momwire >= 

496 0.8.0: bspline dense, sinusoidal field-based, hmatrix/arrayblock fast 

497 paths); "fast" → ("finite-fast", ...), the reflection-coefficient 

498 model everywhere. The response ships the engine's actual eps/sigma so 

499 the frontend far-field Fresnel uses the real constants either way; 

500 `ground_model_applied` reports what the impedance solve really 

501 used.""" 

502 model = _requested_ground_model(req) 

503 if model is None: 

504 return None 

505 if model == "pec": 

506 return "pec" 

507 if model == "fast": 

508 return ("finite-fast",) + DEFAULT_GROUND[1:] 

509 return DEFAULT_GROUND

598def _primary_feed(engine): 

599 """(polyline_idx, arclength) of the driven feed, or None. 

600 

601 MomwireEngine exposes `_feeds = [(polyline_idx, arclength, voltage)]` 

602 post-translator. For network-spec designs the geometry translator 

603 registers a feed for every named edge — including non-driven ports 

604 like trap stubs — so `_feeds[0]` is whichever named tuple appears 

605 first in `build_wires()`, not necessarily the driven feed. Look up 

606 the driven port's `_feeds` entry by index when a Network is present. 

607 """ 

608 feeds = getattr(engine, "_feeds", None) or [] 

609 feed_names = getattr(engine, "_feed_names", None) or [] 

610 if not feeds: 

611 return None 

612 feed_idx = 0 

613 network = getattr(engine, "_network", None) 

614 if network is not None and network.sources: 

615 # The first Driven source is the primary feed. If it resolves to a 

616 # real (PortAtEdge) port, use its position; otherwise (virtual port, 

617 # e.g. delta_looparray_network's "driver"), fall back to feeds[0]. 

618 driven_name = network.sources[0].port 

619 if driven_name in feed_names: 

620 feed_idx = feed_names.index(driven_name) 

621 pl_idx, arclen, _v = feeds[feed_idx] 

622 return int(pl_idx), float(arclen)

625def _interp_polyline(knots, cum, arclen): 

626 """3D point at `arclen` along a polyline (knots + cumulative arclength).""" 

627 arclen = min(max(arclen, 0.0), float(cum[-1])) 

628 seg = int(np.searchsorted(cum, arclen, side="right")) - 1 

629 seg = min(max(seg, 0), len(knots) - 2) 

630 span = cum[seg + 1] - cum[seg] 

631 t = 0.0 if span <= 0 else (arclen - cum[seg]) / span 

632 return (knots[seg] + t * (knots[seg + 1] - knots[seg])).tolist()

635def _feed_indices(engine, currents) -> tuple[int, int]: 

636 """Pick a (wire, knot) for the feed marker — the knot nearest the feed. 

637 

638 Kept for the feed_knot_index the frontend uses to read feed current and 

639 split the current envelope. The visible marker dot uses `_feed_position` 

640 instead (exact, not snapped to a knot). 

641 """ 

642 pf = _primary_feed(engine) 

643 if pf is None: 

644 return 0, 0 

645 pl_idx, arclen = pf 

646 if pl_idx >= len(currents): 

647 return 0, 0 

648 knots = currents[pl_idx].knot_positions 

649 if knots.shape[0] < 2: 

650 return pl_idx, 0 

651 # Cumulative arclength along the polyline. 

652 deltas = np.linalg.norm(np.diff(knots, axis=0), axis=1) 

653 cum = np.concatenate([[0.0], np.cumsum(deltas)]) 

654 j = int(np.argmin(np.abs(cum - arclen))) 

655 return pl_idx, j

658def _feed_position(engine, currents): 

659 """Exact 3D feed point for the primary feed — the physical location the 

660 solver actually feeds, independent of where segment knots fall. Avoids 

661 the half-segment marker shift from snapping to the nearest knot, which 

662 lands on an endpoint when the feed edge has no interior knot (e.g. a 

663 1-segment driven stub under odd-parity bases like sinusoidal/Bspline=2).

665 pf = _primary_feed(engine)

668 pl_idx, arclen = pf 

669 if pl_idx >= len(currents): 

670 return None 

671 knots = currents[pl_idx].knot_positions 

672 if knots.shape[0] < 2: 

673 return knots[0].tolist() if knots.shape[0] else None

675 cum = np.concatenate([[0.0], np.cumsum(deltas)])

750def _discover_variants(cls) -> tuple[str, ...]: 

751 """Names of every class-level `<name>_params` attribute (the variant 

752 convention used across the design library — e.g. `default_params`, 

753 `opt_params`, `z50_params`, `current_physical_params`). The 

754 returned list is suitable for a UI selector; the bare names (no 

755 `_params` suffix) are what the frontend sends back in the request. 

756 

757 `default` is always first if present, so the UI lists it as the 

758 canonical starting point regardless of class attribute order. 

759 """ 

760 suffix = "_params" 

761 found: list[str] = [] 

762 for attr in dir(cls): 

763 if not attr.endswith(suffix) or attr.startswith("_"): 

764 continue 

765 v = getattr(cls, attr, None) 

766 # MappingProxyType / dict only — skip e.g. a method that happens 

767 # to end in _params. 

768 if not hasattr(v, "keys"): 

769 continue 

770 name = attr[: -len(suffix)] 

771 if name: 

772 found.append(name) 

773 # `default` first, rest in stable (alphabetical) order. 

774 found.sort(key=lambda n: (n != "default", n)) 

775 return tuple(found)

822def _auto_target_z0(cls) -> float: 

823 """Default reference impedance for the SWR readout. 

824 

825 Array designs scale 50 Ω by the element count (1×2 → 100, 2×2 → 200, 

826 2×4 → 400, ...) — the convention that each branch in the splitter 

827 sees 50 Ω after the chain of impedance transformers, so the 

828 combined driving point lands at N × 50. 

829 

830 Everything else defaults to 50 Ω. Designs that violate either 

831 convention (turnstiles with per-port 50 Ω matching, designs tuned 

832 to 75 Ω, etc.) override via `ui_params["target_z0"]`. 

833 """ 

834 if not issubclass(cls, _ARRAY_BASES): 

835 return 50.0 

836 try: 

837 b = cls() 

838 n_feeds = sum(1 for *_, ev in b.build_wires() if ev is not None) 

839 except Exception: 

840 return 50.0 

841 return 50.0 * max(1, n_feeds)

864def _auto_default_view(cls) -> str: 

865 """Pick a 2D projection from the spans of the antenna's wires. 

866 

867 Rule: if x_span is small (the antenna lies in the y-z plane — 

868 typical for dipoles, V's, loops, fan/bowtie variants), default to 

869 `yz`. Otherwise return the plane of the two largest spans (xy / yz 

870 / xz). The 0.5 m threshold catches feed-gap micro-offsets like 

871 fan_dipole's 0.22 m without flipping to xy. 

872 

873 Hand-overridden via ui_params['default_view']; designs whose axis 

874 layout doesn't match this rule (vertical, moxonarray) supply the 

875 explicit value. 

876 """ 

877 try: 

878 b = cls() 

879 pts = [] 

880 for p0, p1, _n, _e in b.build_wires(): 

881 pts.append(p0) 

882 pts.append(p1) 

883 a = np.asarray(pts, dtype=float) 

884 except Exception: 

885 return "xy" 

886 sx = float(a[:, 0].max() - a[:, 0].min()) 

887 sy = float(a[:, 1].max() - a[:, 1].min()) 

888 sz = float(a[:, 2].max() - a[:, 2].min()) 

889 if sx < 0.5: 

890 return "yz" 

891 spans = sorted([("x", sx), ("y", sy), ("z", sz)], key=lambda t: t[1], reverse=True) 

892 return "".join(sorted(s[0] for s in spans[:2]))

895@lru_cache(maxsize=None) 

896def _recommended_backend(cls) -> str | None: 

897 """Recommend a default solver for the design, or None to let the UI keep 

898 its own default (the dense B-spline path). 

899 

900 Returns "arrayblock" for true grid arrays — multiple electrically separate 

901 elements with at least one repeated shape — where the element-aware block 

902 solver is dramatically faster than the dense default (e.g. bowtiearray2x4: 

903 ~1 s vs ~8 s). Single-element designs, and multi-element designs whose 

904 elements are all distinct (Yagi-style), keep the dense default so their 

905 basis/results are unchanged. Detection is geometry-only (no solve) and any 

906 failure falls back to None. 

907 

908 Memoised per design class: it already runs only once per design at registry 

909 build (the result is baked into the immutable `AntennaExample`, which the 

910 /examples endpoint and the frontend read at runtime — a slider change never 

911 re-runs it), but `lru_cache` makes that a hard guarantee regardless of 

912 call site. 

913 """ 

914 try: 

915 from momwire.array_block import _wire_to_element 

916 

917 builder = _build_builder(cls, {}) 

918 eng = _make_momwire_engine({}, builder) 

919 polylines = [np.asarray(p, dtype=float) for p in eng._polylines] 

920 except Exception: 

921 return None 

922 if len(polylines) < 2: 

923 return None 

924 wire_elem, n_elem = _wire_to_element(polylines) 

925 # array-block only pays off for a genuine grid array: several elements where 

926 # ONE shape repeats many times (so per-shape block reuse dominates). Require 

927 # at least 4 elements — below that the speedup is marginal and 2-element 

928 # symmetric things (a split dipole, a 1x2) are ambiguous. 

929 if n_elem < 4: 

930 return None 

931 # Signature each element by its points recentred on its own centroid, then 

932 # require repetition to *dominate*: at least half the elements must be 

933 # duplicates of another (len(sigs) * 2 <= n_elem). The earlier test 

934 # (len(sigs) < n_elem) fired on a single repeated pair, which wrongly tagged 

935 # Yagis (their equal-length directors collapse to one signature while the 

936 # driven element and reflector stay distinct) as arrays. 

937 sigs = set() 

938 for e in range(n_elem): 

939 pts = np.vstack( 

940 [polylines[w] for w in range(len(polylines)) if wire_elem[w] == e] 

941 ) 

942 pts = pts - pts.mean(axis=0) 

943 key = np.round(pts / 1e-4).astype(np.int64) 

944 key = key[np.lexsort(key.T)] 

945 sigs.add(key.tobytes()) 

946 return "arrayblock" if len(sigs) * 2 <= n_elem else None

1363 def params_source(req: dict) -> str: 

1364 # Overlay the request's live knob values onto the chosen variant's 

1365 # params (which still carry ui_params and the design's real nesting — 

1366 # bands tuples etc.), then serialise. Knob-values-only by default 

1367 # (include_ui), matching the manual "copy the printed values" workflow 

1368 # this replaces; pass include_ui=true to emit a wholesale block.

1370 from antennaknobs.serialize import params_source as _emit 

1371 

1372 variant = req.get("variant") 

1373 base = dict(_variant_params(cls, variant)) # retains ui_params 

1374 ui = base.get("ui_params") 

1375 nested = req.get("params") or {} 

1376 for k in list(base.keys()): 

1377 if k == "ui_params": 

1378 continue 

1379 if k in req: 

1380 base[k] = _rehydrate_param(base[k], req[k]) 

1381 elif k in nested: 

1382 base[k] = _rehydrate_param(base[k], nested[k])

1394 else None

1398 emit = diff_params(dict(_variant_params(cls, "default")), base) 

1399 else: 

1400 name = ( 

1401 f"{variant}_params" 

1402 if variant and variant != "default" 

1403 else "default_params" 

1404 )

1408 name=name,

1528def list_designs() -> list[str]: 

1529 """Discover every Builder file under designs/. 

1530 

1531 Every design lives in a family subpackage (`dipoles/`, `loops/`, 

1532 `arrays/`, …) and registers under the dotted path the user sees in 

1533 the UI (`dipoles.invvee`) — the same convention as the Python import 

1534 path, minus the leading `antennaknobs.designs.`. The dotted name 

1535 is what `register_all` feeds back to importlib too. Any bare top-level 

1536 `*.py` (none today) would register under its stem. 

1537 """ 

1538 names: list[str] = [] 

1539 for p in sorted(DESIGNS_DIR.glob("*.py")): 

1540 if p.stem.startswith("_"): 

1541 continue 

1542 names.append(p.stem) 

1543 for sub in sorted(d for d in DESIGNS_DIR.iterdir() if d.is_dir()): 

1544 if sub.name.startswith("_") or sub.name == "__pycache__": 

1545 continue 

1546 for p in sorted(sub.glob("*.py")): 

1547 if p.stem.startswith("_"): 

1548 continue 

1549 names.append(f"{sub.name}.{p.stem}") 

1550 return names
