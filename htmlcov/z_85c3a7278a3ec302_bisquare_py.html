1r"""Bi-square: a two-wavelength loop worked as a vertical broadside curtain 

2(L. B. Cebik, W4RNL). 

3 

4Take the half-square (a 1 wl conductor) and double it into a 2 wl loop with 

5four HALF-wave sides, hung as a square standing on one corner (a diamond) and 

6fed at the bottom corner. Current reverses every half wavelength -- i.e. at 

7every corner -- so by symmetry the horizontal (left/right) field components of 

8the four slanting sides cancel while the VERTICAL components add: the bi-square 

9is VERTICALLY POLARISED and fires bidirectionally broadside to the plane of the 

10loop. In free space its gain is modest (a couple of dB over a dipole), but 

11mounted low it is a strong LOW-ANGLE VP performer, the best-known member of 

12Cebik's "self-contained vertical" (SCV) loop family. 

13 

14This fills the "vertical loop curtain" gap: the catalog's loops (delta, 

15diamond, quad) are ~1 wl radiators or horizontally polarised beams; the 

16bi-square is a 2 wl VP curtain needing no ground plane. Cebik's max-pattern 

17proportion runs the sides a touch over a half wave (~0.55 wl here). 

18 

19Geometry, in the framework's (x, y, z) convention: 

20 - z : height; the loop's tall vertical extent is what makes it VP 

21 - y : the left/right corners splay to +/- y 

22 - x : firing axis; broadside off +/- x (planar in x = 0) 

23 

24 o top corner z = base + 2*hd 

25 / \ 

26 left corner o o right corner z = base + hd 

27 \ / 

28 F bottom corner (fed) z = base 

29""" 

30 

31from ... import AntennaBuilder 

32from types import MappingProxyType 

33import math 

34 

35 

36class Builder(AntennaBuilder): 

37 default_params = MappingProxyType( 

38 { 

39 "design_freq": 28.57, 

40 "freq": 28.57, 

41 # Height of the bottom (fed) corner above ground. 

42 "base": 4.0, 

43 # Side length as a fraction of a wavelength (~half-wave per side; 

44 # four sides -> ~2 wl loop). ~0.55 wl is the modelled max-pattern 

45 # proportion. Pinned (hidden) because the geometry depends only on 

46 # the product side_frac * length_factor (see build_wires): exposing 

47 # both would be two knobs for one degree of freedom. length_factor 

48 # is the visible trim knob; this stays at Cebik's max-pattern value. 

49 "side_frac": 0.55, 

50 # Overall scale knob (peak gain / pattern, not a low-Z resonance, 

51 # is the design target -- the corner feed is high and reactive). 

52 "length_factor": 1.0, 

53 "ui_params": MappingProxyType( 

54 { 

55 # Corner-fed 2 wl loop -> high, reactive feed; open-wire + 

56 # tuner in practice. 

57 "target_z0": 300.0, 

58 "default_view": "yz", 

59 # Degenerate with length_factor (product-only); pin it and 

60 # let length_factor be the single scale knob. 

61 "side_frac": {"hidden": True}, 

62 "length_factor": { 

63 "min": 0.9, 

64 "max": 1.1, 

65 }, 

66 } 

67 ), 

68 } 

69 ) 

70 

71 def build_wires(self): 

72 eps = 0.05 

73 wavelength = 299.792458 / self.design_freq 

74 quarter = 0.25 * wavelength 

75 

76 side = self.side_frac * wavelength * self.length_factor 

77 # Square standing on a corner: the half-diagonal is side / sqrt(2). 

78 hd = side / math.sqrt(2) 

79 

80 Bc = (0.0, 0.0, self.base) # bottom corner (fed) 

81 Rc = (0.0, hd, self.base + hd) # right corner 

82 Tc = (0.0, 0.0, self.base + 2 * hd) # top corner 

83 Lc = (0.0, -hd, self.base + hd) # left corner 

84 

85 ns = self.segs_for(side, quarter) 

86 feed = 2 * eps 

87 # Feed gap on the lower-right side, a short `feed` distance up from the 

88 # bottom corner along the Bc -> Rc direction (unit (0, 1, 1)/sqrt(2)). 

89 u = hd / side # = 1/sqrt(2) 

90 F = (0.0, feed * u, self.base + feed * u) 

91 

92 tups = [] 

93 tups.append((Bc, F, 1, 1 + 0j)) # one-segment driven gap at the corner 

94 tups.append((F, Rc, self.segs_for(side - feed, quarter), None)) 

95 tups.append((Rc, Tc, ns, None)) # upper-right side 

96 tups.append((Tc, Lc, ns, None)) # upper-left side 

97 tups.append((Lc, Bc, ns, None)) # lower-left side 

98 return tups
