1r"""The shipped delta loop, built as a **reflection hybrid**: the drone is a 

2trig-free *point finder*, an ``ry`` mirror gives the left half for free, and 

3``build_path`` stitches the corners. 

4 

5One of four interchangeable expressions of the *same* design. ``delta_loop``, 

6``delta_loop_flyby``, ``delta_loop_reflected`` (here) and ``delta_loop_topdown`` 

7all take the same knobs — ``base`` (top-edge height), ``length_factor`` (total 

8wire length in wavelengths) and ``angle_deg`` (slant tilt from horizontal) — and 

9produce byte-identical wires. They differ only in *how you specify the geometry*. 

10 

11This one leans on three idioms at once, anchored at the **feed** and built up: 

12 

13 - the **drone** flies pen-up (laying no wire) up the right slant to read off 

14 the top corner ``A`` — the one point that would otherwise need trig; the 

15 sin/cos lives in the drone's matrices, never in this script; 

16 - **reflection** across the ``y = 0`` plane (``ry``) mirrors the right half to 

17 the left corner ``B`` and feed terminal ``T`` for free; 

18 - **build_path** (the same helper ``delta_loop`` uses) stitches the four points 

19 into the S->A->B->T perimeter plus the T->S feed. 

20 

21Two passes make it match the shipped design's knobs: ``brentq`` solves the slant 

22length so the total wire hits ``length_factor * wavelength`` (build it with the 

23feed at ``z = 0``, no top height involved yet), then a **z-offset pass** lifts 

24every ``z`` so the top edge seats at ``base``, the feed trailing below. 

25 

26 B-----------A top (A found by the drone, B = ry(A)) 

27 \ / 

28 \ / 

29 \ / 

30 T~~~~S feed gap; S seeds the drone, T = ry(S) 

31""" 

32 

33import math 

34from types import MappingProxyType 

35 

36from antennaknobs import AntennaBuilder, Drone 

37 

38 

39class Builder(AntennaBuilder): 

40 default_params = MappingProxyType( 

41 { 

42 "design_freq": 28.47, 

43 "freq": 28.47, 

44 # Top-edge height -- the same knob as the shipped delta_loop. 

45 "base": 7.0, 

46 # Total wire length in wavelengths (~1 -> full-wave loop). 

47 "length_factor": 1.0800, 

48 # Slant tilt from horizontal. 

49 "angle_deg": 62.3894, 

50 "ui_params": MappingProxyType( 

51 { 

52 "target_z0": 100.0, 

53 "default_view": "yz", # loop lies in the x = 0 plane 

54 "length_factor": {"min": 0.95, "max": 1.15}, 

55 "angle_deg": {"min": 30.0, "max": 80.0}, 

56 } 

57 ), 

58 } 

59 ) 

60 

61 def build_wires(self): 

62 from scipy.optimize import brentq 

63 

64 eps = 0.05 

65 wavelength = 299.792458 / self.design_freq 

66 target = self.length_factor * wavelength 

67 n0 = self.nominal_nsegs 

68 n1 = max(3, self.nominal_nsegs // 7) 

69 

70 def ry(p): 

71 return (p[0], -p[1], p[2]) 

72 

73 def build_path(lst, ns, ex): 

74 return [(a, b, ns, ex) for a, b in zip(lst[:-1], lst[1:])] 

75 

76 def geometry(side): 

77 # Build with the feed at z = 0 (no top height involved yet). Fly the 

78 # right slant pen-up to read off corner A, reflect for the left half. 

79 S = (0.0, eps, 0.0) 

80 A = ( 

81 Drone(position=S) 

82 .face(heading=(0.0, 1.0, 0.0), up=(1.0, 0.0, 0.0)) 

83 .yaw(self.angle_deg) 

84 .forward(side) 

85 .position 

86 ) 

87 B, T = ry(A), ry(S) 

88 return build_path([S, A, B, T], n0, None) + build_path([T, S], n1, 1 + 0j) 

89 

90 def total(side): 

91 return sum(math.dist(p0, p1) for p0, p1, _ns, _ex in geometry(side)) 

92 

93 # Size by total wire length, numerically (no closed-form corner position). 

94 side = brentq(lambda s: total(s) - target, 1e-3, 2.0 * wavelength) 

95 wires = geometry(side) 

96 

97 # z-offset pass: lift the whole loop so the top edge seats at `base`. 

98 top_z = max(p[2] for e in wires for p in (e[0], e[1])) 

99 shift = self.base - top_z 

100 

101 def lift(p): 

102 return (p[0], p[1], p[2] + shift) 

103 

104 return [(lift(a), lift(b), ns, ex) for a, b, ns, ex in wires]
