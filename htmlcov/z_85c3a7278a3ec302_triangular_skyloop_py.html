1r"""Triangular horizontal full-wave loop ("skyloop"), fed at a corner. 

2 

3A single closed loop whose perimeter is about one wavelength, lying FLAT in a 

4plane of constant height (z = base) like the square horizontal_loop skywire -- 

5but bent into an EQUILATERAL TRIANGLE and fed at one CORNER rather than at the 

6midpoint of a side. Strung over real ground it is a strong NVIS radiator (it 

7fires nearly straight up, filling the close-in skip zone); in free space it is 

8horizontally polarised with its main lobe broadside to the loop plane, along 

9+/- z. A full-wave loop presents a moderate, near-resistive feed; a corner feed 

10on a triangle runs a touch higher than the ~100 ohm of a side-fed loop. 

11 

12The wire mesh is SYMMETRIC on all three sides AND across each vertex: every 

13corner is chamfered by the same short one-segment wire, centred ON the vertex 

14and perpendicular to its bisector, so the loop is really an equiangular 

15hexagon alternating long runs with short chamfers (a triangle with its 

16corners clipped). Only the apex chamfer is driven; the other two are plain 

17structural wire, so the three corners are meshed identically and the feed 

18straddles its vertex symmetrically (compare horizontal_loop_drone's diagonal 

19feed across a corner). 

20 

21Default design frequency is 3.8 MHz -- the centre of the 75 m voice band (the 

223.6-4.0 MHz phone segment of 80 m), where a full-size triangular skyloop is a 

23classic high-and-flat NVIS/rag-chew antenna. 

24 

25Methodology purpose: a LARGE single closed loop with the feed placed ACROSS A 

26VERTEX (the driven chamfer bridges the corner, half on each side of it), 

27exercising the engines' closed-loop assembly with a corner-symmetric feed 

28rather than the mid-side gap of horizontal_loop. Authored with the Drone (3D 

29turtle): the hexagon is one continuous pen-down flight -- feed chamfer, then 

30alternating -60 deg yaws between runs and chamfers, and a close. 

31 

32Geometry, in the framework's (x, y, z) convention: 

33 - z : height; the whole triangle sits in the plane z = base (HORIZONTAL) 

34 - x, y : the equilateral triangle, apex at the ORIGIN, hanging toward -y 

35 - feed F straddling the apex corner at (0, 0) (a short chamfer centred on 

36 it, running along x); main lobe along +/- z (toward zenith) 

37 

38 V_top feed F straddles this corner 

39 /\ 

40 / \ 

41 / \ (triangle lies flat at z = base; 

42 / \ viewed from above) 

43 / \ 

44 V_left ---------- V_right 

45""" 

46 

47from types import MappingProxyType 

48 

49from ... import AntennaBuilder, Drone 

50 

51 

52class Builder(AntennaBuilder): 

53 default_params = MappingProxyType( 

54 { 

55 # Centre of the 75 m voice band (3.6-4.0 MHz phone segment). 

56 "design_freq": 3.8, 

57 "freq": 3.8, 

58 # Height of the (flat) loop plane above ground. In free space this 

59 # only shifts the structure; over real ground it sets the NVIS 

60 # takeoff. A full-size 75 m loop typically hangs high in tall trees. 

61 "base": 15.0, 

62 # Perimeter in wavelengths, and the single scale/tuning knob. Each 

63 # side is length_factor/3 of a wavelength, so length_factor = 1 is a 

64 # nominal one-wavelength loop; a full-wave loop resonates a few 

65 # percent long, so the free-space feed is near resonant (X ~ 0, 

66 # ~113 ohm) at length_factor = 1.05. 

67 "length_factor": 1.05, 

68 "ui_params": MappingProxyType( 

69 { 

70 # Full-wave loop, corner-fed -> moderate near-resistive feed. 

71 # Free-space solve at the default 1.05 is ~113 - j5 ohm, so 

72 # 112 shows a near-1:1 SWR there. 

73 "target_z0": 112.0, 

74 # The triangle lies flat in z = base; its x and y spans are 

75 # the two largest, so the xy view shows the loop face-on. 

76 "default_view": "xy", 

77 "length_factor": { 

78 "min": 0.95, 

79 "max": 1.15, 

80 }, 

81 } 

82 ), 

83 } 

84 ) 

85 

86 # Band-locked variant: identical geometry to default, but the frontend 

87 # clamps the freq sweep to the amateur band containing the anchor (80 m, 

88 # which holds the 3.8 MHz default) instead of the wide ±multiplier window. 

89 # Only ui_params is stated — it deep-merges over default_params.ui_params 

90 # (inheriting target_z0 / default_view / length_factor) and flips just 

91 # sweep_policy.band_locked; every regular param overlays from default. 

92 band_locked_params = MappingProxyType( 

93 {"ui_params": MappingProxyType({"sweep_policy": {"band_locked": True}})} 

94 ) 

95 

96 def build_wires(self): 

97 eps = 0.05 

98 

99 wavelength = 299.792458 / self.design_freq 

100 quarter = 0.25 * wavelength 

101 

102 # Equilateral triangle, so each side is a third of the perimeter; the 

103 # perimeter is length_factor wavelengths (length_factor = 1 -> a nominal 

104 # one-wavelength loop). 

105 side = (wavelength / 3.0) * self.length_factor 

106 

107 # Every vertex is chamfered by the same short one-segment wire, centred 

108 # ON the vertex and perpendicular to its bisector; only the apex 

109 # chamfer carries the source. Runs of side - gap between chamfers keep 

110 # the total perimeter at exactly length_factor wavelengths. 

111 gap = 2 * eps 

112 run = side - gap 

113 

114 # Fly the hexagon clockwise (viewed from above) as one pen-down stroke, 

115 # with the feed point at the origin and the triangle hanging toward -y. 

116 # The drone's default pose faces +x with up = +z -- at the apex that is 

117 # exactly the chamfer direction (perpendicular to the bisector), so no 

118 # face() is needed: back up half a gap, lay the driven chamfer 

119 # straddling the vertex, then alternate -60 deg yaws between runs and 

120 # chamfers, staying in the z = base plane throughout. 

121 drone = Drone( 

122 position=(0.0, 0.0, self.base), 

123 nominal_nsegs=self.nominal_nsegs, 

124 ref=quarter, 

125 ) 

126 drone.jump(-gap / 2.0) 

127 

128 drone.feed(1 + 0j).forward(gap, nsegs=1) # driven chamfer across apex 

129 drone.pay_out() 

130 drone.yaw(-60).forward(run) # -> v_right 

131 drone.yaw(-60).forward(gap, nsegs=1) # passive chamfer across v_right 

132 drone.yaw(-60).forward(run) # -> v_left 

133 drone.yaw(-60).forward(gap, nsegs=1) # passive chamfer across v_left 

134 drone.yaw(-60).close() # final run, home to the apex chamfer 

135 

136 return drone.wires()
