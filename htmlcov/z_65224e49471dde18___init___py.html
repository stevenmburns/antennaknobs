1"""Antenna example registry. 

2 

3Each antenna geometry exposed by the web UI is defined in its own module 

4under this package and registered via `register(EXAMPLE)`. The dispatchers 

5in `web.server` and `web.pynec_backend` look up the example by its `name` 

6field, so removing an antenna is a two-line change: delete the module file 

7and the matching `from . import <name>` line below. 

8 

9The registry is intentionally web-layer (not solver-core) — each example 

10parses the request dict, calls shared helpers in `web.server` / 

11`web.pynec_backend`, and produces the JSON-shaped response the frontend 

12consumes. The solver package `momwire/` stays free of UI concerns. 

13""" 

14 

15from __future__ import annotations 

16 

17from ._base import AntennaExample, ParamSpec 

18 

19REGISTRY: dict[str, AntennaExample] = {} 

20 

21 

22def register(example: AntennaExample) -> AntennaExample: 

23 if example.name in REGISTRY: 

24 raise ValueError(f"duplicate antenna example: {example.name}") 

25 REGISTRY[example.name] = example 

26 return example 

27 

28

30# the adapter, which derives a ParamSpec schema from each Builder's 

31# default_params and wraps MomwireEngine + PyNECEngine in the SolveFn / 

32# SweepFn contract above. See web/adapter.py for the bridge and what each 

33# Builder can opt into via its `ui_params` dict. 

34from .. import adapter # noqa: E402 

35 

36adapter.register_all() 

37 

38__all__ = ["AntennaExample", "ParamSpec", "REGISTRY", "register"]
