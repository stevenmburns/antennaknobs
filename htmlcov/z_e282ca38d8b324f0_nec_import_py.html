1"""Import a NEC2 card deck (``.nec``) as antennaknobs wire geometry. 

2 

3The read-side twin of :mod:`antennaknobs.nec_export`: where ``export_nec`` 

4emits a card deck for other NEC tools, ``parse_nec`` consumes one — so a deck 

5written by xnec2c, 4nec2, EZNEC, or found in an antenna handbook can be loaded 

6as a data-driven design (see ``read_nec``). 

7 

8Geometry card semantics (GW/GA/GH/GM/GX/GR/GS) are transcribed from the 

9``nec2c`` 1.3.1 sources (``geometry.c``: ``wire``/``arc``/``helix``/``move``/ 

10``reflc``) so transforms replicate what a NEC engine would build, including 

11the quirks: ``GM`` repetitions compound (each copy transforms the previous 

12copy), ``GX`` doubles the tag increment after every reflection plane, tag 0 

13never increments, and ``GS`` supports xnec2c's tag-range extension. Only the 

14*wire* model is translated; patches (SP/SM) and tapered wires (GC) raise. 

15 

16A NEC deck also carries run configuration that antennaknobs manages itself — 

17ground (GN/GD), loading (LD), networks and transmission lines (TL/NT), sweep 

18and output requests (FR/RP/NE/NH/XQ/...). By default those cards are recorded 

19in ``NecDeck.ignored`` (and FR in ``NecDeck.freq_mhz``) rather than 

20translated, so a caller can tell the user what the deck asked for that the 

21app decides differently. 

22 

23With ``network=True``, the LD/TL/NT cards that antennaknobs' port-network 

24system can express exactly are translated instead of ignored (issue #385): 

25lumped LD loads become ``Load`` branches on named 1-segment wires, LD 5 wire 

26conductivity surfaces as ``NecDeck.conductivity`` (feed it to ``WireSpec``), 

27TL cards become ``TL`` branches (crossed lines, zero-length = port 

28separation, end shunts — a conductance as a ``Shunt``, a reactive G+jB as a 

29fixed 1-port ``Admittance``, issue #423), and an NT card becomes its exact 

30resistive pi when the Y matrix is all-real (``TwoPort`` + ``Shunt``) or a 

312-port ``Admittance`` when it carries susceptance. What cannot be expressed 

32exactly — frequency-independent series reactance (LD 4 with X≠0), distributed 

33RLC (LD 2/3), range-limited conductivity — stays in ``ignored`` with a 

34per-card reason in ``ignored_detail``. ``wire_tuples()`` then emits *named* wires (no legacy 

35``ex`` markers) and ``network()`` returns the matching ``Network``, ready to 

36return from ``build_wires`` / ``build_network``. 

37 

38Excitation: only voltage sources (EX type 0 and 5) can drive an antenna in 

39antennaknobs; plane-wave and current-source excitations raise. The engine 

40feeds a wire tuple at its middle segment, so ``NecDeck.wire_tuples`` splits a 

41wire whose EX segment is off-centre into colinear pieces that preserve the 

42deck's exact segment boundaries and put the feed on its own 1-segment wire. 

43"""

241 def dominant_radius(self) -> float: 

242 """The deck's wire radius, length-weighted where wires differ. 

243 

244 ``build_wires()`` tuples carry no radius — the engines take one 

245 radius for the whole antenna via ``build_wire_material()`` — so a 

246 deck with mixed radii is approximated by the radius that makes up 

247 the greatest total wire length. Feed this to 

248 ``WireSpec(radius=...)`` so the import keeps the deck's reactance 

249 (the engines' 0.5 mm idealization is far off a typical 4-10 mm 

250 Yagi element). 

251 """ 

252 length_by_radius: dict[float, float] = {} 

253 for w in self.wires: 

254 ln = math.dist(w.p1, w.p2) 

255 length_by_radius[w.radius] = length_by_radius.get(w.radius, 0.0) + ln 

256 return max(length_by_radius.items(), key=lambda kv: kv[1])[0]

302 @cached_property 

303 def _port_plan(self) -> dict[tuple[int, int], str]: 

304 """(wire index, 1-based local segment) → port name, for every 

305 segment the network attaches to (network mode). Feeds claim names 

306 first — a single feed is ``"feed"``, matching the catalog 

307 convention — and later attachments to an already-claimed segment 

308 share its port: an LD on the fed segment becomes a ``Load`` and a 

309 ``Driven`` on one port (the Group-2 termination branch), a TL 

310 chain's shared element gets one port per segment however many 

311 lines land there.""" 

312 plan: dict[tuple[int, int], str] = {} 

313 single = len(self.feeds) == 1 

314 for k, f in enumerate(self.feeds, 1): 

315 key = (f.wire, f.seg) 

316 if key in plan: 

317 raise ValueError( 

318 f"NEC deck drives segment {f.seg} of wire {f.wire + 1} " 

319 f"with more than one EX card" 

320 ) 

321 plan[key] = "feed" if single else f"feed{k}" 

322 for k, ld in enumerate(self.loads, 1): 

323 plan.setdefault((ld.wire, ld.seg), f"load{k}") 

324 for k, tl in enumerate(self.tls, 1): 

325 plan.setdefault((tl.wire_a, tl.seg_a), f"tl{k}a") 

326 plan.setdefault((tl.wire_b, tl.seg_b), f"tl{k}b") 

327 for k, nt in enumerate(self.nts, 1): 

328 plan.setdefault((nt.wire_a, nt.seg_a), f"nt{k}a") 

329 plan.setdefault((nt.wire_b, nt.seg_b), f"nt{k}b") 

330 return plan

332 @cached_property 

333 def _junction_cuts(self) -> dict[int, frozenset[int]]: 

334 """wire index → interior segment boundaries (1..n_seg−1) where some 

335 OTHER wire has a segment endpoint. 

336 

337 NEC connects *segments* whose ends coincide — the grouping into GW 

338 wires is irrelevant to it — so a deck may run one long wire straight 

339 through another and rely on the crossing carrying current (the W8IO 

340 whip's matching straps cross the whip axis mid-wire). antennaknobs' 

341 engines junction wires at wire ENDS only, so ``wire_tuples()`` must 

342 shatter wires at these boundaries to reproduce the deck's electrical 

343 graph. The split is lossless: same segments, same boundaries, and 

344 the KCL junction at the shared node is exactly NEC's connection. 

345 """ 

346 eps = 1e-9 

347 

348 def key(p): 

349 return tuple(round(c / eps) for c in p) 

350 

351 def boundary(w, k): 

352 # Must match wire_tuples' point() bitwise so the shattered 

353 # pieces land exactly on the detected nodes. 

354 t = k / w.n_seg 

355 return tuple(a + (b - a) * t for a, b in zip(w.p1, w.p2)) 

356 

357 owners: dict[tuple, set[int]] = {} 

358 for i, w in enumerate(self.wires): 

359 for k in range(w.n_seg + 1): 

360 owners.setdefault(key(boundary(w, k)), set()).add(i) 

361 cuts: dict[int, frozenset[int]] = {} 

362 for i, w in enumerate(self.wires): 

363 shared = { 

364 k for k in range(1, w.n_seg) if len(owners[key(boundary(w, k))]) > 1 

365 } 

366 if shared: 

367 cuts[i] = frozenset(shared) 

368 return cuts

370 def wire_tuples(self, specs: bool = False): 

371 """The deck as ``build_wires()`` tuples. 

372 

373 Default mode: ``(p1, p2, n_seg, ex)`` with the deck's EX voltages as 

374 legacy ``ex`` markers. Network mode (``network=True``): every segment 

375 the network attaches to — feeds, loads, TL/NT connections — becomes a 

376 *named* wire instead, ``(p1, p2, n_seg, None, name)``, and no tuple 

377 carries ``ex`` (the drive comes from ``network()``'s ``Driven`` 

378 sources). 

379 

380 ``specs=True`` (issue #388) emits ``Wire`` named tuples instead, 

381 each carrying a per-wire ``WireSpec`` with the deck wire's OWN 

382 radius — no ``dominant_radius()`` compromise — and its effective 

383 conductivity (a ranged LD 5 over the whole wire, else the deck's 

384 whole-structure LD 5). PyNEC honors both per wire; momwire honors 

385 both too since momwire#147 (complete in momwire 0.13.0 across all 

386 four solver bases, the H-matrix family included). With 

387 ``specs=True`` a ``build_wire_material()`` fallback is unnecessary 

388 (every wire carries its spec) — though a design may still define 

389 one for the weight readout of spec-less wires it adds itself. 

390 

391 Wires are split into colinear pieces on the deck's exact segment 

392 boundaries in two situations: a marked (fed / port) segment that is 

393 not the wire's middle segment gets isolated on its own 1-segment 

394 wire so the delta gap lands exactly where the deck put it, and any 

395 boundary another wire touches is cut so the crossing becomes a 

396 wire-end junction (``_junction_cuts`` — NEC connects segment ends 

397 regardless of wire grouping; the engines junction wire ends only). 

398 Same geometry, same segmentation, same electrical graph as a NEC 

399 run of the original deck. A wire with no cuts whose only mark sits 

400 at the middle segment of an odd count stays whole. 

401 """ 

402 if not self.feeds: 

403 raise ValueError( 

404 "NEC deck has no voltage-source EX card — nothing drives the antenna" 

405 ) 

406 # (wire index) → {segment: (ex voltage | None, port name | None)} 

407 marks: dict[int, dict[int, tuple[complex | None, str | None]]] = {} 

408 if self.network_mode: 

409 for (wi, seg), pname in self._port_plan.items(): 

410 marks.setdefault(wi, {})[seg] = (None, pname) 

411 else: 

412 for f in self.feeds: 

413 per = marks.setdefault(f.wire, {}) 

414 if f.seg in per: 

415 raise ValueError( 

416 f"NEC deck drives segment {f.seg} of wire {f.wire + 1} " 

417 f"with more than one EX card" 

418 ) 

419 per[f.seg] = (f.voltage, None) 

420 

421 sigma_by_wire = dict(self.wire_conductivity) 

422 

423 def spec_for(i, w): 

424 """Per-wire spec (issue #388): the deck wire's own radius, with 

425 its effective conductivity baked in — a ranged LD 5 on this wire 

426 wins over the whole-structure one. Baking is required: engines 

427 treat an explicit spec as complete (no field-level fallback to 

428 build_wire_material), so leaving conductivity None would turn a 

429 copper deck into PEC wire by wire.""" 

430 if not specs: 

431 return None 

432 return _net.WireSpec( 

433 radius=w.radius, 

434 conductivity=sigma_by_wire.get(i, self.conductivity), 

435 ) 

436 

437 tups = [] 

438 

439 def emit(p0, p1, n, ex, pname=None, spec=None): 

440 if spec is not None: 

441 tups.append(_net.Wire(p0, p1, n, ex, pname, spec)) 

442 elif pname: 

443 tups.append((p0, p1, n, ex, pname)) 

444 else: 

445 tups.append((p0, p1, n, ex)) 

446 

447 for i, w in enumerate(self.wires): 

448 if i in self.virtual_anchors: 

449 # Remote TL-anchor wire (issue #427): no geometry — its TL end 

450 # is a PortVirtual in network(), so emit nothing here. 

451 continue 

452 per = marks.get(i, {}) 

453 cutset = self._junction_cuts.get(i, frozenset()) 

454 n = w.n_seg 

455 spec = spec_for(i, w) 

456 if not per and not cutset: 

457 emit(w.p1, w.p2, n, None, spec=spec) 

458 continue 

459 if not cutset and len(per) == 1 and n % 2 == 1: 

460 (seg, (ex, pname)) = next(iter(per.items())) 

461 if seg == (n + 1) // 2: 

462 # Marked at the wire's middle segment — the engine's 

463 # native attachment position; keep the wire whole. 

464 emit(w.p1, w.p2, n, ex, pname, spec) 

465 continue 

466 

467 def point(k, w=w, n=n): 

468 """Endpoint after ``k`` of the wire's ``n`` segments. The same 

469 expression for adjoining pieces yields bitwise-equal points, 

470 which is how wires are recognised as connected.""" 

471 t = k / n 

472 return tuple(a + (b - a) * t for a, b in zip(w.p1, w.p2)) 

473 

474 # Cut at every junction boundary, and around every marked 

475 # segment so it sits alone on a 1-segment piece. 

476 bounds = set(cutset) 

477 for seg in per: 

478 bounds.update((seg - 1, seg)) 

479 bounds -= {0, n} 

480 prev = 0 

481 for b in [*sorted(bounds), n]: 

482 count = b - prev 

483 mark = per.get(b) if count == 1 else None 

484 if mark is not None: 

485 ex, pname = mark 

486 emit(point(prev), point(b), 1, ex, pname, spec) 

487 else: 

488 emit(point(prev), point(b), count, None, spec=spec) 

489 prev = b 

490 return tups

492 def network(self): 

493 """The deck's translated LD/TL/NT cards as a ``network.Network``, 

494 with ports on the named wires ``wire_tuples()`` emits and one 

495 ``Driven`` per EX card — ready to return from ``build_network()``. 

496 A deck with no translatable cards still gets its ``Driven`` feeds, 

497 so a network-mode stub can always define ``build_network``. Only 

498 available when the deck was parsed with ``network=True``.""" 

499 if not self.network_mode: 

500 raise ValueError( 

501 "deck was not parsed for network translation — call " 

502 "parse_nec/read_nec with network=True" 

503 ) 

504 plan = self._port_plan 

505 # A port on a virtualized anchor wire (issue #427) is a pure circuit 

506 # node (PortVirtual), no geometry; every other port is on a real wire. 

507 ports = { 

508 pname: ( 

509 _net.PortVirtual(pname) 

510 if wi in self.virtual_anchors 

511 else _net.PortOnWire(pname) 

512 ) 

513 for (wi, _seg), pname in plan.items() 

514 } 

515 branches: list = [] 

516 for ld in self.loads: 

517 branches.append( 

518 _net.Load( 

519 port=plan[(ld.wire, ld.seg)], 

520 r=ld.r, 

521 l=ld.l, 

522 c=ld.c, 

523 parallel=ld.parallel, 

524 ) 

525 ) 

526 for tl in self.tls: 

527 a = plan[(tl.wire_a, tl.seg_a)] 

528 b = plan[(tl.wire_b, tl.seg_b)] 

529 branches.append( 

530 _net.TL(a=a, b=b, z0=tl.z0, length=tl.length, transposed=tl.transposed) 

531 ) 

532 if tl.shunt_r_a is not None:

548 # Admittance branch (issue #416).

550 continue 

551 if nt.series_r is not None:

553 if nt.shunt_r_a is not None:

558 _net.Driven(port=plan[(f.wire, f.seg)], voltage=f.voltage) 

559 for f in self.feeds 

560 ]

580def _float(token: str, where: str) -> float: 

581 try: 

582 return float(token) 

583 except ValueError: 

584 # Old Fortran decks write D exponents ("1.0D+03"). 

585 try: 

586 return float(token.upper().replace("D", "E")) 

587 except ValueError: 

588 raise ValueError(f"{where}: bad number {token!r}") from None 

589 

590 

591# --------------------------------------------------------------------------- 

592# SY symbolic variables (4nec2 extension, issue #417) 

593# --------------------------------------------------------------------------- 

594# 4nec2's expression language is BASIC-flavored: `^` is power, trig works in 

595# DEGREES (the corpus is full of `sin(360*x)`), `sqr` is square root, `atn` 

596# arctangent (returning degrees), `int` truncates. Names are matched 

597# case-insensitively. Evaluation is a whitelisted ast walk — no eval(). 

598_SY_FUNCS = { 

599 "sin": lambda x: math.sin(math.radians(x)), 

600 "cos": lambda x: math.cos(math.radians(x)), 

601 "tan": lambda x: math.tan(math.radians(x)), 

602 "atn": lambda x: math.degrees(math.atan(x)), 

603 "atan": lambda x: math.degrees(math.atan(x)), 

604 "sqr": math.sqrt, 

605 "sqrt": math.sqrt, 

606 "abs": abs, 

607 "int": lambda x: float(int(x)), 

608 "log": math.log, 

609 "exp": math.exp, 

610} 

611_SY_CONSTANTS = { 

612 "pi": math.pi, 

613 # 4nec2 predefined unit-scale symbols (`SY r = 1.5 * mm`): factors to 

614 # metres. A deck's own SY definition of the same name wins (the symbol 

615 # table is consulted before these constants). 

616 "mm": 1e-3, 

617 "cm": 1e-2, 

618 "dm": 0.1, 

619 "m": 1.0, 

620 "in": 0.0254, 

621 "ft": 0.3048, 

622 # electrical component suffixes (`SY C=36.6pF`, `SY L=0.5uH`) 

623 "pf": 1e-12, 

624 "nf": 1e-9, 

625 "uf": 1e-6, 

626 "nh": 1e-9, 

627 "uh": 1e-6, 

628 "mh": 1e-3, 

629} 

630 

631 

632_SY_UNITS = frozenset( 

633 ("mm", "cm", "dm", "m", "in", "ft", "pf", "nf", "uf", "nh", "uh", "mh") 

634) 

635_SY_MAX_LEN = 512 

636_SY_MAX_TOKENS = 128 

637_SY_MAX_DEPTH = 32 

638# number | identifier | operator/paren. Numbers cover 1, 2.5, 3., .5, 1e-3; 

639# "1..5" tokenizes as two adjacent numbers and fails in the parser. 

640_SY_TOKEN = re.compile( 

641 r"\s*(?:" 

642 r"(\d+\.?\d*(?:[eE][+-]?\d+)?|\.\d+(?:[eE][+-]?\d+)?)" 

643 r"|([A-Za-z_]\w*)" 

644 r"|([()+\-*/^%])" 

645 r")" 

646) 

647 

648# Binding powers (precedence climbing). 4nec2's expression language follows 

649# the BASIC convention: ^ is right-associative and binds tighter than unary 

650# minus (-2^2 = -(2^2) = -4; Excel is the famous outlier). Unary sits 

651# between * and ^ so -2*3 = (-2)*3 but -2^2 = -(2^2). 

652_SY_BP = { 

653 "+": (10, 11), 

654 "-": (10, 11), 

655 "*": (20, 21), 

656 "/": (20, 21), 

657 "%": (20, 21), 

658 "^": (40, 39), 

659} 

660_SY_UNARY_BP = 30

663def _eval_sy_expr(expr: str, syms: dict, where: str) -> float: 

664 """Evaluate one 4nec2 expression with a dedicated precedence-climbing 

665 (Pratt) parser (#424) — the grammar is exactly 4nec2's, not Python's: 

666 no CPython parser in the path, explicit size/depth caps, and a hard 

667 contract: every failure is a ValueError, every success a finite float. 

668 """ 

669 text = expr.strip() 

670 

671 def err(msg: str) -> ValueError: 

672 return ValueError(f"{where}: SY expression {expr!r}: {msg}") 

673 

674 if not text: 

675 raise err("empty") 

676 if len(text) > _SY_MAX_LEN: 

677 raise err(f"longer than {_SY_MAX_LEN} characters") 

678 

679 # ---- tokenize ------------------------------------------------------- 

680 tokens: list[tuple[str, object]] = [] 

681 pos = 0 

682 while pos < len(text): 

683 m = _SY_TOKEN.match(text, pos) 

684 if m is None or m.end() == m.start(): 

685 rest = text[pos:].lstrip() 

686 if not rest: 

687 break 

688 raise err(f"unexpected character {rest[0]!r}") 

689 pos = m.end() 

690 num, ident, op = m.group(1), m.group(2), m.group(3) 

691 if num is not None: 

692 v = float(num) 

693 if not math.isfinite(v): 

694 raise err(f"non-finite number literal {num!r}") 

695 tokens.append(("num", v)) 

696 elif ident is not None: 

697 tokens.append(("ident", ident.lower())) 

698 else: 

699 tokens.append(("op", op)) 

700 if len(tokens) > _SY_MAX_TOKENS: 

701 raise err(f"more than {_SY_MAX_TOKENS} tokens") 

702 if not tokens: 

703 raise err("empty") 

704 

705 # ---- parse + evaluate in one pass ------------------------------------ 

706 idx = 0 

707 

708 def peek(): 

709 return tokens[idx] if idx < len(tokens) else (None, None) 

710 

711 def take(): 

712 nonlocal idx 

713 t = tokens[idx] 

714 idx += 1 

715 return t 

716 

717 def finite(v: float, what: str) -> float: 

718 if not math.isfinite(v): 

719 raise err(f"{what} is not finite") 

720 return v 

721 

722 def lookup(name: str) -> float: 

723 if name in syms: 

724 return syms[name] 

725 if name in _SY_CONSTANTS: 

726 return _SY_CONSTANTS[name] 

727 raise ValueError(f"{where}: undefined symbol {name!r}") 

728 

729 def primary(depth: int) -> float: 

730 kind, val = peek() 

731 if kind == "num": 

732 take() 

733 v = float(val) 

734 # juxtaposed/glued unit product: `135 ft`, `61ft`, `36.6pF` 

735 k2, v2 = peek() 

736 if k2 == "ident" and v2 in _SY_UNITS: 

737 nxt = tokens[idx + 1] if idx + 1 < len(tokens) else (None, None) 

738 if nxt != ("op", "("): 

739 take() 

740 v *= lookup(v2) 

741 return v 

742 if kind == "ident": 

743 take() 

744 k2, _v2 = peek() 

745 if k2 == "op" and tokens[idx][1] == "(": 

746 fn = _SY_FUNCS.get(val) 

747 if fn is None: 

748 raise err(f"unknown function {val!r}") 

749 take() # ( 

750 arg = climb(0, depth + 1) 

751 k3, v3 = peek() 

752 if (k3, v3) != ("op", ")"): 

753 raise err(f"expected ')' after {val}(...)") 

754 take() 

755 try: 

756 return finite(float(fn(arg)), f"{val}(...)") 

757 except (ArithmeticError, ValueError) as e: 

758 if isinstance(e, ValueError) and str(e).startswith(where): 

759 raise 

760 raise err(f"{val}({arg:g}) failed: {e}") from None 

761 return lookup(val) 

762 if kind == "op" and val == "(": 

763 take() 

764 v = climb(0, depth + 1) 

765 k2, v2 = peek() 

766 if (k2, v2) != ("op", ")"): 

767 raise err("unbalanced parenthesis") 

768 take() 

769 return v 

770 if kind == "op" and val in ("-", "+"): 

771 take() 

772 v = climb(_SY_UNARY_BP, depth + 1) 

773 return -v if val == "-" else v 

774 if kind is None: 

775 raise err("ends unexpectedly") 

776 raise err(f"unexpected {val!r}") 

777 

778 def apply(op: str, a: float, b: float) -> float: 

779 try: 

780 if op == "+": 

781 r = a + b 

782 elif op == "-": 

783 r = a - b 

784 elif op == "*": 

785 r = a * b 

786 elif op == "/": 

787 r = a / b 

788 elif op == "%": 

789 r = math.fmod(a, b) 

790 if b == 0.0: 

791 raise ZeroDivisionError("modulo by zero") 

792 else: 

793 r = a**b 

794 except (ArithmeticError, ValueError) as e: 

795 raise err(f"'{a:g} {op} {b:g}' failed: {e}") from None 

796 return finite(r, f"'{a:g} {op} {b:g}'") 

797 

798 def climb(min_bp: int, depth: int) -> float: 

799 if depth > _SY_MAX_DEPTH: 

800 raise err(f"nested deeper than {_SY_MAX_DEPTH}") 

801 lhs = primary(depth) 

802 while True: 

803 kind, val = peek() 

804 if kind != "op" or val not in _SY_BP: 

805 return lhs 

806 lbp, rbp = _SY_BP[val] 

807 if lbp < min_bp: 

808 return lhs 

809 take() 

810 rhs = climb(rbp, depth + 1) 

811 lhs = apply(val, lhs, rhs) 

812 

813 result = climb(0, 0) 

814 if idx != len(tokens): 

815 raise err(f"unexpected {tokens[idx][1]!r} after a complete expression") 

816 return finite(float(result), "result")

819def _define_sy(rest: str, syms: dict, where: str) -> None: 

820 """Apply one SY card: ``name=expr[, name=expr...]['comment]``.""" 

821 body = rest.split("'", 1)[0].strip() # 4nec2 trailing comment 

822 if not body: 

823 raise ValueError(f"{where}: SY card without an assignment") 

824 # Split on top-level commas only (function args never contain commas in 

825 # the 4nec2 single-argument vocabulary, but stay paren-aware anyway). 

826 parts, depth, start = [], 0, 0 

827 for i, ch in enumerate(body): 

828 if ch == "(": 

829 depth += 1 

830 elif ch == ")": 

831 depth -= 1 

832 elif ch == "," and depth == 0: 

833 parts.append(body[start:i]) 

834 start = i + 1 

835 parts.append(body[start:]) 

836 for part in parts: 

837 if "=" not in part: 

838 raise ValueError(f"{where}: SY assignment {part.strip()!r} has no '='") 

839 name, expr = part.split("=", 1) 

840 name = name.strip() 

841 if not name.isidentifier(): 

842 raise ValueError(f"{where}: SY name {name!r} is not a valid symbol") 

843 syms[name.lower()] = _eval_sy_expr(expr, syms, where)

846def _value(token: str, where: str, syms: dict | None) -> float: 

847 """A card field: a plain number, 4nec2's ``#nn`` AWG wire-gauge 

848 shorthand (a radius, in metres), or (when the deck defined SY symbols 

849 or the token contains a letter) a 4nec2 expression.""" 

850 if token.startswith("#"): 

851 # AWG gauge n -> diameter 0.127 mm * 92^((36-n)/39); field is a 

852 # radius. 4nec2 writes `#14`-style GW radius fields (#418). 

853 try: 

854 gauge = int(token[1:]) 

855 except ValueError: 

856 raise ValueError(f"{where}: bad wire gauge {token!r}") from None 

857 return 0.5 * 0.127e-3 * 92.0 ** ((36.0 - gauge) / 39.0) 

858 try: 

859 return _float(token, where) 

860 except ValueError: 

861 if syms is not None and any(c.isalpha() or c in "()*/+-^" for c in token): 

862 return _eval_sy_expr(token, syms, where) 

863 raise

909def _ga(card, wires): 

910 """Wire arc in the XZ plane (nec2c ``arc``): ``n_seg`` 1-segment chords.""" 

911 tag, n_seg = card.i(0), card.i(1) 

912 rada, ang1, ang2, radius = card.f(2), card.f(3), card.f(4), card.f(5)

918 raise card.error("arc angle exceeds 360 degrees")

923 ang += dang 

924 x2, z2 = rada * math.cos(ang), rada * math.sin(ang) 

925 wires.append([tag, 1, [x1, 0.0, z1], [x2, 0.0, z2], radius]) 

926 x1, z1 = x2, z2

929def _gh(card, wires): 

930 """Helix/spiral about +Z (nec2c ``helix``): ``n_seg`` 1-segment chords.""" 

931 tag, n_seg = card.i(0), card.i(1) 

932 s, hl = card.f(2), card.f(3) 

933 a1, b1, a2, b2 = card.f(4), card.f(5), card.f(6), card.f(7) 

934 radius = card.f(8) 

935 if n_seg < 1: 

936 raise card.error(f"segment count must be >= 1, got {n_seg}") 

937 if radius <= 0.0: 

938 raise card.error("wire radius must be > 0") 

939 if s == 0.0 or hl == 0.0: 

940 raise card.error("turn spacing and helix length must be nonzero") 

941 zinc = abs(hl / n_seg) 

942 if a2 == a1 and b1 == 0.0: 

943 b1 = a1 

944 if a2 != a1 and b2 == 0.0: 

945 b2 = a2 

946 

947 def point(z): 

948 if a2 == a1: 

949 a, b = a1, b1 

950 else: 

951 a = a1 + (a2 - a1) * z / abs(hl) 

952 b = b1 + (b2 - b1) * z / abs(hl) 

953 x = a * math.cos(2.0 * math.pi * z / s) 

954 y = b * math.sin(2.0 * math.pi * z / s) 

955 # hl < 0 winds the helix left-handed (nec2c swaps x and y). 

956 return [y, x, z] if hl < 0.0 else [x, y, z] 

957 

958 z1 = 0.0 

959 for _ in range(n_seg): 

960 z2 = z1 + zinc 

961 wires.append([tag, 1, point(z1), point(z2), radius]) 

962 z1 = z2

974def _gm(card, wires): 

975 """Move / replicate (nec2c ``move``): rotate about X then Y then Z (deg), 

976 translate, optionally repeat cumulatively with a tag increment.""" 

977 itgi, nrpt = card.i(0), card.i(1) 

978 rox, roy, roz = card.f(2) * _DEG, card.f(3) * _DEG, card.f(4) * _DEG 

979 xs, ys, zs = card.f(5), card.f(6), card.f(7) 

980 its = int(card.f(8) + 0.5) # nec2c reads ITS as a float and rounds 

981 

982 sps, cps = math.sin(rox), math.cos(rox) 

983 sth, cth = math.sin(roy), math.cos(roy) 

984 sph, cph = math.sin(roz), math.cos(roz) 

985 m = ( 

986 (cph * cth, cph * sth * sps - sph * cps, cph * sth * cps + sph * sps), 

987 (sph * cth, sph * sth * sps + cph * cps, sph * sth * cps - cph * sps), 

988 (-sth, cth * sps, cth * cps), 

989 ) 

990 

991 def xf(p): 

992 return [ 

993 p[0] * m[0][0] + p[1] * m[0][1] + p[2] * m[0][2] + xs, 

994 p[0] * m[1][0] + p[1] * m[1][1] + p[2] * m[1][2] + ys, 

995 p[0] * m[2][0] + p[1] * m[2][1] + p[2] * m[2][2] + zs, 

996 ] 

997 

998 i1 = _first_wire_with_tag(wires, its, card) 

999 if nrpt == 0: 

1000 for w in wires[i1:]: 

1001 w[2], w[3] = xf(w[2]), xf(w[3]) 

1002 return 

1003 block = wires[i1:] 

1004 for _ in range(nrpt): 

1005 # Each repetition transforms the *previous* copy, so rotations and 

1006 # translations compound — that is how one loop side GM-replicates 

1007 # into a square, or one bay into a stack. 

1008 block = [ 

1009 [tag + itgi if tag != 0 else 0, ns, xf(p1), xf(p2), rad] 

1010 for tag, ns, p1, p2, rad in block 

1011 ] 

1012 wires.extend(block)

1017 for tag, ns, p1, p2, rad in wires: 

1018 e1, e2 = p1[axis], p2[axis] 

1019 if abs(e1) + abs(e2) <= 1.0e-5 or e1 * e2 < -1.0e-6:

1063def _gs(card, wires): 

1064 """Scale all dimensions; xnec2c extension: nonzero I1..I2 scales only 

1065 wires whose tag falls in that range.""" 

1066 lo, hi, factor = card.i(0), card.i(1), card.f(2)

1078def _locate_segment(wires, tag, seg, card): 

1079 """Resolve NEC's (tag, segment) addressing to (wire index, 1-based local 

1080 segment): the ``seg``-th segment among wires with that tag, or the 

1081 absolute ``seg``-th segment when ``tag`` is 0.""" 

1082 if seg <= 0: 

1083 seg = 1 

1084 acc = 0 

1085 for i, w in enumerate(wires): 

1086 if tag != 0 and w[0] != tag: 

1087 continue 

1088 if acc + w[1] >= seg: 

1089 return i, seg - acc 

1090 acc += w[1] 

1091 if tag != 0 and acc == 0: 

1092 raise card.error(f"no wire has tag {tag}") 

1093 raise card.error( 

1094 f"segment {seg} is out of range — " 

1095 + (f"tag {tag} has" if tag != 0 else "the deck has") 

1096 + f" only {acc} segments" 

1097 )

1101# to expand it into per-segment Load branches: each expanded segment becomes 

1102# its own named 1-segment wire + MoM port + MNA row, so a wide range (which 

1103# usually means "the whole element" — really distributed loading) would 

1104# shred the mesh for no fidelity gain. 

1105_LD_EXPAND_MAX = 8

1108def _seg_mid(w, seg: int): 

1109 """Midpoint of 1-based local segment ``seg`` of a parse-time wire 

1110 ``[tag, n_seg, p1, p2, radius]`` — NEC's connection point for a 

1111 zero-length TL's straight-line-distance rule.""" 

1112 t = (seg - 0.5) / w[1] 

1113 return [a + (b - a) * t for a, b in zip(w[2], w[3])]

1116def _end_shunt(y: complex, virtual: bool) -> tuple[float | None, complex | None]: 

1117 """Translate one TL end admittance ``y = G + jB`` into a ``(shunt_r, 

1118 shunt_y)`` pair for ``NecTL`` — at most one is non-None: 

1119 

1120 - conductance-only (``B == 0``, real end) → ``shunt_r = 1/G`` (a 

1121 frequency-dependent ``Shunt`` in ``network()``); 

1122 - reactive (``B != 0``, issue #423) or any virtual-node termination 

1123 (remote TL anchor, issue #427) → ``shunt_y = y``, a fixed 1-port 

1124 ``Admittance`` carrying the full complex Y, exact at every frequency; 

1125 - a zero end is an open (both ``None``). 

1126 

1127 A reactive end is no longer dropped: NEC's constant ``B`` is 

1128 frequency-independent, which is exactly what the fixed ``Admittance`` 

1129 primitive (issue #416) expresses. 

1130 """ 

1131 if y == 0: 

1132 return None, None 

1133 if y.imag != 0.0 or virtual: 

1134 return None, y 

1135 return 1.0 / y.real, None

1138def _segment_range(wires, tag, sf, st, card): 

1139 """All (wire index, local segment) pairs an LD card's range covers: 

1140 NEC's tag/range addressing — tag 0 + range 0 is the whole structure, 

1141 tag 0 + a range is absolute segment numbers, a tag with range 0 is 

1142 every segment of that tag, and a tag with a range is local segment 

1143 numbers within the tag.""" 

1144 if tag == 0 and sf == 0: 

1145 return [(i, s) for i, w in enumerate(wires) for s in range(1, w[1] + 1)] 

1146 if sf == 0: 

1147 pairs = [ 

1148 (i, s) 

1149 for i, w in enumerate(wires) 

1150 if w[0] == tag 

1151 for s in range(1, w[1] + 1) 

1152 ] 

1153 if not pairs: 

1154 raise card.error(f"no wire has tag {tag}") 

1155 return pairs 

1156 st = max(st, sf) 

1157 return [_locate_segment(wires, tag, s, card) for s in range(sf, st + 1)]

1160# Clearance, in wavelengths, beyond which a 1-segment TL-terminated wire is 

1161# treated as an electrically irrelevant remote anchor (issue #427). The 

1162# corpus family parks anchors ~100–500 λ away; NEC's own thin-wire coupling is 

1163# long dead by 10 λ, so a wire this far from everything else is there only to 

1164# give a TL card a far-end segment. Kept conservative so nothing intentional 

1165# (a real end-loaded stub a fraction of a wavelength away) is ever swallowed. 

1166_ANCHOR_CLEARANCE_LAMBDA = 10.0 

1167_C_MPS = 299_792_458.0 # speed of light, for wavelength = c / f

1170def _anchor_wires(wires, tls_raw, nts_raw, feeds, lds_raw, freq_mhz): 

1171 """Wire indices that are remote TL-anchor wires (issue #427). 

1172 

1173 A wire qualifies (all must hold) when it is a 1-segment wire referenced 

1174 ONLY as a TL endpoint — not driven (EX), not carrying a lumped/distributed 

1175 LD, not an NT endpoint, not sharing a node with any other wire — and sits 

1176 a clearance of more than ``_ANCHOR_CLEARANCE_LAMBDA`` wavelengths (and far 

1177 more than its own extent) from the rest of the structure. Such a wire is a 

1178 NEC modeling artifact: it exists to terminate a ``TL`` card and is designed 

1179 to be electrically irrelevant, so ``network()`` replaces it with a 

1180 ``PortVirtual`` termination and ``wire_tuples()`` drops it. 

1181 

1182 Needs a frequency to measure clearance in wavelengths: with no FR card 

1183 (``freq_mhz is None``) nothing is virtualized — the safe default is to 

1184 model the deck exactly as written. 

1185 """ 

1186 if freq_mhz is None or len(wires) < 2: 

1187 return set() 

1188 # Largest wavelength in the sweep (lowest frequency) — the most 

1189 # conservative clearance threshold. 

1190 lam = _C_MPS / (min(freq_mhz) * 1e6) 

1191 

1192 def loc(card, a, b): 

1193 return _locate_segment(wires, card.i(a), card.i(b), card)[0] 

1194 

1195 tl_refs: set[int] = set() 

1196 for card in tls_raw: 

1197 tl_refs.add(loc(card, 0, 1)) 

1198 tl_refs.add(loc(card, 2, 3)) 

1199 if not tl_refs: 

1200 return set() 

1201 

1202 # Wires the network otherwise uses electrically — never anchors. 

1203 excluded: set[int] = {f.wire for f in feeds} 

1204 for card in nts_raw: 

1205 excluded.add(loc(card, 0, 1)) 

1206 excluded.add(loc(card, 2, 3)) 

1207 for card in lds_raw: 

1208 if card.i(0) == 5: 

1209 continue # LD 5 is a material conductivity, not an element 

1210 tag, sf, st = card.i(1), card.i(2), card.i(3) 

1211 if tag == 0 and sf == 0: 

1212 continue # whole-structure load — does not single out a wire 

1213 for wi, _ in _segment_range(wires, tag, sf, st, card): 

1214 excluded.add(wi) 

1215 

1216 # Node-coincidence over every wire's segment boundaries: an anchor touches 

1217 # nothing, so any shared node disqualifies it (same key() as wire_tuples). 

1218 eps = 1e-9 

1219 

1220 def key(p): 

1221 return tuple(round(c / eps) for c in p) 

1222 

1223 def boundary(w, k): 

1224 t = k / w[1] 

1225 return tuple(a + (b - a) * t for a, b in zip(w[2], w[3])) 

1226 

1227 owners: dict[tuple, set[int]] = {} 

1228 for i, w in enumerate(wires): 

1229 for k in range(w[1] + 1): 

1230 owners.setdefault(key(boundary(w, k)), set()).add(i) 

1231 

1232 def junctioned(i): 

1233 w = wires[i] 

1234 return any(len(owners[key(boundary(w, k))]) > 1 for k in range(w[1] + 1)) 

1235 

1236 def clearance(i): 

1237 """Nearest endpoint-to-endpoint distance from wire ``i`` to any other 

1238 wire — a lower bound on true separation, ample at ≫10 λ scales.""" 

1239 wi = wires[i] 

1240 pts_i = (wi[2], wi[3]) 

1241 best = math.inf 

1242 for j, wj in enumerate(wires): 

1243 if j == i: 

1244 continue 

1245 for pj in (wj[2], wj[3]): 

1246 for pi in pts_i: 

1247 d = math.dist(pi, pj) 

1248 if d < best: 

1249 best = d 

1250 return best 

1251 

1252 anchors: set[int] = set() 

1253 for i in tl_refs: 

1254 w = wires[i] 

1255 if w[1] != 1 or i in excluded or junctioned(i): 

1256 continue 

1257 extent = math.dist(w[2], w[3]) 

1258 clr = clearance(i) 

1259 if clr > _ANCHOR_CLEARANCE_LAMBDA * lam and clr > 100.0 * extent: 

1260 anchors.add(i) 

1261 return anchors

1264def _translate_network_cards( 

1265 wires, lds_raw, tls_raw, nts_raw, feeds, freq_mhz, virtualize_anchors 

1266): 

1267 """Turn the collected LD/TL/NT cards into NecLoad/NecTL/NecNT records, 

1268 plus (mnemonic, reason) detail for every card instance that stays 

1269 unmodelled. TL/NT resolve first so loads can refuse to co-locate with a 

1270 connection point — our ``Load`` is the port's termination branch, which 

1271 would sit in *parallel* with a TL/NT attached to the same port, not in 

1272 series inside the segment the way NEC composes them.""" 

1273 detail: list[tuple[str, str]] = [] 

1274 skipped: set[str] = set() 

1275 

1276 def skip(mnemonic: str, reason: str) -> None: 

1277 skipped.add(mnemonic) 

1278 if (mnemonic, reason) not in detail: 

1279 detail.append((mnemonic, reason)) 

1280 

1281 anchors = ( 

1282 _anchor_wires(wires, tls_raw, nts_raw, feeds, lds_raw, freq_mhz) 

1283 if virtualize_anchors 

1284 else set() 

1285 ) 

1286 virtualized: set[int] = set() # anchors actually replaced by a virtual end 

1287 

1288 tls: list[NecTL] = [] 

1289 for card in tls_raw: 

1290 wa, sa = _locate_segment(wires, card.i(0), card.i(1), card) 

1291 wb, sb = _locate_segment(wires, card.i(2), card.i(3), card) 

1292 va, vb = wa in anchors, wb in anchors 

1293 # End admittances G+jB: a conductance-only end becomes a Shunt(1/G), 

1294 # a reactive one (#423) or a virtual-node termination (#427) a fixed 

1295 # 1-port Admittance. See _end_shunt — susceptance is expressible now, 

1296 # so a reactive end no longer sinks the whole TL. 

1297 shunt_r_a, shunt_y_a = _end_shunt(complex(card.f(6), card.f(7)), va) 

1298 shunt_r_b, shunt_y_b = _end_shunt(complex(card.f(8), card.f(9)), vb) 

1299 z0 = card.f(4) 

1300 if z0 == 0.0: 

1301 raise card.error("characteristic impedance must be nonzero") 

1302 length = card.f(5) 

1303 if length == 0.0: 

1304 # NEC: zero length means the straight-line distance between 

1305 # the connection points. 

1306 length = math.dist(_seg_mid(wires[wa], sa), _seg_mid(wires[wb], sb)) 

1307 tls.append( 

1308 NecTL( 

1309 wire_a=wa, 

1310 seg_a=sa, 

1311 wire_b=wb, 

1312 seg_b=sb, 

1313 z0=abs(z0), 

1314 length=length, 

1315 transposed=z0 < 0.0, 

1316 shunt_r_a=shunt_r_a, 

1317 shunt_r_b=shunt_r_b, 

1318 virtual_a=va, 

1319 virtual_b=vb, 

1320 shunt_y_a=shunt_y_a, 

1321 shunt_y_b=shunt_y_b, 

1322 ) 

1323 ) 

1324 if va: 

1325 virtualized.add(wa) 

1326 if vb: 

1327 virtualized.add(wb) 

1328 

1329 nts: list[NecNT] = [] 

1330 for card in nts_raw: 

1331 wa, sa = _locate_segment(wires, card.i(0), card.i(1), card) 

1332 wb, sb = _locate_segment(wires, card.i(2), card.i(3), card) 

1333 # NEC's NT card is reciprocal: it gives Y11, Y12, Y22 (real+imag each) 

1334 # and Y21 = Y12. 

1335 y11 = complex(card.f(4), card.f(5)) 

1336 y12 = complex(card.f(6), card.f(7)) 

1337 y22 = complex(card.f(8), card.f(9)) 

1338 if y11 == y12 == y22 == 0.0: 

1339 skip("NT", "all-zero Y-parameters") 

1340 continue 

1341 if card.f(5) or card.f(7) or card.f(9): 

1342 # Susceptance anywhere: no resistive pi. Keep the full complex Y — 

1343 # network() emits it as a general Admittance branch (issue #416). 

1344 nts.append( 

1345 NecNT( 

1346 wire_a=wa, 

1347 seg_a=sa, 

1348 wire_b=wb, 

1349 seg_b=sb, 

1350 y=((y11, y12), (y12, y22)), 

1351 ) 

1352 ) 

1353 continue 

1354 # All-real Y → exact resistive pi: series −Y12 between the ports, 

1355 # shunts Y11+Y12 / Y22+Y12 at each. Real Y is frequency-independent, 

1356 # so this holds at every frequency. 

1357 ys, ya, yb = -y12.real, y11.real + y12.real, y22.real + y12.real 

1358 nts.append( 

1359 NecNT( 

1360 wire_a=wa, 

1361 seg_a=sa, 

1362 wire_b=wb, 

1363 seg_b=sb, 

1364 series_r=1.0 / ys if ys else None, 

1365 shunt_r_a=1.0 / ya if ya else None, 

1366 shunt_r_b=1.0 / yb if yb else None, 

1367 ) 

1368 ) 

1369 

1370 connected = {(t.wire_a, t.seg_a) for t in tls} | {(t.wire_b, t.seg_b) for t in tls} 

1371 connected |= {(t.wire_a, t.seg_a) for t in nts} | {(t.wire_b, t.seg_b) for t in nts} 

1372 

1373 loads: list[NecLoad] = [] 

1374 loaded: set[tuple[int, int]] = set() 

1375 conductivity: float | None = None 

1376 wire_conductivity: dict[int, float] = {} 

1377 for card in lds_raw: 

1378 ldtyp = card.i(0) 

1379 tag, sf, st = card.i(1), card.i(2), card.i(3) 

1380 if ldtyp in (0, 1, 4): 

1381 if ldtyp == 4 and card.f(5) != 0.0: 

1382 skip( 

1383 "LD", 

1384 "type 4 fixed reactance is frequency-independent — " 

1385 "not representable with R/L/C", 

1386 ) 

1387 continue 

1388 pairs = _segment_range(wires, tag, sf, st, card) 

1389 if len(pairs) > _LD_EXPAND_MAX: 

1390 skip( 

1391 "LD", 

1392 f"range spans {len(pairs)} segments — the importer " 

1393 f"expands at most {_LD_EXPAND_MAX} into per-segment loads", 

1394 ) 

1395 continue 

1396 r = card.f(4) or None 

1397 le = None if ldtyp == 4 else (card.f(5) or None) 

1398 c = None if ldtyp == 4 else (card.f(6) or None) 

1399 if r is None and le is None and c is None: 

1400 continue # zero-valued load — a no-op 

1401 for pair in pairs: 

1402 if pair in connected: 

1403 skip( 

1404 "LD", 

1405 "load on a segment with a TL/NT connection — the " 

1406 "series-inside-the-segment composition is not modelled", 

1407 ) 

1408 continue 

1409 if pair in loaded: 

1410 skip("LD", "a second load on one segment is not merged") 

1411 continue 

1412 loaded.add(pair) 

1413 loads.append(NecLoad(pair[0], pair[1], r, le, c, ldtyp == 1)) 

1414 elif ldtyp in (2, 3): 

1415 skip("LD", f"type {ldtyp} distributed per-metre loading is not translated") 

1416 elif ldtyp == 5: 

1417 if tag == 0 and sf == 0: 

1418 conductivity = card.f(4) 

1419 else: 

1420 # Ranged conductivity (issue #388): whole wires can carry 

1421 # their own WireSpec conductivity, so a range that covers 

1422 # each touched wire in full translates per wire (NEC's 

1423 # last-card-wins per segment becomes last-wins per wire). 

1424 pairs = _segment_range(wires, tag, sf, st, card) 

1425 by_wire: dict[int, set[int]] = {} 

1426 for wi, s in pairs: 

1427 by_wire.setdefault(wi, set()).add(s) 

1428 if all( 

1429 segs == set(range(1, wires[wi][1] + 1)) 

1430 for wi, segs in by_wire.items() 

1431 ): 

1432 for wi in by_wire: 

1433 wire_conductivity[wi] = card.f(4) 

1434 else: 

1435 skip( 

1436 "LD", 

1437 "type 5 conductivity on a partial-wire segment " 

1438 "range — per-wire specs cover whole wires only", 

1439 ) 

1440 else: 

1441 skip("LD", f"type {ldtyp} is not recognised") 

1442 

1443 return ( 

1444 tuple(loads), 

1445 tuple(tls), 

1446 tuple(nts), 

1447 conductivity, 

1448 tuple(sorted(wire_conductivity.items())), 

1449 detail, 

1450 skipped, 

1451 frozenset(virtualized), 

1452 )

1455def parse_nec( 

1456 text: str, 

1457 *, 

1458 name: str = "NEC deck", 

1459 network: bool = False, 

1460 virtualize_anchors: bool = True, 

1461) -> NecDeck: 

1462 """Parse the text of a NEC2 card deck into a :class:`NecDeck`. 

1463 

1464 ``network=True`` additionally translates the deck's expressible LD/TL/NT 

1465 cards into ``deck.network()`` branches instead of recording them in 

1466 ``ignored`` — see the module docstring for exactly what translates. 

1467 

1468 ``virtualize_anchors`` (network mode only, default on) replaces a remote 

1469 1-segment TL-anchor wire with a ``PortVirtual`` termination (issue #427): 

1470 a wire ≫10 λ from everything else, referenced only to give a TL card a 

1471 far-end segment, is dropped from ``wire_tuples()`` and its TL end becomes 

1472 a pure circuit node — an ideal open, or a 1-port ``Admittance`` for a 

1473 shorted-stub far-Y. Set it ``False`` to model such wires as real geometry. 

1474 

1475 Raises ``ValueError`` (with ``name`` and the line number) on cards that 

1476 are malformed or describe things antennaknobs cannot model — patches, 

1477 tapered wires, plane-wave or current-source excitation. 

1478 """

1480 comments: list[str] = [] 

1481 ignored: set[str] = set() 

1482 feeds_raw: list[tuple[int, int, complex, str]] = [] 

1483 lds_raw: list[_Card] = [] 

1484 tls_raw: list[_Card] = [] 

1485 nts_raw: list[_Card] = [] 

1486 freq_mhz: tuple[float, float] | None = None 

1487 ground = False 

1488 extended_kernel = False 

1489 syms: dict[str, float] = {} # SY symbol table (#417) 

1490 

1491 geometry = { 

1492 "GW": _gw, 

1493 "GA": _ga, 

1494 "GH": _gh, 

1495 "GM": _gm, 

1496 "GX": _gx, 

1497 "GR": _gr, 

1498 "GS": _gs, 

1499 } 

1500 

1501 for line_no, raw in enumerate(text.splitlines(), 1): 

1502 stripped = raw.strip() 

1503 if not stripped: 

1504 continue 

1505 where = f"{name}, line {line_no}" 

1506 # 4nec2 comment convention (#418): a leading ' comments out the 

1507 # whole line (including commented-out cards, `'GW ...`); anywhere 

1508 # else ' starts an end-of-line comment. CM/CE lines are exempt — 

1509 # their free text legitimately contains apostrophes. 

1510 if stripped.startswith("'"): 

1511 continue 

1512 if stripped[:2].upper() == "CM": 

1513 # NEC identifies cards by the first two columns, so a glued 

1514 # "CMtext..." is a comment too (wild decks write "cmRP ..." to 

1515 # comment out cards). Tolerated after CE as well (#418). 

1516 comments.append(stripped[2:].strip()) 

1517 continue 

1518 if stripped[:2].upper() != "CE": 

1519 stripped = stripped.split("'", 1)[0].rstrip() 

1520 if not stripped: 

1521 continue 

1522 # Cards are free-format in practice: mnemonic, then numbers separated 

1523 # by spaces and/or commas. 

1524 tokens = stripped.replace(",", " ").split() 

1525 # Fused mnemonics (#418): ARRL-era decks glue the mnemonic to the 

1526 # first field ("GW1,8,...", "GE1", "EX5,1,..."). Split when two 

1527 # alphabetic characters run straight into a number. 

1528 if ( 

1529 len(tokens[0]) > 2 

1530 and tokens[0][:2].isalpha() 

1531 and tokens[0][2] in "0123456789.+-" 

1532 ): 

1533 tokens = [tokens[0][:2], tokens[0][2:], *tokens[1:]] 

1534 mnemonic = tokens[0].upper() 

1535 if len(mnemonic) != 2 or not mnemonic.isalpha(): 

1536 raise ValueError( 

1537 f"{where}: expected a NEC card mnemonic, got {tokens[0]!r}" 

1538 ) 

1539 

1540 if mnemonic == "CE": 

1541 continue 

1542 

1543 if mnemonic == "EN": 

1544 break 

1545 if mnemonic == "SY": 

1546 # 4nec2 symbolic variables (#417): bind name=expr (possibly 

1547 # several per card, possibly with a trailing ' comment) into the 

1548 # symbol table consulted by every later card field. 

1549 _define_sy(stripped[2:], syms, where) 

1550 continue 

1551 if mnemonic in _UNSUPPORTED_CARDS: 

1552 raise ValueError( 

1553 f"{where}: this deck uses {_UNSUPPORTED_CARDS[mnemonic]}, " 

1554 f"which antennaknobs cannot model" 

1555 ) 

1556 if mnemonic == "GN": 

1557 ground = True 

1558 ignored.add(mnemonic) 

1559 continue

1562 # list is final (their tag/segment addressing resolves against 

1563 # the transformed geometry, exactly like EX). 

1564 card = _Card(mnemonic, tokens[1:], where, syms) 

1565 if mnemonic == "LD": 

1566 if card.i(0) == -1: 

1567 lds_raw.clear() # NEC: nullify all previous loads 

1568 else: 

1569 lds_raw.append(card) 

1570 elif mnemonic == "TL": 

1571 tls_raw.append(card) 

1572 else: 

1573 if card.i(0) == -1: 

1574 # NEC: an NT with I1 = -1 cancels all previous 

1575 # network AND transmission-line data. 

1576 nts_raw.clear() 

1577 tls_raw.clear() 

1578 else: 

1579 nts_raw.append(card) 

1580 continue 

1581 if mnemonic == "EK": 

1582 # Extended thin-wire kernel (#414): honored (via PyNEC), not 

1583 # ignored. `EK -1` switches back to the standard kernel; any 

1584 # other form turns it on. NEC scopes EK to subsequent geometry; 

1585 # decks in the wild use it globally, so one deck-level flag. 

1586 card = _Card(mnemonic, tokens[1:], where, syms) 

1587 extended_kernel = card.i(0) != -1 

1588 continue 

1589 if mnemonic in _IGNORED_CARDS: 

1590 ignored.add(mnemonic) 

1591 continue 

1592 

1593 card = _Card(mnemonic, tokens[1:], where, syms) 

1594 

1595 if mnemonic in geometry: 

1596 geometry[mnemonic](card, wires) 

1597 elif mnemonic == "GE": 

1598 ground = ground or card.i(0) != 0 

1599 elif mnemonic == "FR":

1601 nfrq = max(card.i(1), 1) 

1602 start, step = card.f(4), card.f(5) 

1603 if card.i(0) == 0: # linear sweep 

1604 end = start + (nfrq - 1) * step 

1605 else: # multiplicative sweep 

1606 end = start * step ** (nfrq - 1) if step > 0 else start
