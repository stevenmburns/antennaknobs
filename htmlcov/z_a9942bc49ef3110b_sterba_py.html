1"""Sterba curtain: a broadside, bidirectional, horizontally-polarised curtain 

2array (E. J. Sterba, Bell Labs, 1930s). 

3 

4Topology follows L. B. Cebik's (W4RNL) "models with remarkable ease" recipe 

5for the Sterba in NEC. The curtain is a single continuous closed wire that 

6meanders between a bottom rail (z = base) and a top rail (z = base + h), 

7where h = half a wavelength. Each horizontal half-wave run radiates; the 

8vertical half-wave runs phase-reverse the standing wave so every horizontal 

9section ends up in phase -> broadside gain off both faces. 

10 

11The "twisted vertical line pairs" of the textbook drawing are NOT modelled as 

12transmission lines. Cebik's trick: run two conductors offset by a tiny space 

13(`spacing`) in x, one out-and-back, so the verticals sit as ordinary wires 

14correctly oriented in both planes and no half-twist or TL card is needed. 

15At 7.2 MHz Cebik used 0.5 ft of offset for a 2.5 wl curtain "without the 

16slightest distortion in antenna performance"; we scale a comparable offset. 

17 

18Geometry, in the framework's (x, y, z) convention: 

19 - y : the long axis of the curtain (Cebik's x) 

20 - z : height; bottom rail at `base`, top rail at `base + h` 

21 - x : the small twisted-pair offset (Cebik's y); curtain fires +/- x 

22 

23The curtain has `n_cells` full half-wave sections in the middle plus a 

24quarter-wave section at each end, so the total length is 

25(n_cells + 1) * (lambda/2 * length_factor). n_cells must be odd so the 

26structure is symmetric and a bottom-rail half-wave section sits dead centre. 

27 

28Feedpoint: the centre of the central lower (bottom-rail) half-wave section -- 

29the classic "feed the middle of a lower 1/2-wl dipole section" point. The 

30input impedance is high (~600 ohm, reactive); in practice you feed it with 

31open-wire line to a tuner. With n_cells = 3 (~2 wl) this model gives, at 

3228.47 MHz, Z ~ 600 ohm, ~10.5 dBi bidirectional broadside in free space, and 

33~16 dBi at a ~14-deg elevation over average ground with the bottom rail at 

347 m. length_factor near 1.0 maximises gain (and resonates near ~0.99); 

35gain falls off quickly below ~0.96 as the half-wave phasing breaks down. 

36""" 

37 

38from antennaknobs import AntennaBuilder 

39import math 

40from types import MappingProxyType 

41 

42 

43class Builder(AntennaBuilder): 

44 default_params = MappingProxyType( 

45 { 

46 "design_freq": 28.47, 

47 "freq": 28.47, 

48 "base": 7.0, 

49 # Scales the half-wave unit (both horizontal section length and 

50 # vertical rail spacing). Tuned by the optimiser for resonance; 

51 # Cebik's table sits near ~0.98. 

52 "length_factor": 1.0, 

53 # Number of full half-wave sections in the middle of the curtain. 

54 # Must be odd. 3 -> ~2 lambda (Cebik's example); 5 -> ~3 lambda. 

55 "n_cells": 3, 

56 # Twisted-pair offset in x, in metres. Small vs wavelength; just 

57 # large enough to keep the two conductors from merging. 

58 "spacing": 0.04, 

59 "ui_params": MappingProxyType( 

60 { 

61 # The curtain's driving-point impedance is ~600 ohm; it's 

62 # fed with open-wire line to a tuner, so the SWR readout is 

63 # referenced to 600 ohm rather than the auto 50 ohm default. 

64 "target_z0": 600.0, 

65 # Narrowband, 10m-specific antenna: snap the GUI freq 

66 # sweep to the band containing design_freq (28.0-29.7 MHz) 

67 # instead of the default broad +/-25% out-of-band view. 

68 "sweep_policy": {"band_locked": True}, 

69 # Below ~0.96 the half-wave phasing breaks down (gain 

70 # collapses) and the input reactance swings through a 

71 # sharp anti-resonance; keep the slider/optimiser in the 

72 # window where the curtain actually behaves as a Sterba. 

73 "length_factor": { 

74 "min": 0.96, 

75 "max": 1.05, 

76 }, 

77 "n_cells": { 

78 "min": 1, 

79 "max": 7, 

80 "step": 2, 

81 }, 

82 } 

83 ), 

84 } 

85 ) 

86 

87 def build_wires(self): 

88 wavelength = 299.792458 / self.design_freq 

89 

90 h = 0.5 * wavelength * self.length_factor # half-wave unit 

91 q = 0.5 * h # quarter-wave end section 

92 bot = self.base 

93 top = self.base + h 

94 dx = self.spacing 

95 n = int(self.n_cells) 

96 

97 assert n >= 1 and n % 2 == 1, "n_cells must be a positive odd integer" 

98 

99 # y breakpoints: quarter-wave, then n half-waves, then quarter-wave. 

100 yb = [0.0, q] + [q + k * h for k in range(1, n + 1)] + [2 * q + n * h] 

101 Ltot = yb[-1] 

102 ymid = 0.5 * Ltot 

103 

104 # Conductor A (x = 0): horizontal section i is on the top rail when i 

105 # is even, bottom rail when odd. Starts top-left, and -- because there 

106 # are n + 2 sections and n is odd -> odd count -> ends top-right. 

107 def lvl_a(i): 

108 return top if i % 2 == 0 else bot 

109 

110 A = [(0.0, yb[0], lvl_a(0))] 

111 for i in range(n + 2): 

112 A.append((0.0, yb[i + 1], lvl_a(i))) # horizontal run 

113 if i < n + 1: 

114 A.append((0.0, yb[i + 1], lvl_a(i + 1))) # vertical riser 

115 A.append((0.0, Ltot, bot)) # right-end vertical down to bottom rail 

116 

117 # Conductor B (x = dx): the return pass, right -> left, levels flipped 

118 # so it interleaves with A. Its first quarter-wave run starts from A's 

119 # bottom-right corner (so x ramps 0 -> dx within that run, exactly as 

120 # Cebik's wire table does -- no degenerate tiny jog wire). 

121 ybr = list(reversed(yb)) 

122 

123 def lvl_b(j): 

124 return bot if j % 2 == 0 else top 

125 

126 B = [(dx, ybr[1], lvl_b(0))] 

127 B.append((dx, ybr[1], lvl_b(1))) 

128 for j in range(1, n + 2): 

129 B.append((dx, ybr[j + 1], lvl_b(j))) 

130 if j < n + 1: 

131 B.append((dx, ybr[j + 1], lvl_b(j + 1))) 

132 # B ends at (dx, 0, bot); the loop closes B[-1] -> A[0] (left-end 

133 # near-vertical riser, with the small x offset folded in). 

134 

135 loop = A + B 

136 N = len(loop) 

137 

138 tups = [] 

139 feed_count = 0 

140 for k in range(N): 

141 p0 = loop[k] 

142 p1 = loop[(k + 1) % N] 

143 seg_len = math.dist(p0, p1) 

144 nseg = self.segs_for(seg_len, h) 

145 

146 is_horizontal = abs(p0[2] - p1[2]) < 1e-9 

147 at_bottom = abs(p0[2] - bot) < 1e-9 and abs(p1[2] - bot) < 1e-9 

148 midy = 0.5 * (p0[1] + p1[1]) 

149 # Feed = centre of the central lower half-wave section. 

150 is_feed = ( 

151 is_horizontal 

152 and at_bottom 

153 and seg_len > 0.6 * h 

154 and abs(midy - ymid) < 0.01 * h 

155 ) 

156 

157 ev = None 

158 if is_feed: 

159 feed_count += 1 

160 if nseg % 2 == 0: 

161 nseg += 1 # odd -> a segment sits exactly at centre 

162 ev = 1 + 0j 

163 

164 tups.append((p0, p1, nseg, ev)) 

165 

166 assert feed_count == 1, f"expected exactly one feed edge, got {feed_count}" 

167 

168 return tups
